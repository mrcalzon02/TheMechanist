    ArrayList<Point> linePoints(int x0, int y0, int x1, int y1) {
        ArrayList<Point> pts = new ArrayList<>();
        int dx = Math.abs(x1 - x0), dy = Math.abs(y1 - y0);
        int sx = x0 < x1 ? 1 : -1;
        int sy = y0 < y1 ? 1 : -1;
        int err = dx - dy;
        int x = x0, y = y0;
        while (true) {
            pts.add(new Point(x,y));
            if (x == x1 && y == y1) break;
            int e2 = 2 * err;
            if (e2 > -dy) { err -= dy; x += sx; }
            if (e2 < dx) { err += dx; y += sy; }
            if (pts.size() > 256) break;
        }
        return pts;
    }

    Map<String,String> decodeEditorGrid(String raw) {
        LinkedHashMap<String,String> out = new LinkedHashMap<>();
        if (raw == null || raw.isBlank()) return out;
        for (String part : raw.split(";")) {
            int eq = part.indexOf('=');
            if (eq <= 0) continue;
            String key = part.substring(0, eq).trim();
            String val = part.substring(eq+1).trim();
            if (key.matches("\\d+,\\d+") && !val.isBlank()) out.put(key, val);
        }
        return out;
    }
```

    String encodeEditorGrid(Map<String,String> cells) {
        if (cells == null || cells.isEmpty()) return "";
        ArrayList<String> keys = new ArrayList<>(cells.keySet());
        keys.sort((a,b) -> {
            String[] aa=a.split(","); String[] bb=b.split(",");
            int ay=Integer.parseInt(aa[1]), by=Integer.parseInt(bb[1]);
            if (ay != by) return Integer.compare(ay, by);
            return Integer.compare(Integer.parseInt(aa[0]), Integer.parseInt(bb[0]));
        });
        StringBuilder sb = new StringBuilder();
        for (String k : keys) {
            if (sb.length() > 0) sb.append(';');
            sb.append(k).append('=').append(cells.get(k).replace(';', ',').replace('=', ':'));
        }
        return sb.toString();
    }
```
