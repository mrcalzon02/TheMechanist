    void startGamepadInputEngine(boolean enabled) {
        if (!enabled || gamepadInputThread != null) return;
        gamepadInputEngine = new GamepadInputEngine(inputRegistry);
        gamepadInputThread = new Thread(gamepadInputEngine, "mechanist-gamepad-input");
        gamepadInputThread.setDaemon(true);
        gamepadInputThread.start();
    }

    void beginWindowModeReconfigure() {
        windowModeReconfiguring = true;
        try { DebugLog.audit("WINDOW_RECONFIGURE_BEGIN", "suppressRuntimeShutdown=true timerRunning=" + timer.isRunning()); } catch (Throwable ignored) {}
    }

    void endWindowModeReconfigure() {
        windowModeReconfiguring = false;
        if (!timer.isRunning()) {
            timer.start();
            try { DebugLog.warn("WINDOW_RECONFIGURE_TIMER_RESTART", "Swing timer was not running after display reconfiguration; restarted for UI/loading progress."); } catch (Throwable ignored) {}
        }
        try { DebugLog.audit("WINDOW_RECONFIGURE_END", "suppressRuntimeShutdown=false timerRunning=" + timer.isRunning()); } catch (Throwable ignored) {}
    }

    void ensureRuntimeTimerRunning(String reason) {
        if (!timer.isRunning()) {
            timer.start();
            try { DebugLog.warn("RUNTIME_TIMER_RESTART", "Swing timer was stopped; restarted for " + reason + "."); } catch (Throwable ignored) {}
        }
    }

    void shutdownRuntime() {
        if (windowModeReconfiguring) {
            try { DebugLog.audit("RUNTIME_SHUTDOWN_SUPPRESSED", "reason=window-mode-reconfigure timerRunning=" + timer.isRunning()); } catch (Throwable ignored) {}
            return;
        }
        try { timer.stop(); } catch (Throwable ignored) {}
        try { if (gamepadInputEngine != null) gamepadInputEngine.stop(); } catch (Throwable ignored) {}
        try { inputRegistry.clearSource(InputSource.GAMEPAD); } catch (Throwable ignored) {}
        try { if (chatRuntime != null) chatRuntime.shutdown(); } catch (Throwable ignored) {}
        try { if (renderStressTest != null && renderStressTest.active()) DebugLog.metric("render-stress", renderStressTest.toggle(frameLimiter)); } catch (Throwable t) { DebugLog.error("RUNTIME_SHUTDOWN_STRESS", "Could not stop render stress test cleanly.", t); }
        try { if (multiplayerMenu != null) multiplayerMenu.close(); } catch (Throwable t) { DebugLog.error("RUNTIME_SHUTDOWN_MULTIPLAYER", "Could not close multiplayer host/menu controller.", t); }
        try { singlePlayerSectorBridge.close(); } catch (Throwable t) { DebugLog.error("RUNTIME_SHUTDOWN_SECTOR", "Could not close single-player sector runtime bridge.", t); }
        try { DebugLog.audit("RUNTIME_SHUTDOWN", "timerStopped=true gamepadStopRequested=" + (gamepadInputEngine != null) + " sectorRuntimeClosed=true"); } catch (Throwable ignored) {}
    }

    void updateDynamicMusic(String reason) {
        if (sounds == null) return;
        String key = desiredMusicContext();
        sounds.requestMusic(key, options);
        if (!key.equals(lastMusicAuditKey) || (reason != null && reason.toLowerCase(Locale.ROOT).contains("combat"))) {
            lastMusicAuditKey = key;
            DebugLog.audit("MUSIC_ROUTE", "reason=" + reason + " key=" + key + " screen=" + screen + (world == null ? "" : " zone=" + world.zoneType.label) + " combatUntil=" + combatMusicUntilTurn + " governorUntil=" + governorBattleMusicUntilTurn + " turn=" + turn);
        }
    }

    String desiredMusicContext() {
        if (options == null || !options.musicEnabled || options.musicVolume <= 0) return DynamicMusicManager.SILENCE;
        if (screen == Screen.BOOT) return DynamicMusicManager.SILENCE;
        if (screen == Screen.MENU || screen == Screen.OPTIONS || screen == Screen.LOST) return "MAIN_MENU";
        if (screen == Screen.INTRO_CRAWL) return DynamicMusicManager.SILENCE;
        if (screen == Screen.CHARACTER || screen == Screen.KNOWLEDGE || screen == Screen.WORLD_SELECT || screen == Screen.WORLD_SETUP || (screen == Screen.LOADING && loadingForNewGame)) return "CHARACTER_SELECT";
        if (screen == Screen.GAME || screen == Screen.PANEL || screen == Screen.PAUSE || screen == Screen.ZONE_SPLASH || screen == Screen.LOADING) {
            if (governorBattleMusicUntilTurn > turn || hostileGovernorPressureNearby()) return "ESTATE_BATTLE";
            if (combatMusicUntilTurn > turn || hostilePressureNearby()) return "COMBAT";
            return zoneAmbientMusicContext();
        }
        return "LOW_HABITATION";
    }

    String zoneAmbientMusicContext() {
        if (world == null) return "LOW_HABITATION";
        RoomProfile rp = world.roomProfile(world.roomIdAt(playerX, playerY));
        String roomText = ((rp == null ? "" : rp.name) + " " + (rp == null ? "" : rp.descriptor) + " " + (rp == null ? "" : rp.featureText)).toLowerCase(Locale.ROOT);
        if (roomText.contains("clinic") || roomText.contains("triage") || roomText.contains("infirmary")) return "TRIAGE";
        if (roomText.contains("rite") || roomText.contains("chapel") || roomText.contains("choir")) return "CIVIC_RECORDS";
        switch (world.zoneType) {
            case GANGER_TURF: return "GANG_TURF";
            case ARBITES_PRECINCT_EDGE: return "PUBLIC_ORDER";
            case ADMINISTRATUM_ARCHIVE: return "CIVIC_RECORDS";
            case IMPERIAL_GUARD_BILLET: return "CHARTER_GUARD";
            case MECHANICUS_FORGE_CLOISTER:
            case MECHANICUS_RELIC_DUCT: return "ENGINE_HOUSE";
            case CULTIST_SEWER_CAMP: return "PROSCRIBED_CELL";
            case MUTANT_SEWER_CAMP:
            case MUTANT_WARRENS: return "OUTCAST";
            case TRAIN_SERVICE_YARD:
            case NEUTRAL_RAIL_DEPOT: return "TRAIN";
            case SECTOR_GOVERNORS_MANSION: return "ESTATE_PALACE";
            case NOBLE_SERVICE_SPINE: return "ESTATE";
            case NEUTRAL_CIVILIAN_FLOOR:
            case HAB_STACK:
            case SUMP_MARKET:
            case TRASH_WARREN:
            case SEWER_CONDUIT:
            default: return "LOW_HABITATION";
        }
    }

    boolean hostilePressureNearby() {
        if (world == null || world.npcs == null || active == null) return false;
        for (NpcEntity npc : world.npcs) {
            if (!isHostileNpc(npc)) continue;
            int d = Math.abs(npc.x - playerX) + Math.abs(npc.y - playerY);
            if (d <= 4) return true;
            if (d <= 12 && visibleTiles != null && isVisible(npc.x, npc.y)) return true;
        }
        return false;
    }

    boolean hostileGovernorPressureNearby() {
        if (world == null || world.npcs == null || active == null) return false;
        for (NpcEntity npc : world.npcs) {
            if (!isHostileNpc(npc) || !isGovernorCombatNpc(npc)) continue;
            int d = Math.abs(npc.x - playerX) + Math.abs(npc.y - playerY);
            if (d <= 12) return true;
        }
        return false;
    }

    boolean isGovernorCombatNpc(NpcEntity npc) {
        if (npc == null) return false;
        String hay = ((npc.name == null ? "" : npc.name) + " " + (npc.role == null ? "" : npc.role)).toLowerCase(Locale.ROOT);
        return hay.contains("governor") || (world != null && world.zoneType == ZoneType.SECTOR_GOVERNORS_MANSION && isHostileNpc(npc));
    }

    void noteCombatMusicActivity(boolean governor, String reason) {
        combatMusicUntilTurn = Math.max(combatMusicUntilTurn, turn + COMBAT_MUSIC_TIMEOUT_TURNS);
        if (governor) governorBattleMusicUntilTurn = Math.max(governorBattleMusicUntilTurn, turn + COMBAT_MUSIC_TIMEOUT_TURNS * 2);
        updateDynamicMusic(reason);
    }

    void initFactionState() {
        for (Faction f : Faction.values()) {
            factionStanding.put(f, 0);
            temporaryHostileTurns.put(f, 0);
            factionMarketPressure.put(f, 0);
        }
    }

    void generateCandidates() {
        candidates.clear();
        characterNameEditActive = false;
        cachedOperatorProfileName = resolveOperatorProfileName();
        NameLockedProfilePortraitAuthority.setActiveOperatorProfileName(cachedOperatorProfileName);
        Optional<NameLockedProfilePortraitAuthority.Entry> preferred = NameLockedProfilePortraitAuthority.match(cachedOperatorProfileName);
        for (int i = 0; i < 12; i++) {
            Candidate c = Candidate.random(rng);
            if (i == 0 && preferred.isPresent()) applyNameLockedProfileCandidate(c, preferred.get());
            refreshNameLockedCandidateState(c);
            AgeAndWorldTimeAuthority.initializeCandidateAge(c, new Random(Objects.hash(c.name, c.portraitIndex, i, seed)));
            candidates.add(c);
        }
        CharacterCreationAudit audit = CharacterCreationAuditApi.audit(candidates);
        logEvent("Generated " + candidates.size() + " candidate characters with bounded stat spreads: " + audit.summaryLine());
        if (preferred.isPresent()) logEvent("Profile-linked portrait available for matching character name: " + preferred.get().displayName + ".");
        DebugLog.audit("CHARACTER_CREATION", audit.toLogBlock());
        DebugLog.audit("NAME_LOCKED_PROFILE_PORTRAIT", "candidateSeed profile=" + cachedOperatorProfileName + " preferred=" + (preferred.isPresent() ? preferred.get().key : "none"));
    }

    String resolveOperatorProfileName() {
        try {
            if (userProfile != null && userProfile.wrapperDetected && userProfile.displayName != null && !userProfile.displayName.isBlank()) return userProfile.displayName;
            FallbackProfileManagementAuthority.LocalProfile p = FallbackProfileManagementAuthority.loadOrCreate(userProfile);
            if (p != null && p.profileName != null && !p.profileName.isBlank()) return p.profileName;
        } catch (Throwable t) {
            DebugLog.warn("NAME_LOCKED_PROFILE_PORTRAIT", "profile-name resolution fell back: " + t.getMessage());
        }
        return userProfile == null ? "Local Operator" : userProfile.displayName;
    }

    void applyNameLockedProfileCandidate(Candidate c, NameLockedProfilePortraitAuthority.Entry e) {
        if (c == null || e == null) return;
        c.name = e.displayName;
        c.nameLockedProfileKey = e.key;
        c.portraitIndex = Math.abs(Objects.hash(e.key, seed));
    }

    void refreshNameLockedCandidateState(Candidate c) {
        if (c == null) return;
        NameLockedProfilePortraitAuthority.Entry e = NameLockedProfilePortraitAuthority.unlockedEntry(cachedOperatorProfileName, c.name);
        c.nameLockedProfileKey = e == null ? "" : e.key;
    }

    void setScreen(Screen next) {
        Screen oldScreen = screen;
        PanelMode oldPanel = panelMode;
        if (screen != next) previousScreen = screen;
        if (next == Screen.PANEL || next == Screen.OPTIONS || next == Screen.PAUSE) sounds.play("panelOpen", options);
        if (oldScreen == Screen.PANEL && next != Screen.PANEL) clearExpiredPanelState(oldPanel, "screen change to " + next);
        if (oldScreen == Screen.OPTIONS && next != Screen.OPTIONS) clearExpiredOptionsState("screen change to " + next);
        if (oldScreen == Screen.PAUSE && next != Screen.PAUSE) clearExpiredPauseState("screen change to " + next);
        if (oldScreen == Screen.INTRO_CRAWL && next != Screen.INTRO_CRAWL) sounds.stopIntroCrawlNarration("screen change to " + next);
        if (next == Screen.INTRO_CRAWL) {
            introCrawlStartMillis = System.currentTimeMillis();
            if (introCrawlLines.isEmpty()) introCrawlLines = images.loadIntroCrawlLines();
            sounds.playIntroCrawlNarration(options);
            DebugLog.audit("INTRO_CRAWL_BEGIN", "lines=" + introCrawlLines.size() + " queuedZoneSplash=" + introCrawlQueuedZoneSplash + " textTargetMs=" + INTRO_CRAWL_TEXT_TARGET_MS + " audioExpectedMs=" + INTRO_CRAWL_AUDIO_EXPECTED_MS + " audioTailSafetyMs=" + INTRO_CRAWL_AUDIO_TAIL_SAFETY_MS);
        }
        screen = next;
        selectedButton = 0;
        activeScrollTag = "";
        requestFocusInWindow();
        updateDynamicMusic("screen change to " + next);
        repaint();
    }

    void clearExpiredPanelState(PanelMode expired, String reason) {
        if (expired == PanelMode.NONE) return;
        DebugLog.audit("UI_CLOSE", "Clearing expired panel " + expired + " due to " + reason + " | " + stateSummary());
        activeScrollTag = "";
        scrollbarDragActive = false;
        if (expired == PanelMode.LOOK) {
            lookCursorActive = false;
            lookStackIndex = 0;
            lookX = playerX;
            lookY = playerY;
        }
        if (expired == PanelMode.COMBAT) {
            combatCursorActive = false;
            combatX = combatY = -1;
            lastTargetingReport = "No target selected.";
        }
        if (expired == PanelMode.INTERACT) {
            interactCursorActive = false;
            pendingInteractionSummary = "No interaction selected.";
        }
        if (expired == PanelMode.BUILD || expired == PanelMode.WORKBENCH) {
            buildPlacementActive = false;
            pendingBuildRecipe = null;
            pendingBuildSummary = "No build selected.";
            buildX = -1;
            buildY = -1;
        }
        if (expired == PanelMode.TRADE) {
            activeTrader = null;
            selectedTradeOffer = 0;
            conversationBranch = "No active conversation.";
        }
        if (expired == PanelMode.INTERACT || expired == PanelMode.TRADE) {
            activeConversationNpc = null;
        }
        if (expired == PanelMode.INFOPEDIA) {
            infopediaScroll = 0;
            infopediaDetailScroll = 0;
        }
        if (panelMode == expired) panelMode = PanelMode.NONE;
    }

    void clearExpiredOptionsState(String reason) {
        if (graphicsDropdown != -1 || scrollbarDragActive) DebugLog.audit("UI_CLOSE", "Clearing expired options controls due to " + reason + " | dropdown=" + graphicsDropdown + " | " + stateSummary());
        graphicsDropdown = -1;
        scrollbarDragActive = false;
        activeScrollTag = "";
    }

    void clearExpiredPauseState(String reason) {
        if (scrollbarDragActive) DebugLog.audit("UI_CLOSE", "Clearing expired pause state due to " + reason + " | " + stateSummary());
        scrollbarDragActive = false;
        activeScrollTag = "";
    }

    void openPanel(PanelMode mode) {
        panelMode = mode;
        setScreen(Screen.PANEL);
        if (mode == PanelMode.WORKBENCH && universalWindowAuthority != null) {
            universalWindowAuthority.open("machine_operations", turn, "workbench machine-operation status bridge");
            DebugLog.audit("MACHINE_OPERATION_STATUS_BRIDGE", MachineOperationStatusBridge.compactSummary(this));
            DebugLog.audit("STAFFING_LABOR_BRIDGE", StaffingLaborBridgeAuthority.compactSummary(this));
            }
        logEvent("Opened panel: " + mode);
    }

    void closePanel() {
        sounds.play("panelClose", options);
        PanelMode closing = panelMode;
        clearExpiredPanelState(panelMode, "explicit close");
        if (!lookCursorActive) { lookX = playerX; lookY = playerY; }
        if (previousScreen == Screen.LOST) setScreen(Screen.LOST);
        else if (previousScreen == Screen.MENU || previousScreen == Screen.PAUSE || previousScreen == Screen.OPTIONS || previousScreen == Screen.WORLD_SELECT || previousScreen == Screen.WORLD_SETUP || previousScreen == Screen.CHARACTER) setScreen(previousScreen);
        else if (previousScreen == Screen.MODS || previousScreen == Screen.SECTOR_AUDIT) setScreen(previousScreen);
        else if (world == null && closing == PanelMode.SAVELOAD) setScreen(Screen.MENU);
        else setScreen(Screen.GAME);
    }

    void logEvent(String s) {
        DebugLog.log(s);
        if (recentActionEligible(s)) {
            eventLog.add(s);
            while (eventLog.size() > 80) eventLog.remove(0);
        }
    }

    boolean recentActionEligible(String s) {
        if (s == null || s.isBlank()) return false;
        String low = s.toLowerCase(Locale.ROOT);
        if (low.startsWith("opened panel:") || low.startsWith("closed panel:") || low.startsWith("benchmark") || low.startsWith("metric/") || low.startsWith("audit/") || low.startsWith("settings/") || low.startsWith("debug") || low.contains("worstpaintms") || low.contains("renderprofile=") || low.contains("current.log") || low.contains("runtime.log") || low.contains("errors.log")) return false;
        if (low.contains("viewport zoom") || low.contains("world zoom") || low.startsWith("graphics dropdown") || low.startsWith("font scale") || low.startsWith("gui scale") || low.startsWith("tile icon rendering") || low.startsWith("screensaver") || low.startsWith("floating hover help") || low.startsWith("move preview") || low.contains("movement ghost") || low.startsWith("selected inventory") || low.startsWith("inventory item selected") || low.startsWith("equipment: selected")) return false;
        if (low.contains("route selected:") || low.contains("launcher shell") || low.contains("authority=") || low.contains("local-server boundary") || low.contains("zone audit generated") || low.contains("not-yet-implemented")) return false;
        return true;
    }

        try {
            DebugLog.audit(system, detail + " | state=" + stateSummary());

        }
    }

    void acceptEulaGate() {
        try {
            EulaConsentAuthority.accept(userProfile);
            eulaGateActive = false;
            showTimedAlert("EULA ACCEPTED", "Consent recorded locally. Runtime access is now unlocked.", 2600);
            repaint();
        } catch (Throwable t) {
            DebugLog.error("EULA_ACCEPT", "Consent failed; closing by fail-closed policy.", t);
            requestApplicationExit("EULA consent persistence failure");
        }
    }

    void requestApplicationExit(String reason) {
        DebugLog.audit("APPLICATION_EXIT", reason == null ? "unspecified" : reason);
        shutdownRuntime();
        Window w = SwingUtilities.getWindowAncestor(this);
        if (w != null) w.dispose();
        System.exit(0);
    }

    boolean toggleConsoleFlag(String key) {
        String safe = ChatRuntimeAuthority.ChatSecurity.sanitizeIdentifier(key == null ? "flag" : key);
        boolean next = !consoleBooleanFlags.getOrDefault(safe, false);
        consoleBooleanFlags.put(safe, next);
        logEvent("CONSOLE FLAG: " + safe + " " + (next ? "enabled" : "disabled") + ".");
        return next;
    }

    }

    void setConsoleStringFlag(String key, String value) {

    void writeSaveFile(int slot, boolean auto) throws IOException {
        if (active == null || world == null || atlas == null) { if (!auto) logEvent("Nothing to save yet: no active run."); return; }
        ServerRuntimePaths.ensureSinglePlayerDirectories();
        Path file = savePathForSlot(slot);
        Properties pr = new Properties();
        Persistence.writeCore(this, pr);
        Properties worldDefinition = SaveEfficiencyAuthority.worldRuntimeProperties(this);
        SaveEfficiencyAuthority.prepareSinglePlayerBundledSlot(this, pr, worldDefinition);
        SaveEfficiencyAuthority.saveRuntimeWorldFile(this, worldDefinition, ServerRuntimePaths.SaveDomain.SINGLE_PLAYER);
        try (OutputStream out = Files.newOutputStream(file)) { pr.store(out, "The Mechanist " + slotLabel(slot)); }
        if (!auto) logEvent("Saved " + slotLabel(slot) + " to " + file.toString() + ".");
        DebugLog.audit("SAVE_CATALOG", SaveEfficiencyAuthority.auditLine(slotLabel(slot), pr, worldDefinition));
        DebugLog.audit(auto ? "AUTOSAVE_COMPLETE" : "SAVE_COMPLETE", "slot=" + slot + " file=" + file.toAbsolutePath() + " keys=" + pr.size() + " state=" + stateSummary());
    }

    void saveGameSlot(int slot) {

        });
    }

    void maybeAutosave(String reason) {
        if (active == null || world == null || atlas == null || screen == Screen.LOST) return;
        if (turn <= 0 || turn == lastAutosaveTurn) return;
        if (turn % AUTOSAVE_INTERVAL_TURNS != 0) return;
        try {
            writeSaveFile(AUTOSAVE_HOURLY_SLOT, true);
            lastAutosaveTurn = turn;
            logEvent("AUTOSAVE: hourly save at turn " + turn + " after " + reason + ".");
        } catch (Throwable t) {
            DebugLog.error("AUTOSAVE_FAILED", "Hourly autosave failed at turn " + turn + " reason=" + reason, t);
            logEvent("AUTOSAVE WARNING: could not write hourly autosave. Diagnostic details were recorded.");
        }
    }

    void autosaveZoneTransition(String reason) {
        if (active == null || world == null || atlas == null || screen == Screen.LOST) return;
        if (turn <= 0 || turn == lastZoneAutosaveTurn) return;
        try {
            writeSaveFile(AUTOSAVE_ZONE_SLOT, true);
            lastZoneAutosaveTurn = turn;
            logEvent("AUTOSAVE: zone-transition save at turn " + turn + " after " + reason + ".");
        } catch (Throwable t) {
            DebugLog.error("AUTOSAVE_ZONE_FAILED", "Zone-transition autosave failed at turn " + turn + " reason=" + reason, t);
            logEvent("AUTOSAVE WARNING: could not write zone-transition autosave. Diagnostic details were recorded.");
        }
    }

    void loadGameSlot(int slot) {
        runGuarded("LOAD", "slot=" + slot, () -> {
            Path file = savePathForSlot(slot);
            if (!Files.exists(file)) { logEvent("No save file found for " + slotLabel(slot) + " at " + file.toString() + "."); return; }
            Properties pr = new Properties();
            try { try (InputStream in = Files.newInputStream(file)) { pr.load(in); } } catch (IOException ex) { throw new RuntimeException(ex); }
            SaveEfficiencyAuthority.mergeReferencedWorldState(pr);
            loadingForNewGame = false;
            loadingForTransition = false;
            loadingProgress = 0;
            loadingStep = 0;
            loadingStartMillis = System.currentTimeMillis();
            loadingStatus = "Loading saved underhive state.";
            Persistence.readCore(this, pr);
            updateSensoryModel("load " + slotLabel(slot));
            logEvent("Loaded " + slotLabel(slot) + ": " + stateSummary());
            DebugLog.audit("LOAD_COMPLETE", "slot=" + slot + " file=" + file.toAbsolutePath() + " keys=" + pr.size() + " state=" + stateSummary());
            setScreen(Screen.GAME);
        });
    }

    }

    void sanityCheck(String system) {
        if (selectedButton < 0) {
            DebugLog.warn(system, "selectedButton below zero; correcting. " + stateSummary());
            selectedButton = 0;
        }
        if (buttons.size() > 0 && selectedButton >= buttons.size()) {
            DebugLog.warn(system, "selectedButton outside button list; correcting. buttons=" + buttons.size() + " " + stateSummary());
            selectedButton = buttons.size() - 1;
        }
        if ((screen == Screen.GAME || screen == Screen.PANEL) && world == null) {
            DebugLog.warn(system, "Game/panel screen active without world. " + stateSummary());
        }
        if (world != null && (playerX < 0 || playerY < 0 || playerX >= world.w || playerY >= world.h)) {
            DebugLog.warn(system, "Player outside world bounds. " + stateSummary());
        }
    }
