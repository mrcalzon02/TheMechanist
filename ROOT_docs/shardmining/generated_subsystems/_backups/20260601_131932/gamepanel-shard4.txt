    void interactVendingMachine(int tx, int ty, char ch) {
        String key = passiveKey(tx, ty);
        int until = passiveCooldownUntilTurn.getOrDefault(key, 0);
        int hackedCount = vendingHackVendCount.getOrDefault(key, 0);
        if (turn < until) {
            int need = Math.max(0, until - turn);
            if (hackedCount >= PublicServiceMediaAuthority.maxHackedCycles()) {
                logEvent("The " + vendingName(ch) + " is hacked dry. Its machine-spirit sulks in empty spirals.");
                advanceTurn("checks an empty vending machine.");
                return;
            }
            if (countDataSpikes() <= 0) {
                logEvent("The " + vendingName(ch) + " is cooling down for " + need + " more turns. Illegal override requires a Data spike.");
                DebugLog.audit("VENDING_HACK_NO_SPIKE", vendingName(ch)+" key="+key+" remaining="+need+" state="+stateSummary());
                advanceTurn("checks a vending machine access port.");
                return;
            }
            int intellect = stat("Intellect", 3);
            int hackDifficulty = PublicServiceMediaAuthority.hackDifficulty(hackedCount);
            int check = intellect + stat("Mechanics", 3) / 2 + rng.nextInt(20) + 1;
            consumeDataSpike();
            if (check >= hackDifficulty) {
                vendingHackVendCount.put(key, hackedCount + 1);
                passiveCooldownUntilTurn.put(key, turn + PublicServiceMediaAuthority.hackedVendingCooldownTurns());
                suspicion += 2;
                logEvent("You burn a Data spike into the " + vendingName(ch) + " and illegally force a vend cycle. It will now vend on a 1-hour cycle until 8 hacked items are dispensed. Suspicion +2.");
                vendItem(ch, true);
                DebugLog.audit("VENDING_HACK", vendingName(ch)+" key="+key+" hackedCount="+(hackedCount+1)+" check="+check+" diff="+hackDifficulty+" intellect="+intellect+" dataSpikeConsumed=true state="+stateSummary());
            } else {
                suspicion += 1;
                logEvent("Data spike burned. Hack failed: roll " + check + " < " + hackDifficulty + ". The " + vendingName(ch) + " remains on cooldown for " + need + " more turns. Suspicion +1.");
                DebugLog.audit("VENDING_HACK_FAIL", vendingName(ch)+" key="+key+" remaining="+need+" check="+check+" diff="+hackDifficulty+" dataSpikeConsumed=true state="+stateSummary());
            }
            advanceTurn("tampered with " + vendingName(ch) + ".");
            return;
        }
        if (hackedCount >= PublicServiceMediaAuthority.maxHackedCycles()) {
            logEvent("The " + vendingName(ch) + " is hacked dry. Its machine-spirit sulks in empty spirals.");
            advanceTurn("checks an empty vending machine.");
            return;
        }
        int cost = vendingCost(ch);
        if (!spendImperialScript(cost)) {
            logEvent("The " + vendingName(ch) + " demands " + cost + " Concord Script. You have " + countMoney() + ".");
            DebugLog.audit("VENDING_PURCHASE_FAIL", vendingName(ch)+" cost="+cost+" money="+countMoney()+" state="+stateSummary());
            advanceTurn("fails to purchase from " + vendingName(ch) + ".");
            return;
        }
        passiveCooldownUntilTurn.put(key, turn + PublicServiceMediaAuthority.legalVendingCooldownTurns());
        vendItem(ch, false);
        DebugLog.audit("VENDING_PURCHASE", vendingName(ch)+" key="+key+" cost="+cost+" cooldownUntil="+passiveCooldownUntilTurn.get(key)+" state="+stateSummary());
        advanceTurn("purchases from " + vendingName(ch) + ".");
    }

    int sleepTurnCost(int hours) { return Math.max(1, hours * TURNS_PER_HOUR); }
    int sleepFoodWaterCost(int hours) {
        boolean uninjured = wounds <= 0 && totalBodyDamagePercent() <= 0.1;
        double rate = uninjured ? 5.0 / 32.0 : 5.0 / 8.0;
        return Math.max(1, (int)Math.ceil(hours * rate));
    }
    double totalBodyDamagePercent() {
        if (active == null || active.body == null || active.body.isEmpty()) return wounds;
        double lost = 0, max = 0;
        for (BodyPart bp : active.body.values()) { lost += Math.max(0, bp.maxHealth() - bp.currentHealth()); max += Math.max(1, bp.maxHealth()); }
        return max <= 0 ? 0 : (lost / max) * 100.0;
    }

    void sleepOnCot(int hours) {
        BaseObject obj = baseObjectAt(lookX, lookY);
        if (obj == null || obj.symbol != 'c') { logEvent("Sleep failed: interact cursor is not on a cot."); return; }
        int cost = sleepFoodWaterCost(hours);
        if (food < cost || water < cost) { logEvent("Sleep blocked: needs " + cost + " food and " + cost + " water. Current food " + food + ", water " + water + "."); return; }
        food -= cost; water -= cost;
        int oldFatigue = fatigue;
        int oldWounds = wounds;
        int oldSleepNeed = sleepNeed;
        fatigue = Math.max(0, fatigue - hours);
        wounds = Math.max(0, wounds - Math.max(1, cost));
        int turnsToSpend = sleepTurnCost(hours);
        sleepNeed = Math.max(0, sleepNeed - turnsToSpend);
        DebugLog.audit("SLEEP", "hours=" + hours + " turns=" + turnsToSpend + " foodCost=" + cost + " waterCost=" + cost + " fatigue " + oldFatigue + "->" + fatigue + " wounds " + oldWounds + "->" + wounds + " sleepNeed " + oldSleepNeed + "->" + sleepNeed + " state=" + stateSummary());
        logEvent("You sleep " + hours + " hour(s). Food " + food + "/15 Water " + water + "/15. Sleep need " + oldSleepNeed + "->" + sleepNeed + "/" + MAX_SLEEP_NEED + ". Fatigue " + oldFatigue + "->" + fatigue + ", wounds " + oldWounds + "->" + wounds + ".");
        for (int i=0; i<turnsToSpend; i++) advanceTurnSleeping("sleeps under bad pipes and worse dreams.");
        updatePendingInteractionSummary();
    }

    void consumeEmergencyRation() {
        int idx = -1;
        for (int i=0; i<inventory.size(); i++) {
            String item = inventory.get(i).toLowerCase(Locale.ROOT);
            if (item.contains("ration") || item.contains("canteen") || item.contains("water")) { idx = i; break; }
        }
        if (idx < 0) { logEvent("No emergency rations, canteens, or water items found in inventory."); return; }
        String item = inventory.get(idx);
        ItemActionResult consumed = transferContainerItemResultAt(CONTAINER_PLAYER_INVENTORY, inventory, idx, CONTAINER_VOID_CONSUMED, "consumed item ledger", null, "consumed by player as emergency ration");
        if (!consumed.success) { logEvent("Cannot consume emergency ration: " + consumed.playerText); return; }
        int oldFood = food, oldWater = water;
        if (item.toLowerCase(Locale.ROOT).contains("water") || item.toLowerCase(Locale.ROOT).contains("canteen")) water = Math.min(MAX_FOOD_WATER, water + 4);
        else { food = Math.min(MAX_FOOD_WATER, food + 4); water = Math.min(MAX_FOOD_WATER, water + 2); }
        DebugLog.audit("CONSUME_ITEM", "item=" + item + " food " + oldFood + "->" + food + " water " + oldWater + "->" + water + " state=" + stateSummary());
        logEvent("Consumed " + item + ". Food " + oldFood + "->" + food + ", water " + oldWater + "->" + water + ".");
        advanceTurn("uses an inventory ration item.");
    }

    void applyAwakeSurvivalTick() {
        if (turn > 0 && turn % TURNS_PER_HOUR == 0) {
            sleepNeed = Math.min(MAX_SLEEP_NEED, sleepNeed + TURNS_PER_HOUR);
            DebugLog.audit("SLEEP_PRESSURE", "sleepNeed=" + sleepNeed + "/" + MAX_SLEEP_NEED + " time=" + timeText() + " state=" + stateSummary());
        }
        if (turn > 0 && turn % (8 * TURNS_PER_HOUR) == 0) {
            if (food > 0) food--;
            if (water > 0) water--;
            DebugLog.audit("AWAKE_METABOLISM", "food=" + food + " water=" + water + " sleepNeed=" + sleepNeed + " time=" + timeText() + " state=" + stateSummary());
            logEvent("SURVIVAL CLOCK: eight hours pass. Food " + food + "/" + MAX_FOOD_WATER + ", water " + water + "/" + MAX_FOOD_WATER + ".");
        }
        if (sleepNeed >= MAX_SLEEP_NEED) passOutFromSleepNeed();
    }


    void applyHourlyProgressionTick(boolean sleeping) {
        if (turn <= 0 || turn % TURNS_PER_HOUR != 0) return;
        int oldHeat = gangHeat, oldSuspicion = suspicion;
        int heatDecay = sleeping ? 2 : 1;
        int suspicionDecay = sleeping ? 2 : 1;
        if (!baseClaimed) suspicionDecay += 1;
        gangHeat = Math.max(0, gangHeat - heatDecay);
        suspicion = Math.max(0, suspicion - suspicionDecay);
        if (oldHeat != gangHeat || oldSuspicion != suspicion) {
            logEvent("PRESSURE DECAY: an hour passes. Heat " + oldHeat + "->" + gangHeat + ", suspicion " + oldSuspicion + "->" + suspicion + ".");
            DebugLog.audit("PRESSURE_DECAY", "sleeping=" + sleeping + " heat=" + oldHeat + "->" + gangHeat + " suspicion=" + oldSuspicion + "->" + suspicion + " time=" + timeText() + " state=" + stateSummary());
        }
    }

    void gainXp(String skill, int amount, String source) {
        amount = Math.max(1, amount);
        int oldXp = xp;
        int oldCredits = knowledgeCredits;
        xp += amount;
        int oldBands = oldXp / 25;
        int newBands = xp / 25;
        if (newBands > oldBands) knowledgeCredits += (newBands - oldBands);
        String msg = "XP GAINED: +" + amount + " " + skill + " XP from " + source + ". Total XP " + oldXp + "->" + xp + ".";
        if (knowledgeCredits != oldCredits) msg += " Knowledge credits " + oldCredits + "->" + knowledgeCredits + ".";
        logEvent(msg);
        DebugLog.audit("XP_GAIN", "skill=" + skill + " amount=" + amount + " source=" + source + " xp=" + oldXp + "->" + xp + " knowledgeCredits=" + oldCredits + "->" + knowledgeCredits + " state=" + stateSummary());
    }

    void handlePlayerUnconsciousness(String cause, String attacker, String weapon, boolean theftRisk) {
        if (screen == Screen.LOST) return;
        runUnconsciousEvents++;
        int stolenItems = 0;
        int stolenScript = 0;
        if (theftRisk) {
            stolenScript = unconsciousScriptLoss();
            if (stolenScript > 0) carriedScript = Math.max(0, carriedScript - stolenScript);
            stolenItems = unconsciousItemLoss();
        }
        Point wake = unconsciousWakePoint();
        playerX = wake.x; playerY = wake.y; lookX = playerX; lookY = playerY;
        fatigue = Math.max(0, Math.min(10, fatigue / 2));
        sleepNeed = 0;
        pain = Math.max(0, pain / 2);
        bleeding = 0;
        wounds = Math.max(0, wounds - 2);
        if (active != null && active.body != null) healWorstBodyPart(25.0);
        updateSensoryModel("unconscious relocation");
        DebugLog.audit("PLAYER_UNCONSCIOUS", "cause=" + cause + " attacker=" + attacker + " weapon=" + weapon + " scriptLost=" + stolenScript + " inventoryLost=" + stolenItems + " wake=" + playerX + "," + playerY + " state=" + stateSummary());
        logEvent("UNCONSCIOUS: " + cause + ". You are carried, dragged, ransomed, or simply found later and wake at " + wakePointLabel(wake) + ".");
        if (stolenScript > 0) logEvent("SCRIPT STOLEN: " + stolenScript + " Concord Script was lifted from your pockets while you were out.");
        if (stolenItems > 0) logEvent("ITEMS LOST: " + stolenItems + " carried item(s) were gone when you woke.");
        maybeAutosave("unconscious recovery");
    }

    int unconsciousScriptLoss() {
        if (carriedScript <= 0) return 0;
        int pct = 10 + rng.nextInt(6); // usually around a tenth; enough to sting without total liquidation.
        return Math.max(1, Math.min(carriedScript, (int)Math.ceil(carriedScript * pct / 100.0)));
    }

    int unconsciousItemLoss() {
        if (inventory.isEmpty()) return 0;
        int target = Math.max(0, (int)Math.floor(inventory.size() * 0.10));
        if (inventory.size() >= 5) target = Math.max(1, target);
        target = Math.min(target, Math.max(0, inventory.size() / 3));
        int lost = 0;
        for (int i=0; i<target && !inventory.isEmpty(); i++) {
            int idx = rng.nextInt(inventory.size());
            String removed = removeInventoryAtWithTrace(idx);
            if (removed != null) lost++;
        }
        return lost;
    }

    String wakePointLabel(Point p) {
        if (baseClaimed && p != null && p.x == baseX && p.y == baseY) return "your claimed base";
        if (world != null && p != null && world.inBounds(p.x, p.y) && world.tiles[p.x][p.y] == 'C') return "the nearest medicae clinic";

    Point nearestMedicalFacilityPoint() {
        if (world == null) return null;
        Point best = null; int bestD = Integer.MAX_VALUE;
        for (int x=1;x<world.w-1;x++) for (int y=1;y<world.h-1;y++) {
            boolean clinicTile = world.tiles[x][y] == 'C';
            boolean medRoom = false;
            int rid = world.roomIdAt(x,y);
            if (rid >= 0 && rid < world.roomProfiles.size() && world.roomProfiles.get(rid) != null) {
                String n = world.roomProfiles.get(rid).name.toLowerCase(Locale.ROOT);
                medRoom = n.contains("medicae") || n.contains("clinic") || n.contains("triage") || n.contains("infirmary") || n.contains("aid");
            }
            if ((clinicTile || medRoom) && world.walkable(x,y)) {
                int d = Math.abs(x-playerX) + Math.abs(y-playerY);
                if (d < bestD) { bestD = d; best = new Point(x,y); }
            }
        }
        return best;
    }

    boolean shouldPlayerDieNow() {
        if (active == null || active.body == null) return false;
        BodyPart head = active.body.get("Head");
        BodyPart torso = active.body.get("Torso");
        boolean vitalDestroyed = (head != null && head.destroyed()) || (torso != null && torso.destroyed());
        return vitalDestroyed || wounds >= 20 || (wounds >= 16 && bleeding > 0);
    }

    boolean shouldPlayerFallUnconsciousNow() {
        return wounds >= 12 || fatigue >= 24 || pain >= 18;
    }

    void evaluatePlayerDefeatState(String cause, String attacker, String weapon, String location) {
        if (screen == Screen.LOST || active == null) return;
        if (shouldPlayerDieNow()) {
            triggerPlayerDeath(cause, attacker, weapon, location);
        } else if (shouldPlayerFallUnconsciousNow()) {
            if (arbitesAuthorityText(attacker) || arbitesAuthorityText(weapon)) handleArbitesCapture("incapacitated by Civic Wardens authority", Math.max(2, suspicion / 4 + wounds / 6));
            else handlePlayerUnconsciousness(cause, attacker, weapon, false);
        }
    }

        evaluateFactionPressure();

    int movementBaseTurnCost() {
        int cost = 1 + Math.max(0, inventoryWeight() - carryCapacity());
        int legPenalty = crippledLegCount();
        if (legPenalty > 0) cost += legPenalty;
        if (pain >= 6) cost += 1;
        return Math.max(1, cost);
    }

    int movementTurnCost() {
        int cost = movementBaseTurnCost();
        if (movementModeIndex == MOTION_SNEAK) cost += 2;      // SNEAK: slow, crouched, concealment-first travel.
        else if (movementModeIndex == MOTION_WALK) cost += 1; // WALK: deliberate and observant.
        else if (movementModeIndex == MOTION_SPRINT) cost = Math.max(1, cost - 1); // SPRINT: fast, loud, sloppy.
        return Math.max(1, cost);
    }

    Point slidePixelOffset(TileMotion m, int cellW, int cellH) {
        if (m == null || m.done(System.currentTimeMillis())) return new Point(0, 0);
        double p = m.progress(System.currentTimeMillis());
        int ox = (int)Math.round((m.fromX - m.toX) * cellW * (1.0 - p));
        int oy = (int)Math.round((m.fromY - m.toY) * cellH * (1.0 - p));
        return new Point(ox, oy);
    }

    Point playerSlideOffsetPixels(int cellW, int cellH) {
        if (playerTileMotion == null) return new Point(0,0);
        if (playerTileMotion.toX != playerX || playerTileMotion.toY != playerY || playerTileMotion.done(System.currentTimeMillis())) { playerTileMotion = null; return new Point(0,0); }
        return slidePixelOffset(playerTileMotion, cellW, cellH);
    }

    Point npcSlideOffsetPixels(NpcEntity npc, int cellW, int cellH) {
        if (npc == null) return new Point(0,0);
        TileMotion m = npcTileMotions.get(npc.id);
        if (m == null) return new Point(0,0);
        if (m.toX != npc.x || m.toY != npc.y || m.done(System.currentTimeMillis())) { npcTileMotions.remove(npc.id); return new Point(0,0); }
        return slidePixelOffset(m, cellW, cellH);
    }

    String pickWeightedCue(String[] keys, int[] weights) {
        if (keys == null || keys.length == 0) return "ambient_machine";
        int total = 0;
        for (int w : weights) total += Math.max(0, w);
        if (total <= 0) return keys[0];
        int roll = rng.nextInt(total);
        for (int i=0; i<keys.length; i++) {
            roll -= Math.max(0, weights[Math.min(i, weights.length-1)]);
            if (roll < 0) return keys[i];
        }
        return keys[keys.length-1];
    }

    String coreAmbientCueForZone() {
        if (world == null || world.zoneType == null) return "ambient_machine";

            new String[]{"ambient_pipe","ambient_vent","ambient_drip","ambient_sludge","ambient_alarm_far"},
            new int[]{32,24,24,16,4});
        if (z.contains("noble") || z.contains("governor")) return pickWeightedCue(
            new String[]{"ambient_tv","ambient_radio","ambient_chime","ambient_servitor","ambient_door_servo"},
            new int[]{30,26,16,16,12});
        if (z.contains("industrial") || z.contains("mechanist Collegia") || z.contains("factory")) return pickWeightedCue(
            new String[]{"ambient_machine","ambient_spark","ambient_press","ambient_servo","ambient_alarm_far","ambient_vent"},
            new int[]{30,20,18,14,8,10});
        if (z.contains("market") || z.contains("hab")) return pickWeightedCue(
            new String[]{"ambient_radio","ambient_tv","ambient_machine","ambient_door_servo","ambient_pipe"},
            new int[]{32,22,18,14,14});

    }

            if (f.contains("sludge") || f.contains("runoff") || f.contains("sewer")) return "sludge";
            if (f.contains("wire") || f.contains("electric")) return "grate";
        }
        char t = world.tiles[x][y];
        if (t == '~' || t == ',' || t == ';') return "sludge";
        if (t == ':' || t == '=') return "grate";
        if (t == '*' || t == '`') return "debris";
        return "metal";
    }

    void updateMoveGhostTarget(int dx, int dy) {
        if (world == null) { moveGhostX = moveGhostY = -1; moveGhostRange = 1; return; }
        if (dx == 0 && dy == 0) { dx = lastMoveDx; dy = lastMoveDy; }
        if (dx == 0 && dy == 0) { moveGhostX = playerX; moveGhostY = playerY; moveGhostRange = 0; return; }
        lastMoveDx = dx; lastMoveDy = dy;
        int tx = playerX, ty = playerY;
        int range = movementModeRange();
        for (int i=0; i<range; i++) {
            int nx = tx + dx, ny = ty + dy;
            if (world == null || !world.walkable(nx, ny)) break;
            tx = nx; ty = ny;
            if (world.npcAt(nx, ny) != null) break;
        }
        moveGhostX = tx; moveGhostY = ty; moveGhostRange = Math.max(1, range);
    }

    int inputPacingIntervalMillis() {
        switch (movementModeIndex) {
            case MOTION_SNEAK: return INPUT_PACING_SNEAK_INTERVAL_MS;
            case MOTION_WALK: return INPUT_PACING_WALK_INTERVAL_MS;
            case MOTION_SPRINT: return INPUT_PACING_SPRINT_INTERVAL_MS;
            case MOTION_RUN:
            default: return INPUT_PACING_RUN_INTERVAL_MS;
        }
    }

    void clearPendingMovementInput(String reason) {
        if (pendingHeldMove || movementGhostArmed) DebugLog.audit("INPUT_PACING_CLEAR", "reason=" + reason + " pending=" + pendingMoveDx + "," + pendingMoveDy + " armed=" + armedMoveDx + "," + armedMoveDy + " turn=" + turn);
        movementGhostArmed = false;
        armedMoveDx = 0;
        armedMoveDy = 0;
        pendingMoveDx = 0;
        pendingMoveDy = 0;
        pendingHeldMove = false;
    }

    boolean movementInputWouldAdvanceWorld() {
        return screen == Screen.GAME && world != null && active != null && screen != Screen.LOST;
    }

    void queueOrExecuteMovementInput(int dx, int dy) {
        if (dx == 0 && dy == 0) return;
        clearMouseMovementPreview("keyboard movement");
        if (!movementInputWouldAdvanceWorld()) { clearPendingMovementInput("not-gameplay"); return; }
        long now = System.currentTimeMillis();
        if (now >= nextWorldActionAllowedMillis && !pendingHeldMove) {
            executePacedMovement(dx, dy, "direct");
            return;
        }
        // Strict buffer: one active move may have one pending held-key move behind it. Nothing deeper.
        pendingMoveDx = dx;
        pendingMoveDy = dy;
        pendingHeldMove = true;
        DebugLog.audit("INPUT_PACING_QUEUE", "pending=" + dx + "," + dy + " cap=" + STRICT_INPUT_MOVEMENT_BUFFER_CAP + " nextAllowedInMs=" + Math.max(0L, nextWorldActionAllowedMillis - now) + " mode=" + movementModeLabel() + " turn=" + turn);
    }

    void executePacedMovement(int dx, int dy, String source) {
        if (!movementInputWouldAdvanceWorld()) { clearPendingMovementInput("execute-not-gameplay"); return; }
        if (movementModeRange() > 1 && !(movementGhostArmed && armedMoveDx == dx && armedMoveDy == dy)) {
            updateMoveGhostTarget(dx, dy);
            movementGhostArmed = true;
            armedMoveDx = dx;
            armedMoveDy = dy;
            logEvent("MOVE PREVIEW: " + movementModeLabel() + " projects to " + moveGhostX + "," + moveGhostY + ". Press the same direction again to confirm.");
            DebugLog.audit("MOVE_GHOST_ARMED", "dx=" + dx + " dy=" + dy + " target=" + moveGhostX + "," + moveGhostY + " range=" + movementModeRange() + " source=" + source + " turn=" + turn);
            return;
        }
        if (singlePlayerSectorBridge.shouldOwnAuthoritativeTurns(this) && !singlePlayerSectorBridge.insideAuthoritativeTurn()) {
            singlePlayerSectorBridge.submitCommand(this, new MovePlayerCommand(SinglePlayerSectorRuntimeBridge.LOCAL_PLAYER_ID, dx, dy, source));
            return;
        }
        executePacedMovementBody(dx, dy, source);
    }

    void executePacedMovementBody(int dx, int dy, String source) {
        if (!movementInputWouldAdvanceWorld()) { clearPendingMovementInput("execute-not-gameplay"); return; }
        long now = System.currentTimeMillis();
        lastWorldActionMillis = now;
        nextWorldActionAllowedMillis = now + inputPacingIntervalMillis();
        movementGhostArmed = false;
        boolean moved = movementModeRange() > 1 ? attemptPlayerMove(dx, dy) : attemptPlayerMoveOneTile(dx, dy);
        if (!moved) clearPendingMovementInput("blocked-or-failed-" + source);
        else DebugLog.audit("INPUT_PACING_EXECUTE", "source=" + source + " moved=" + dx + "," + dy + " intervalMs=" + inputPacingIntervalMillis() + " mode=" + movementModeLabel() + " range=" + movementModeRange() + " turn=" + turn);
    }

    void processPendingMovementInput() {
        if (!pendingHeldMove) return;
        if (!movementInputWouldAdvanceWorld()) { clearPendingMovementInput("timer-not-gameplay"); return; }
        long now = System.currentTimeMillis();
        if (now < nextWorldActionAllowedMillis) return;
        int dx = pendingMoveDx, dy = pendingMoveDy;
        pendingMoveDx = 0;
        pendingMoveDy = 0;
        pendingHeldMove = false;
        executePacedMovement(dx, dy, "held-buffer");
        repaint();
    }

    void spendDeferredSimulationBudget(String reason, int budgetMillis) {
        if (budgetMillis <= 0 || deferredSimulationDebt <= 0) return;
        if (singlePlayerSectorBridge.shouldOwnAuthoritativeTurns(this) && !singlePlayerSectorBridge.insideAuthoritativeTurn()) {
            singlePlayerSectorBridge.mutate(this, "deferred simulation budget: " + reason, () -> spendDeferredSimulationBudgetBody(reason, budgetMillis));
            return;
        }
        spendDeferredSimulationBudgetBody(reason, budgetMillis);
    }

    void spendDeferredSimulationBudgetBody(String reason, int budgetMillis) {
        if (budgetMillis <= 0 || deferredSimulationDebt <= 0) return;
        long start = System.nanoTime();
        long deadline = start + budgetMillis * 1_000_000L;
        int paid = 0;
        while (deferredSimulationDebt > 0 && System.nanoTime() < deadline) {
            int chunk = Math.min(deferredSimulationDebt, 8);
            AbstractDistantZoneSimulation.TickResult ar = abstractDistantZoneSimulation.spend(world, playerX, playerY, turn, chunk);
            int spent = Math.max(1, ar.deferredPaid);
            deferredSimulationDebt = Math.max(0, deferredSimulationDebt - spent);
            paid += spent;
            profilingAbstractZoneTicks += ar.ledgersTouched;
            profilingAbstractZoneWork += ar.spentWork;
            if (abstractDistantZoneSimulation.shouldAudit(turn)) DebugLog.audit("ABSTRACT_DISTANT_ZONE", "reason=" + reason + " " + ar.summary + " global=" + abstractDistantZoneSimulation.globalSummary() + " debt=" + deferredSimulationDebt);
            if (ar.spentWork <= 0 && chunk <= 1) break;
        }
        if (paid > 0) {
            profilingDeferredBudgetRuns++;
            profilingIdleBudgetNanos += Math.max(0L, System.nanoTime() - start);
            if (profilingDeferredBudgetRuns % 100 == 0) DebugLog.audit("SIM_BUDGET", "reason=" + reason + " paid=" + paid + " remainingDebt=" + deferredSimulationDebt + " budgetMs=" + budgetMillis + " turn=" + turn);
        }
    }

    void auditSimulationEfficiency(String reason) {
        long avgTurn = profilingAdvanceTurnSamples <= 0 ? 0L : profilingAdvanceTurnNanos / profilingAdvanceTurnSamples;
        long avgNpc = profilingNpcTickSamples <= 0 ? 0L : profilingNpcTickNanos / profilingNpcTickSamples;
        long avgIdle = profilingDeferredBudgetRuns <= 0 ? 0L : profilingIdleBudgetNanos / profilingDeferredBudgetRuns;
        DebugLog.audit("SIM_EFFICIENCY", "reason=" + reason + " avgAdvanceTurnNs=" + avgTurn + " samples=" + profilingAdvanceTurnSamples + " avgNpcTickNs=" + avgNpc + " npcSamples=" + profilingNpcTickSamples + " npcSchedulerSelections=" + profilingNpcSchedulerSelections + " npcSchedulerDeferred=" + profilingNpcSchedulerDeferred + " deferredDebt=" + deferredSimulationDebt + " avgBudgetNs=" + avgIdle + " budgetRuns=" + profilingDeferredBudgetRuns + " inputPending=" + pendingHeldMove + " nextActionMs=" + Math.max(0L, nextWorldActionAllowedMillis - System.currentTimeMillis()) + " lightRebuilds=" + profilingLightCacheRebuilds + " noiseRebuilds=" + profilingNoiseCacheRebuilds + " visLocalClears=" + profilingVisibilityLocalClears + " visFullClears=" + profilingVisibilityFullClears + " dirty=" + lastDirtyRegionReport + " abstractTicks=" + profilingAbstractZoneTicks + " abstractWork=" + profilingAbstractZoneWork + " abstract=" + abstractDistantZoneSimulation.globalSummary());
    }

    boolean attemptPlayerMove(int dx, int dy) {
        if (world == null) return false;
        updateMoveGhostTarget(dx, dy);
        int steps = movementModeRange();
        boolean movedAny = false;
        for (int i=0; i<steps; i++) {
            int nx = playerX + dx, ny = playerY + dy;
            boolean directClear = stepIsClearForPath(nx, ny);
            boolean moved = directClear ? attemptPlayerMoveOneTile(dx, dy) : attemptPlayerPathSeekingStep(dx, dy);
            if (!moved) break;
            movedAny = true;
            if (screen == Screen.LOST || world == null) break;
            // Multi-tile movement remains conservative around entity contact.  If the next
            // target is occupied, stop here so shove/push-past is deliberate, not automated.
            nx = playerX + dx; ny = playerY + dy;
            if (world.inBounds(nx, ny) && world.npcAt(nx, ny) != null) break;
        }
        updateMoveGhostTarget(dx, dy);
        return movedAny;
    }


    boolean stepIsClearForPath(int x, int y) {
        return world != null && world.inBounds(x, y) && world.walkable(x, y) && world.npcAt(x, y) == null;
    }

    int[][] pathSeekingAlternates(int dx, int dy) {
        if (dx == 0 && dy < 0) return new int[][]{{1,0},{-1,0},{0,-1}};   // north: try right first, then left, then straight if it opens.
        if (dx == 0 && dy > 0) return new int[][]{{-1,0},{1,0},{0,1}};    // south: mirror the preference so corridors do not jitter.
        if (dx > 0 && dy == 0) return new int[][]{{0,1},{0,-1},{1,0}};    // east: try down/up around corners.
        if (dx < 0 && dy == 0) return new int[][]{{0,-1},{0,1},{-1,0}};   // west: try up/down around corners.
        return new int[][]{{dx,dy}};
    }

    boolean attemptPlayerPathSeekingStep(int dx, int dy) {
        if (world == null) return false;
        for (int[] alt : pathSeekingAlternates(dx, dy)) {
            int ax = alt[0], ay = alt[1];
            if (ax == 0 && ay == 0) continue;
            int nx = playerX + ax, ny = playerY + ay;
            if (!stepIsClearForPath(nx, ny)) continue;
            DebugLog.audit("MOVE_PATH_SEEK", "requested=" + dx + "," + dy + " detour=" + ax + "," + ay + " from=" + playerX + "," + playerY + " to=" + nx + "," + ny + " mode=" + movementModeLabel() + " turn=" + turn);
            movePlayerInto(nx, ny, movementModeLabel().toLowerCase(Locale.ROOT) + " detours to " + nx + "," + ny + " around a blocked direct path.");
            checkTrapAfterPlayerMove(nx, ny);
            return true;
        }
        resetPushPastIntent();
        logEvent("Movement blocked: no clear detour found for " + movementModeLabel() + ".");
        return false;
    }

    boolean attemptPlayerMoveOneTile(int dx, int dy) {
        if (world == null) return false;
        int nx = playerX + dx, ny = playerY + dy;
        if (!world.walkable(nx, ny)) {
            resetPushPastIntent();
            logEvent("Movement blocked at " + nx + "," + ny + ".");
            return false;
        }
        NpcEntity blocker = world.npcAt(nx, ny);
        if (blocker != null) {
            return handleOccupiedMovementTarget(blocker, dx, dy, nx, ny);
        }
        resetPushPastIntent();
        movePlayerInto(nx, ny, movementModeLabel().toLowerCase(Locale.ROOT) + "s to " + nx + "," + ny + " over " + describeTile(world.tiles[nx][ny]) + ".");
        checkTrapAfterPlayerMove(nx, ny);
        return true;
    }

    void movePlayerInto(int nx, int ny, String reason) {
        int oldX = playerX, oldY = playerY;
        beginPlayerTileSlide(oldX, oldY, nx, ny);
        sounds.playFootstep(movementSurfaceKeyAt(nx, ny), 0, options);
        playerX = nx;
        playerY = ny;
        lookX = playerX;
        lookY = playerY;
        lookCursorActive = false;
        interactCursorActive = false;
        advanceMovementTurn(reason);
        enforceEntityOccupancy("player-move");
    }

    boolean handleOccupiedMovementTarget(NpcEntity blocker, int dx, int dy, int nx, int ny) {
        if (blocker == null) return false;
        if (blocker.isProtectedCleric()) {
            resetPushPastIntent();
            logEvent("Movement blocked: " + blocker.name + " holds the sanctuary space and cannot be shoved aside.");
            advanceTurn("stops short before " + blocker.name + ".");
            return false;
        }
        if (isHostileNpc(blocker)) {
            resetPushPastIntent();
            blocker.state = "Hostile";
            detectHostileIntent(blocker, 1);
            logEvent("BODY CHECK: " + blocker.name + " occupies the space and refuses your advance.");
            hostileAdjacentAttack(blocker);
            advanceTurn("is checked by " + blocker.name + " instead of entering the occupied tile.");
            enforceEntityOccupancy("hostile-body-check");
            return false;
        }
        Point shove = shoveDestinationFor(blocker, dx, dy);
        if (shove != null) {
            String old = blocker.x + "," + blocker.y;
            blocker.x = shove.x;
            blocker.y = shove.y;
            blocker.state = "Jostled";
            DebugLog.audit("ENTITY_SHOVE", "player=" + playerX + "," + playerY + " blocker=" + blocker.id + " from=" + old + " to=" + shove.x + "," + shove.y + " dir=" + dx + "," + dy);
            resetPushPastIntent();
            logEvent("SHOVE: " + blocker.name + " is pushed from " + old + " to " + shove.x + "," + shove.y + ". Two bodies do not get to occupy the same scrap of floor.");
            movePlayerInto(nx, ny, "shoves past " + blocker.name + " into " + nx + "," + ny + ".");
            return true;
        }
        if (tryPlayerPushPast(blocker, dx, dy, nx, ny)) return false;
        logEvent("Movement blocked: " + blocker.name + " is wedged in the way and there is nowhere legal to shove them.");
        advanceTurn("jostles against " + blocker.name + " in the crowded underhive.");
        enforceEntityOccupancy("failed-shove");
        return false;
    }

    boolean tryPlayerPushPast(NpcEntity blocker, int dx, int dy, int nx, int ny) {
        if (world == null || blocker == null) return false;
        if (blocker.isProtectedCleric() || isHostileNpc(blocker)) return false;
        if (blocker.x != nx || blocker.y != ny) return false;
        if (!world.walkable(playerX, playerY)) return false;
        String id = blocker.id == null ? blocker.name : blocker.id;
        boolean sameIntent = id.equals(pushPastIntentNpcId) && pushPastIntentX == nx && pushPastIntentY == ny && pushPastIntentDx == dx && pushPastIntentDy == dy;
        if (!sameIntent) {
            pushPastIntentNpcId = id;
            pushPastIntentX = nx;
            pushPastIntentY = ny;
            pushPastIntentDx = dx;
            pushPastIntentDy = dy;
            pushPastIntentCount = 1;
        } else pushPastIntentCount++;

        if (pushPastIntentCount < PUSH_PAST_REPEAT_THRESHOLD) {
            logEvent("Crowded passage: " + blocker.name + " cannot be shoved aside here. Press the same direction again to push past slowly instead of escalating to violence.");
            DebugLog.audit("ENTITY_PUSH_PAST_ARMED", "blocker=" + id + " at=" + nx + "," + ny + " dir=" + dx + "," + dy + " count=" + pushPastIntentCount);
            advanceTurn("leans into the crowding around " + blocker.name + " but does not force the passage yet.");
            enforceEntityOccupancy("push-past-armed");
            return true;
        }

        int oldPlayerX = playerX, oldPlayerY = playerY;
        blocker.x = oldPlayerX;
        blocker.y = oldPlayerY;
        blocker.state = "Jostled";
        playerX = nx;
        playerY = ny;
        lookX = playerX;
        lookY = playerY;
        lookCursorActive = false;
        interactCursorActive = false;
        int cost = Math.max(1, movementTurnCost()) + PUSH_PAST_EXTRA_TURNS;
        DebugLog.audit("ENTITY_PUSH_PAST", "playerFrom=" + oldPlayerX + "," + oldPlayerY + " playerTo=" + playerX + "," + playerY + " blocker=" + id + " blockerTo=" + blocker.x + "," + blocker.y + " cost=" + cost + " mode=" + movementModeLabel());
        logEvent("PUSH PAST: you spend extra time squeezing past " + blocker.name + " through the one-wide choke instead of starting a gunfight in a civilian corridor.");
        resetPushPastIntent();
        for (int i=0; i<cost; i++) {
            advanceTurn(i == 0 ? "pushes past " + blocker.name + " through a crowded one-wide passage. Push-past time cost " + cost + " turns." : "is still untangling from the corridor shuffle.");
        }
        enforceEntityOccupancy("push-past-complete");
        return true;
    }

    void resetPushPastIntent() {
        pushPastIntentNpcId = "";
        pushPastIntentX = Integer.MIN_VALUE;
        pushPastIntentY = Integer.MIN_VALUE;
        pushPastIntentDx = 0;
        pushPastIntentDy = 0;
        pushPastIntentCount = 0;
    }

    Point shoveDestinationFor(NpcEntity blocker, int dx, int dy) {
        if (world == null || blocker == null) return null;
        int[][] candidates;
        if (dx != 0) candidates = new int[][]{{dx,0},{0,-1},{0,1},{-dx,0}};
        else candidates = new int[][]{{0,dy},{-1,0},{1,0},{0,-dy}};
        for (int[] d : candidates) {
            int sx = blocker.x + d[0], sy = blocker.y + d[1];
            if (canNpcOccupyForShove(sx, sy)) return new Point(sx, sy);
        }
        return null;
    }

    boolean canNpcOccupyForShove(int x, int y) {
        return world != null && world.walkable(x, y) && world.npcAt(x, y) == null && !(x == playerX && y == playerY);
    }

    void enforceEntityOccupancy(String phase) {
        if (world == null || world.npcs == null) return;
        int repaired = world.repairNpcOccupancy(playerX, playerY, rng);
        if (repaired > 0) DebugLog.audit("ENTITY_OCCUPANCY_REPAIR", "phase=" + phase + " repaired=" + repaired + " player=" + playerX + "," + playerY + " zone=" + world.zoneType.label + " turn=" + turn);
    }

    void advanceMovementTurn(String reason) {
        int cost = movementTurnCost();
        markPlayerMovedThisTurn();
        DebugLog.audit("MOVEMENT_COST", "selected=" + movementModeLabel() + " active=" + activeMotionStateLabel() + " range=" + movementModeRange() + " cost=" + cost + " load=" + inventoryWeight() + " capacity=" + carryCapacity() + " reason=" + reason);
        for (int i=0; i<cost; i++) {
            advanceTurn(i == 0 ? reason + " [" + movementModeLabel() + "/active " + activeMotionStateLabel() + "]" + (cost > 1 ? " Movement cost " + cost + " turns." : "") : "continues moving under current pace and load.");
        }
        if (movementModeIndex == MOTION_SPRINT || movementModeIndex == MOTION_SNEAK) {
            fatigue = Math.min(20, fatigue + movementModeFatigueCost());
            if (movementModeIndex == MOTION_SPRINT) addSoundPing(playerX, playerY, movementModeNoiseWord());
            DebugLog.audit("MOVEMENT_STRAIN", "mode=" + movementModeLabel() + " fatigue=" + fatigue + " player=" + playerX + "," + playerY + " turn=" + turn);
        } else if (movementModeIndex == MOTION_RUN) addSoundPing(playerX, playerY, movementModeNoiseWord());
        maybeGrantSneakXp("movement");
    }

    void waitOneTurn() {

            singlePlayerSectorBridge.submitCommand(this, new WaitCommand(SinglePlayerSectorRuntimeBridge.LOCAL_PLAYER_ID));
            repaint();

    }

    void decayTemporaryHostilities() {
        for (Faction f : Faction.values()) {
            int v = temporaryHostileTurns.getOrDefault(f, 0);
            if (v > 0) temporaryHostileTurns.put(f, v - 1);
        }
    }

    void tickNearbyNpcs() {
        long profileStartNanos = System.nanoTime();
        if (world == null || world.npcs == null || world.npcs.isEmpty()) return;
        long schedulerStartNanos = System.nanoTime();
        boolean dangerClose = hostilePressureNearby();
        NpcTurnBudgetScheduler.Selection selection = npcScheduler.select(world.npcs, playerX, playerY, turn, activeMotionIsSneak(), dangerClose);
        profilingNpcSchedulerNanos += Math.max(0L, System.nanoTime() - schedulerStartNanos);
        profilingNpcSchedulerSelections++;
        profilingNpcSchedulerDeferred += Math.max(0, selection.farDeferred + selection.budgetLimited);
        if (selection.debtSuggested > 0) deferredSimulationDebt = Math.min(20000, deferredSimulationDebt + selection.debtSuggested);
        int activeTicks = 0;
        for (NpcEntity npc : selection.selected) {
            if (npc == null) continue;
            activeTicks++;
            int dist = Math.abs(npc.x - playerX) + Math.abs(npc.y - playerY);
            int npcOldX = npc.x, npcOldY = npc.y;
            ensureNpcCombatLoadout(npc);
            boolean suppressHostilePursuit = isHostileNpc(npc) && activeMotionIsSneak() && !hostileAwareOfPlayer(npc, dist);
            detectHostileIntent(npc, dist);
            if (!suppressHostilePursuit && maybeNpcRangedAttack(npc, dist)) { /* spent turn firing or reloading */ }
            else if (!suppressHostilePursuit) npc.tick(world, playerX, playerY, turn);
            else if (turn % 3 == 0) npc.wander(world, turn, playerX, playerY);
            if (npc.x != npcOldX || npc.y != npcOldY) beginNpcTileSlide(npc, npcOldX, npcOldY, npc.x, npc.y);
            int postDist = Math.abs(npc.x - playerX) + Math.abs(npc.y - playerY);
            if (isHostileNpc(npc) && postDist <= 1 && turn % 2 == 0) hostileAdjacentAttack(npc);
            if (dist <= 7 && turn % 5 == 0) maybeNpcAmbient(npc, dist);
        }
        if (activeTicks > 0 && npcScheduler.shouldAudit(turn)) {
            DebugLog.audit("NPC_SCHEDULER", selection.auditLine(turn, playerX, playerY, selection.selected.size()) + " dangerClose=" + dangerClose + " deferredDebt=" + deferredSimulationDebt);
        }
        profilingNpcTickNanos += Math.max(0L, System.nanoTime() - profileStartNanos);
        profilingNpcTickSamples++;
    }

    void processPersonnelReplacementQueue() {
        if (world == null || world.replacementQueue == null || world.replacementQueue.isEmpty()) return;
        int pruned = PersonnelPopulationApi.pruneReplacementQueueToCapacity(world);
        if (pruned > 0) DebugLog.audit("PERSONNEL_REPLACEMENT_PRUNE", "removed=" + pruned + " remaining=" + world.replacementQueue.size() + " zone=" + world.zoneType.label + " turn=" + turn);
        ArrayList<PersonnelReplacementRequest> matured = new ArrayList<>();
        for (PersonnelReplacementRequest req : world.replacementQueue) if (req.dueTurn <= turn) matured.add(req);
        if (matured.isEmpty()) return;
        for (PersonnelReplacementRequest req : matured) {
            Point p = PersonnelPopulationApi.spawnPointForReplacement(world, req, rng);
            if (p == null) continue;
            NpcEntity n = NpcEntity.create(req.faction, world.zoneType, p.x, p.y, rng);
            PersonnelPopulationApi.consumeReplacementSlot(world, req, rng);
            PersonnelProvenanceApi.assignReplacementProvenance(n, world, req, rng);
            world.npcs.add(n);
            logEvent("PERSONNEL REPLACEMENT: " + n.name + " arrives from " + req.source + " to refill " + req.faction.label + " losses. " + n.originSummary());
            DebugLog.audit("PERSONNEL_REPLACEMENT_ARRIVE", "dead=" + req.deadNpcId + " replacement=" + n.id + " faction=" + req.faction.label + " source=" + req.source + " due=" + req.dueTurn + " turn=" + turn + " zone=" + world.zoneType.label);
        }
        world.replacementQueue.removeAll(matured);
    }

    void maybeNpcAmbient(NpcEntity npc, int dist) {
        if (npc == null || rng.nextDouble() > 0.22) return;
        String line;
        if (npc.isAnimalActor()) {
            line = npc.name + (npc.isPetActor() ? " pads after its keeper, claws faint on the deck." : " scrapes, sniffs, or chitters somewhere in the local gloom.");
            logEvent(line);
            addSoundPing(npc.x, npc.y, "animal");
            return;
        }
        switch(npc.faction) {
            case MINISTORUM: line = npc.name + " murmurs a lawful prayer beside candle smoke and hungry pilgrims."; break;

    boolean isHostileNpc(NpcEntity npc) {
        if (npc == null) return false;
        if (npc.isUntargetableAnchor()) return false;
        if ("Hostile".equalsIgnoreCase(npc.state)) return true;
        if (npc.faction == Faction.MUTANT || npc.faction == Faction.CULTIST || npc.faction == Faction.HERETIC || npc.faction == Faction.ROGUE_MACHINE) return true;
        if (npc.faction == Faction.BANDIT || npc.faction.name().startsWith("GANGER")) return true;
        return temporaryHostileTurns.getOrDefault(npc.faction, 0) > 0;
    }

    String hostileIntentBark(NpcEntity npc, int dist) {
        if (npc == null) return "Something notices you.";
        if (npc.isAnimalActor()) return npc.name + " notices you and breaks into animal violence.";
        switch(npc.faction) {
            case MUTANT: return npc.name + " sees you and surges forward, wet claws scraping for purchase.";

            case BANDIT: case GANGER_IRON_RATS: case GANGER_BLACK_SUMP: case GANGER_CANDLE_JACKS: case GANGER_RED_GRIN: case GANGER_CHAIN_SAINTS: case GANGER_ASH_MARKET: case GANGER_WIRE_WOLVES: case GANGER_DROWNED_9TH:
                return npc.name + " clocks your movement, reaches for a weapon, and commits to violence.";

            default: return npc.name + " turns hostile and closes distance.";
        }
    }

    boolean hostileAwareOfPlayer(NpcEntity npc, int dist) {
        if (npc == null || !isHostileNpc(npc)) return false;
        int detectPenalty = detectionObstructionPenalty(npc.x, npc.y, playerX, playerY, true) + detectionLightPenalty(npc.x, npc.y, playerX, playerY);
        boolean normalAwareness = dist <= 2 || ((dist <= visionRange() && isVisible(npc.x, npc.y)) || dist <= Math.max(4, hearingRange()/2)) && detectPenalty < 38;
        if (!activeMotionIsSneak()) return normalAwareness;
        if (dist <= 1) return rng.nextInt(100) < 70;
        int chance = 0;
        if (isVisible(npc.x, npc.y)) chance += 12 + Math.max(0, detailVisionRange() - dist) * 3;
        if (dist <= Math.max(2, hearingRange()/3)) chance += 10;
        if (fatigue > 12) chance += 6;
        chance += doorwayExposureBonus(playerX, playerY);
        chance += doorwayExposureBonus(npc.x, npc.y) / 2;
        chance -= detectPenalty;
        chance = Math.max(3, Math.min(75, chance));
        int roll = rng.nextInt(100);
        DebugLog.audit("STEALTH_DETECTION", "npc=" + npc.id + " dist=" + dist + " roll=" + roll + " chance=" + chance + " active=" + activeMotionStateLabel() + " vision=" + visionRange() + " hearing=" + hearingRange() + " turn=" + turn);
        return roll < chance;
    }

    void suspiciousStealthBark(NpcEntity npc, int dist) {
        if (npc == null || !activeMotionIsSneak() || dist > 8) return;
        if (npc.suspiciousBarkCooldown > 0) { npc.suspiciousBarkCooldown--; return; }
        String[] barks = {"must have been the wind", "that was strange", "what was that?", "Hold. Did you hear something?", "No. Nothing. Keep walking."};
        String bark = barks[Math.abs(Objects.hash(npc.id, turn, dist)) % barks.length];
        npc.suspiciousBarkCooldown = 6 + Math.max(0, dist);
        logEvent("SUSPICION: " + npc.name + " mutters, \"" + bark + "\" but has not fixed on you.");
        maybeGrantSneakXp("evaded detection by " + npc.name);
    }

    void maybeGrantSneakXp(String source) {
        if (!activeMotionIsSneak() || world == null || world.npcs == null || turn == lastSneakXpTurn) return;
        for (NpcEntity npc : world.npcs) {
            if (npc == null || !isHostileNpc(npc)) continue;
            int dist = Math.abs(npc.x-playerX)+Math.abs(npc.y-playerY);
            if (dist <= 8 && !npc.hostileBarked) {
                lastSneakXpTurn = turn;
                gainXp("Stealth", 1, "successful sneaking action — " + source);
                return;
            }
        }
    }

    void detectHostileIntent(NpcEntity npc, int dist) {
        if (npc == null || !isHostileNpc(npc)) return;
        boolean aware = hostileAwareOfPlayer(npc, dist);
        if (!aware) { suspiciousStealthBark(npc, dist); return; }
        if (!"Hostile".equalsIgnoreCase(npc.state)) {
            npc.state = "Hostile";
            npc.hostileBarked = true;
            logEvent("HOSTILE INTENT: " + hostileIntentBark(npc, dist));
            addSoundPing(npc.x, npc.y, "threat");
            noteCombatMusicActivity(isGovernorCombatNpc(npc), "hostile intent acquired");
        } else if (!npc.hostileBarked) {
            npc.hostileBarked = true;
            logEvent("HOSTILE INTENT: " + hostileIntentBark(npc, dist));
            noteCombatMusicActivity(isGovernorCombatNpc(npc), "hostile intent bark");
        }
    }

    NpcEntity nearestAttackTarget() {
        if (world == null || world.npcs == null) return null;
        NpcEntity best = null; int bestDist = 999;
        for (NpcEntity npc : world.npcs) {
            if (!isHostileNpc(npc) || npc.isUntargetableAnchor()) continue;
            int d = Math.abs(npc.x-playerX) + Math.abs(npc.y-playerY);
            if (d <= 1 && d < bestDist) { best = npc; bestDist = d; }
        }
        return best;
    }

    String bestWeaponName() {

        }
        return best;
    }

    boolean consumeLoadedShotsForAttack(TargetingSolution ts) {
        if (ts == null || !ts.ranged) return true;
        int need = Math.max(1, ts.ammoCost);
        int have = loadedShotsForWeapon(ts.weaponName);
        if (have < need) {
            logEvent("RELOAD REQUIRED: " + ts.weaponName + " has " + have + "/" + magazineCapacityForWeapon(ts.weaponName) + " shots loaded. Press X to reload using " + ammoItemForWeapon(ts.weaponName) + ".");
            sounds.playReloadEffect(options);
            return false;
        }
        setLoadedShotsForWeapon(ts.weaponName, have - need);
        return true;
    }

    void reloadCurrentRangedWeapon() {
        String weapon = bestWeaponName();
        if (weapon == null || !ItemCatalog.isFirearmLike(weapon)) { logEvent("RELOAD: no carried ranged weapon is selected as the best weapon."); return; }
        String ammo = ammoItemForWeapon(weapon);
        int cap = magazineCapacityForWeapon(weapon);
        int loaded = loadedShotsForWeapon(weapon);
        if (cap <= 0 || ammo.isBlank()) { logEvent("RELOAD: " + weapon + " has no known reload profile yet."); return; }
        if (loaded >= cap) { logEvent("RELOAD: " + weapon + " is already loaded (" + loaded + "/" + cap + ")."); return; }
        if (!removeInventoryOne(ammo)) { logEvent("RELOAD FAILED: need " + ammo + " to load " + weapon + ". Current load " + loaded + "/" + cap + "."); return; }
        setLoadedShotsForWeapon(weapon, cap);
        sounds.playReloadEffect(options);
        logEvent("RELOAD: " + active.name + " loads " + weapon + " with " + ammo + ". Loaded " + cap + "/" + cap + " shots.");
        advanceTurn("reloads " + weapon + ".");
    }

    void throwSelectedExplosiveAtCursor() {
        if (world == null || active == null) return;
        String item = selectedExplosiveItem == null || selectedExplosiveItem.isBlank() ? firstCarriedExplosiveItem() : selectedExplosiveItem;
        if (item.isBlank()) { logEvent("EXPLOSIVE: no carried explosive to use."); return; }
        ExplosiveProfile ep = ExplosiveProfile.forItem(item);
        int tx = combatX < 0 ? playerX + lastMoveDx : combatX;
        int ty = combatY < 0 ? playerY + lastMoveDy : combatY;
        tx = Math.max(0, Math.min(world.w-1, tx));
        ty = Math.max(0, Math.min(world.h-1, ty));
        int dist = Math.abs(tx-playerX) + Math.abs(ty-playerY);
        if (dist > ep.throwRange) { logEvent("EXPLOSIVE RANGE: " + item + " target distance " + dist + " exceeds range " + ep.throwRange + "."); return; }
        if (!removeInventoryOne(item)) { logEvent("EXPLOSIVE: " + item + " is no longer in inventory."); return; }
        Point land = explosiveLandingPoint(item, tx, ty, dist);
        if (ep.plantable && dist <= 1 && !item.toLowerCase(Locale.ROOT).contains("grenade")) {
            plantExplosiveAt(land.x, land.y, item, true);
            for (int i=0;i<Math.max(1, ep.disarmTurns);i++) advanceTurn(i==0 ? "plants " + item + " as a passive explosive." : "finishes arming " + item + ".");
        } else {
            MapObjectState m = MapObjectState.thrownExplosive(land.x, land.y, item, world.zoneType, turn + Math.max(1, ep.fuseTurns), world.tiles[land.x][land.y]);
            world.mapObjects.add(m);
            world.tiles[land.x][land.y] = '*';
            explosionAnimations.add(ExplosionAnimation.grenade(playerX, playerY, land.x, land.y, item, System.currentTimeMillis()));
            sounds.playWeaponEffect(item, true, options);
            logEvent("GRENADE THROWN: " + active.name + " throws " + item + " toward " + tx + "," + ty + "; it bounces to " + land.x + "," + land.y + " as a blinking * and will detonate at turn " + m.cooldownUntilTurn + ".");
            advanceTurn("throws " + item + ".");
        }
        explosiveTargetingMode = false;
        selectedExplosiveItem = "";
    }

    Point explosiveLandingPoint(String item, int tx, int ty, int dist) {
        ExplosiveProfile ep = ExplosiveProfile.forItem(item);
        int chance = Math.max(15, Math.min(90, stat("Agility",5) * 5 + stat("Firearms",5) * 3 + (activeMotionStateIndex == MOTION_STATIONARY ? 14 : activeMotionStateIndex == MOTION_WALK ? 7 : -6) - dist * 4));
        if (rng.nextInt(100) < chance) return new Point(tx, ty);
        int dx = Integer.compare(tx-playerX, 0), dy = Integer.compare(ty-playerY, 0);
        int sideX = dy, sideY = -dx;
        int scatter = 1 + rng.nextInt(Math.max(1, Math.min(3, ep.radius)));
        if (rng.nextBoolean()) scatter = -scatter;
        return new Point(Math.max(0, Math.min(world.w-1, tx + sideX * scatter)), Math.max(0, Math.min(world.h-1, ty + sideY * scatter)));
    }

    void plantExplosiveAt(int x, int y, String item, boolean playerPlaced) {
        if (world == null || !world.inBounds(x,y)) return;
        if (world.npcAt(x,y) != null || (x == playerX && y == playerY)) { logEvent("EXPLOSIVE PLANTING blocked: occupied tile."); addInventoryItem(item, ItemProvenanceRecord.found(item, world, turn, "unplanted explosive returned to hand")); return; }
        ExplosiveProfile ep = ExplosiveProfile.forItem(item);
        MapObjectState m = MapObjectState.plantedExplosive(x, y, item, world.zoneType, turn, world.tiles[x][y], playerPlaced ? "player" : "faction");
        m.vendCount = ep.complexity;
        world.mapObjects.add(m);
        world.tiles[x][y] = '!';
        logEvent("EXPLOSIVE PLANTED: " + item + " armed at " + x + "," + y + ". Good sensory stats may spot it; INTERACT can attempt disarm and recovery.");
        DebugLog.audit("EXPLOSIVE_PLANTED", "item=" + item + " at=" + x + "," + y + " player=" + playerPlaced + " turn=" + turn);
    }

    void tickThrownAndPlantedExplosives() {
        if (world == null || world.mapObjects == null || world.mapObjects.isEmpty()) return;
        ArrayList<MapObjectState> copy = new ArrayList<>(world.mapObjects);
        for (MapObjectState m : copy) {
            if (m == null) continue;
            if ("thrown-explosive".equals(m.type) && turn >= m.cooldownUntilTurn) { detonateExplosiveObject(m, "fuse expiry"); continue; }
            if ("planted-explosive".equals(m.type)) {
                if ((playerX == m.x && playerY == m.y) || Math.abs(playerX-m.x)+Math.abs(playerY-m.y) <= 1) maybeTriggerPlantedExplosive(m, true, null);
                if (world == null || !world.mapObjects.contains(m)) continue;
                for (NpcEntity n : new ArrayList<>(world.npcs)) if (Math.abs(n.x-m.x)+Math.abs(n.y-m.y) <= 0) { maybeTriggerPlantedExplosive(m, false, n); break; }
            }
        }
    }

    void maybeTriggerPlantedExplosive(MapObjectState m, boolean player, NpcEntity npc) {
        if (m == null || !"planted-explosive".equals(m.type)) return;
        ExplosiveProfile ep = ExplosiveProfile.forItem(m.stockState);
        int sensory = player ? stat("Vision",5) + stat("Hearing",5) + stat("Intellect",5) : (npc == null ? 8 : npc.intellect + npc.idleBias + 4);
        int difficulty = 12 + ep.complexity * 3;
        int roll = rng.nextInt(20) + sensory;
        if (roll >= difficulty) {
            if (player) logEvent("TRAP SPOTTED: you catch the telltale wire, blinking rune, or bad smell of " + ep.name + " before setting it off. Interact to disarm.");
            else if (isVisible(npc.x, npc.y)) logEvent("TRAP EVADED: " + npc.name + " hesitates around a planted " + ep.name + ".");
            return;
        }
        detonateExplosiveObject(m, player ? "player triggered" : (npc == null ? "actor triggered" : npc.name + " triggered"));
    }

    void disarmExplosiveAt(int tx, int ty) {
        MapObjectState m = mapObjectAt(tx, ty, "planted-explosive");
        if (m == null) { logEvent("DISARM: no planted explosive selected."); return; }
        ExplosiveProfile ep = ExplosiveProfile.forItem(m.stockState);
        int turns = Math.max(1, ep.disarmTurns);
        int score = stat("Intellect",5) + stat("Mechanics",5) + stat("Vision",5) / 2;
        int difficulty = 12 + ep.complexity * 4;
        int roll = rng.nextInt(20) + score;
        for (int i=0;i<turns;i++) advanceTurn(i==0 ? "attempts to disarm " + ep.name + "." : "continues the disarm work.");
        if (roll >= difficulty) {
            restoreMapObjectUnderlyingTile(m);
            world.mapObjects.remove(m);
            addInventoryItem(ep.name, ItemProvenanceRecord.found(ep.name, world, turn, "disarmed and recovered planted explosive"));
            logEvent("DISARM SUCCESS: recovered " + ep.name + ". Roll " + roll + " vs " + difficulty + ".");
        } else {
            logEvent("DISARM FAILED: " + ep.name + " bites back. Roll " + roll + " vs " + difficulty + ".");
            detonateExplosiveObject(m, "failed disarm");
        }
    }

    void detonateExplosiveAt(int cx, int cy, ExplosiveProfile ep, String cause) {
        if (world == null || ep == null) return;
        sounds.playWeaponEffect(ep.name, true, options);
        explosionAnimations.add(ExplosionAnimation.blast(cx, cy, ep.radius, ep.name, System.currentTimeMillis()));
        while (explosionAnimations.size() > 10) explosionAnimations.remove(0);
        logEvent("EXPLOSION: " + ep.name + " detonates at " + cx + "," + cy + " from " + cause + ". Radius " + ep.radius + ", damage " + ep.damage + ".");
        ArrayList<NpcEntity> killed = new ArrayList<>();
        for (NpcEntity n : new ArrayList<>(world.npcs)) {
            int d = Math.max(Math.abs(n.x-cx), Math.abs(n.y-cy));
            if (d <= ep.radius) {
                if (n.isUntargetableAnchor()) {
                    if (isVisible(n.x,n.y)) logEvent("BLAST REFUSED: " + n.name + " is a protected continuity anchor and remains unharmed.");
                    continue;
                }
                int dmg = Math.max(1, ep.damage - d * Math.max(2, ep.damage / Math.max(2, ep.radius + 2)) - factionArmor(n));
                n.hp -= dmg;
                if (isVisible(n.x,n.y)) logEvent("BLAST HIT: " + n.name + " takes " + dmg + " from " + ep.name + ". HP " + Math.max(0,n.hp) + ".");
                if (n.hp <= 0) killed.add(n);
            }
        }
        for (NpcEntity n : killed) { String corpseLine = createCorpseLootContainer(n, ep.name + " explosion"); world.npcs.remove(n); logEvent("BLAST KILL: " + n.name + " is dropped by the explosion. " + corpseLine); }
        int pd = Math.max(Math.abs(playerX-cx), Math.abs(playerY-cy));
        if (pd <= ep.radius) {
            int shield = shieldMitigationPercentAt(playerX, playerY, equippedClothing == null ? Faction.NONE : equippedClothing.alignedFaction);
            int dmg = Math.max(1, ((ep.damage - pd * 3) * Math.max(20, 100 - shield)) / 100);
            applyPlayerBodyDamage(dmg, randomBodyHitLocation(), ep.name + " blast", ep.name);
        }
        for (int x=Math.max(0,cx-ep.radius); x<=Math.min(world.w-1,cx+ep.radius); x++) for (int y=Math.max(0,cy-ep.radius); y<=Math.min(world.h-1,cy+ep.radius); y++) {
            int d = Math.max(Math.abs(x-cx), Math.abs(y-cy));
            if (d <= ep.radius && rng.nextInt(100) < 70 - d*10) applyStrayImpactDamage(x, y, Math.max(1, ep.damage/3 - d*2), ep.name + " blast");
        }
        if (ep.smoke) seedSmokeCloud(cx, cy, ep.radius);
        DebugLog.audit("EXPLOSIVE_DETONATE", "name=" + ep.name + " at=" + cx + "," + cy + " radius=" + ep.radius + " cause=" + cause + " turn=" + turn);
    }

    void seedSmokeCloud(int cx, int cy, int radius) {

        }
        logEvent("SMOKE: a dirty occluding cloud boils out. Full smoke vision math is queued for the lighting/occlusion pass.");
    }

    int weaponRange(String weapon) {
        if (weapon == null || !ItemCatalog.isFirearmLike(weapon)) return 1;
        String low = ItemQuality.stripManufacturingIdentity(weapon).toLowerCase(Locale.ROOT);
        if (low.contains("shotgun")) return 5;
        if (low.contains("flamer") || low.contains("melta")) return 4;
        if (low.contains("pistol") || low.contains("revolver")) return 7;
        if (low.contains("stubber") || low.contains("carbine") || low.contains("webber")) return 9;
        if (low.contains("rifle") || low.contains("las") || low.contains("mass-Reactive Carbine")) return 12;
        if (low.contains("autocannon") || low.contains("long")) return 14;
        return 8;
    }

    TargetingSolution targetingSolutionAt(int tx, int ty) {
        TargetingSolution ts = new TargetingSolution();
        ts.targetX = tx; ts.targetY = ty;
        if (world == null || active == null) { ts.canFire=false; ts.summary="No active tactical world."; return ts; }
        if (!world.inBounds(tx, ty)) { ts.canFire=false; ts.summary="Target outside world bounds."; return ts; }
        ts.weaponName = bestWeaponName();
        ts.ranged = ts.weaponName != null && ItemCatalog.isFirearmLike(ts.weaponName);
        if (ts.weaponName == null) ts.weaponName = "Bare hands";
        ts.range = weaponRange(ts.weaponName);
        ts.distance = Math.abs(tx-playerX) + Math.abs(ty-playerY);
        ts.targetNpc = world.npcAt(tx, ty);
        ts.fireMode = fireModeLabel();
        ts.loadedShots = loadedShotsForWeapon(ts.weaponName);
        ts.magazineCapacity = magazineCapacityForWeapon(ts.weaponName);
        ts.ammoCost = fireModeAmmoCost(ts.weaponName);
        ts.ammoItem = ammoItemForWeapon(ts.weaponName);
        ts.needsReload = ts.ranged && ts.loadedShots < Math.max(1, ts.ammoCost);
        ts.shieldReduction = shieldMitigationPercentAt(tx, ty, ts.targetNpc == null ? Faction.NONE : ts.targetNpc.faction);
        ts.coverPenalty = rangedCoverPenalty(playerX, playerY, tx, ty, true, ts);
        ts.darknessPenalty = ts.ranged ? rangedDarknessPenalty(playerX, playerY, tx, ty) : 0;
        int skill = ts.ranged ? stat("Firearms", 5) : stat("Melee", 5);
        if (ts.ranged && !hasUsableWeaponLimb()) { ts.canFire=false; ts.summary="Arms/hands too damaged to aim a ranged weapon."; ts.hitPercent=0; return ts; }
        if (ts.ranged && ts.needsReload) { ts.canFire=false; ts.summary="Reload required: " + ts.weaponName + " has " + ts.loadedShots + "/" + ts.magazineCapacity + " loaded; needs " + Math.max(1,ts.ammoCost) + " for " + ts.fireMode + ". Press X to reload using " + ts.ammoItem + "."; ts.hitPercent=0; return ts; }
        if (!ts.ranged && ts.distance > 1) { ts.canFire=false; ts.summary="Melee attacks require an adjacent tile."; ts.hitPercent=0; return ts; }
        if (ts.ranged && ts.distance > ts.range) { ts.canFire=false; ts.summary="Out of range: " + ts.distance + "/" + ts.range + "."; ts.hitPercent=0; return ts; }
        if (ts.targetNpc == null && !ts.ranged) { ts.canFire=false; ts.summary="No adjacent entity to strike."; ts.hitPercent=0; return ts; }
        int distancePenalty = ts.ranged ? Math.max(0, ts.distance - 2) * 4 : 0;
        int targetMotionDefense = ts.targetNpc == null ? 0 : npcEvasionModifier(ts.targetNpc);
        int targetArmor = ts.targetNpc == null ? 0 : factionArmor(ts.targetNpc);
        int base = ts.ranged ? 52 : 65;
        int hit = base + skill * 5 + playerAttackMotionSkillModifier() + fireModeAccuracyModifier() - distancePenalty - ts.coverPenalty - ts.darknessPenalty - targetMotionDefense - targetArmor * 2;
        if (ts.targetNpc == null && ts.ranged && !ts.coverHardBlocked) {
            // Directly attacking empty floor, wall, or an object is not a contest
            // against a dodging actor. If the line is clear and the target is in
            // range, the shot should be highly reliable so terrain/object attacks
            // are usable instead of displaying a miserable 0% on nothing.
            hit = Math.max(hit, 92 + Math.max(0, fireModeAccuracyModifier() / 3));
        } else if (ts.targetNpc == null) hit -= ts.ranged ? 8 : 30;
        if (ts.coverHardBlocked) hit = 0;
        ts.hitPercent = Math.max(0, Math.min(95, hit));
        ts.canFire = ts.hitPercent > 0 && (ts.ranged || ts.targetNpc != null);
        String targetLabel = ts.targetNpc == null ? ("direct terrain/object " + describeTile(world.tiles[tx][ty])) : ts.targetNpc.name;
        ts.summary = (ts.ranged ? "Ranged" : "Melee") + " target " + tx + "," + ty + " | " + targetLabel + " | hit " + ts.hitPercent + "% | " + (ts.ranged ? ("ammo " + ts.loadedShots + "/" + ts.magazineCapacity + " cost " + ts.ammoCost + " | light -" + ts.darknessPenalty + " | ") : "") + ts.coverSummary;
        return ts;
    }

    int npcEvasionModifier(NpcEntity npc) {
        if (npc == null) return 0;
        switch(npc.activeMotionStateIndex) {
            case MOTION_SNEAK: return 6;
            case MOTION_SPRINT: return 5;
            case MOTION_RUN: return 3;
            case MOTION_WALK: return 1;
            default: return 0;
        }
    }

    int shieldMitigationPercentAt(int tx, int ty, Faction defenderFaction) {
        if (world == null || baseObjects == null || baseObjects.isEmpty()) return 0;
        int room = world.inBounds(tx, ty) ? world.roomIds[tx][ty] : -9999;
        int shield = 0;
        for (BaseObject obj : baseObjects) {
            if (obj == null || obj.symbol != 'H' || !obj.armed || obj.charges <= 0) continue;
            int oroom = world.inBounds(obj.x, obj.y) ? world.roomIds[obj.x][obj.y] : -9998;
            boolean sameRoom = room >= 0 && room == oroom;
            boolean nearby = Math.abs(obj.x - tx) + Math.abs(obj.y - ty) <= 5;
            if (sameRoom || nearby) shield += Math.max(2, obj.cover);
        }
        return Math.max(0, Math.min(60, shield * 4));
    }

    int rangedCoverPenalty(int x0, int y0, int x1, int y1, boolean includeEntities, TargetingSolution ts) {
        int penalty = 0;
        ArrayList<Point> pts = linePoints(x0, y0, x1, y1);
        int total = Math.max(1, pts.size() - 1);
        ArrayList<String> blockers = new ArrayList<>();
        for (int i=1; i<pts.size(); i++) {
            Point p = pts.get(i);
            boolean targetTile = (p.x == x1 && p.y == y1);
            if (world == null || !world.inBounds(p.x,p.y)) continue;
            int base = 0;
            char ch = world.tiles[p.x][p.y];
            BaseObject obj = baseObjectAt(p.x,p.y);
            NpcEntity n = world.npcAt(p.x,p.y);
            if (!targetTile && blocksLineOfSight(ch)) base = (ch == '#' || InterstitialInfrastructureApi.isInterstitialSolid(ch) || ch == InterstitialInfrastructureApi.VOID_SPACE) ? 100 : 45;
            else if (!targetTile && isDoorTile(ch)) base = 32;
            if (!targetTile && obj != null) base = Math.max(base, 16 + PassiveDefenseEffectsAuthority.coverPenalty(obj));
            else if (!targetTile && tileProvidesSoftCover(ch)) base = Math.max(base, 12);
            if (includeEntities && !targetTile && n != null) base = Math.max(base, isHostileNpc(n) ? 22 : 28);
            if (base > 0) {
                double progress = i / (double) total;
                int scaled = (int)Math.round(base * (0.35 + 0.65 * progress));
                penalty += scaled;
                if (blockers.size() < 4) blockers.add(blockerLabel(ch, obj, n) + "@" + p.x + "," + p.y + " -" + scaled);
                if (base >= 100) {
                    penalty = Math.max(penalty, 100);
                    if (ts != null) ts.coverHardBlocked = true;
                }
            }
        }
        if (ts != null) ts.coverSummary = blockers.isEmpty() ? "clear firing lane" : String.join("; ", blockers);
        return Math.max(0, Math.min(140, penalty));
    }

    boolean tileProvidesSoftCover(char ch) {
        return ch == 's' || ch == 'c' || ch == 'u' || ch == 'w' || ch == 'N' || ch == 'B' || ch == 'T' || ch == 'H' || ch == 'M' || ch == 'G' || ch == 'k' || ch == 'q' || ch == 'd' || ch == 'S' || ch == 'R' || ch == 'W' || ch == 'D' || ch == 'Y' || ch == 'J' || ch == 'U' || ch == 'Z' || ch == 'P' || ch == 'a' || ch == 'p' || ch == 'g' || ch == 'x' || ch == '1' || ch == '2' || ch == '3' || ch == '4' || ch == '5';
    }

    String blockerLabel(char ch, BaseObject obj, NpcEntity n) {
        if (n != null) return "entity " + n.name;
        if (obj != null) return "object " + obj.name;
        if (isDoorTile(ch)) return "door " + ch;
        if (blocksLineOfSight(ch)) return "hard cover " + ch;
        return "feature " + describeTile(ch);
    }

    ArrayList<Point> linePoints(int x0, int y0, int x1, int y1) {
        ArrayList<Point> pts = new ArrayList<>();
        int dx = Math.abs(x1 - x0), dy = Math.abs(y1 - y0);
        int sx = x0 < x1 ? 1 : -1;
        int sy = y0 < y1 ? 1 : -1;
        int err = dx - dy;
        int x = x0, y = y0;
        while (true) {
            pts.add(new Point(x,y));
            if (x == x1 && y == y1) break;
            int e2 = 2 * err;
            if (e2 > -dy) { err -= dy; x += sx; }
            if (e2 < dx) { err += dx; y += sy; }
            if (pts.size() > 256) break;
        }
        return pts;
    }



    Point strayImpactPoint(TargetingSolution ts) {
        if (ts == null || world == null) return new Point(playerX, playerY);
        int dx = Integer.compare(ts.targetX - playerX, 0);
        int dy = Integer.compare(ts.targetY - playerY, 0);
        int sideX = dy;
        int sideY = -dx;
        int range = Math.max(1, ts.distance);
        int scatter = 1 + rng.nextInt(Math.max(1, Math.min(3, 1 + range / 4)));
        if (rng.nextBoolean()) scatter = -scatter;
        int forward = rng.nextInt(3) - 1;
        int ix = ts.targetX + sideX * scatter + dx * forward;
        int iy = ts.targetY + sideY * scatter + dy * forward;
        ix = Math.max(0, Math.min(world.w-1, ix));
        iy = Math.max(0, Math.min(world.h-1, iy));
        return new Point(ix, iy);
    }

    }

    void tickCorpseEcology() {
        if (world == null || world.mapObjects == null || world.mapObjects.isEmpty()) return;
        ArrayList<MapObjectState> copy = new ArrayList<>(world.mapObjects);
        for (MapObjectState m : copy) {
            if (m == null || !"corpse-container".equals(m.type)) continue;
            if (m.cooldownUntilTurn <= 0) m.cooldownUntilTurn = turn + (isPublicCorpseLocation(m) ? 280 : 900);
            int witnesses = nearbyWitnessCount(m.x, m.y, 8);
            if (isPublicCorpseLocation(m) && witnesses >= 3 && turn >= m.cooldownUntilTurn - 180) { authorityCollectCorpse(m, witnesses); continue; }
            if (turn >= m.cooldownUntilTurn) decayHiddenCorpse(m);
        }
    }

    int nearbyWitnessCount(int x, int y, int range) {
        if (world == null) return 0;
        int c = 0;
        for (NpcEntity n : world.npcs) if (n != null && Math.abs(n.x-x)+Math.abs(n.y-y) <= range && !isHostileNpc(n)) c++;
        return c;
    }

    java.util.List<String> drainContainerItemNames(String cid, int max) {
        ArrayList<String> out = new ArrayList<>();
        ContainerRecord c = itemContainers.get(cid);
        if (c == null) return out;
        int limit = Math.max(0, max);
        while (!c.itemInstanceIds.isEmpty() && out.size() < limit) {
            String iid = c.itemInstanceIds.remove(0);
            ItemInstance inst = itemInstances.remove(iid);
            if (inst != null && inst.displayName != null && !inst.displayName.isBlank()) out.add(decayCorpseItemName(inst.displayName));
        }
        return out;
    }

    String decayCorpseItemName(String item) {
        if (item == null || item.isBlank()) return "Scrap bit";

        if (low.contains("scrap") || low.contains("junk") || low.contains("detritus")) return "Scrap bit";

        String base = ItemQuality.stripQuality(item);
        return ItemQuality.NAMES[Math.max(0, Math.min(ItemQuality.NAMES.length-1, tier))] + " " + base;
    }

        logEvent("CORPSE DECAY: an unnoticed body has become a skeleton and a pile of degraded belongings. Quality is not improved by rot. Stunning revelation.");
        DebugLog.audit("CORPSE_DECAY", "id=" + m.id + " decayedItems=" + decayed.size() + " turn=" + turn);
    }

    void tickFactionWallRepairAndHiveQuake() {
        if (world == null || terrainIntegrity.isEmpty()) return;
        if (turn % 37 != 0) return;
        for (String key : new ArrayList<>(terrainIntegrity.keySet())) {
            int hp = terrainIntegrity.getOrDefault(key, 1);
            if (hp > 0) continue;
            String[] a = key.split(",");
            if (a.length != 2) continue;
            int x, y;
            try { x = Integer.parseInt(a[0]); y = Integer.parseInt(a[1]); } catch(Exception ex) { continue; }
            if (!world.inBounds(x,y)) continue;
            char ch = world.tiles[x][y];
            if (InterstitialInfrastructureApi.isInterstitialSolid(ch) || ch == '%' || ch == '&' || ch == '^' || ch == '8' || ch == '0') {
                if (rng.nextInt(100) < 35) { world.tiles[x][y] = '*'; terrainIntegrity.remove(key); logEvent("ARCOLOGY-QUAKE FILL: overhead debris slumps into an intra-wall breach at " + x + "," + y + ". The arcology repairs itself by dropping masonry on the problem."); }
            } else if (ch == '*' || ch == '#' || isDoorTile(ch)) {
                Faction f = factionControllingTile(x,y);
                if (f != Faction.NONE && rng.nextInt(100) < 45) { world.tiles[x][y] = '#'; terrainIntegrity.put(key, defaultTerrainIntegrity('#')); logEvent("FACTION REPAIR: " + f.label + " workers draw supplies from nearby stock and patch damaged wall/door structure at " + x + "," + y + "."); }
            }
        }
    }

    Faction factionControllingTile(int x, int y) {
        if (world == null) return Faction.NONE;
        int rid = world.roomIdAt(x,y);
        if (rid >= 0 && rid < world.roomFactions.size()) return world.roomFactions.get(rid);
        return world.zoneType == ZoneType.ARBITES_PRECINCT_EDGE ? Faction.CIVIC_WARDENS : (world.zoneType == ZoneType.IMPERIAL_GUARD_BILLET ? Faction.IMPERIAL_GUARD : Faction.NONE);
    }

    String applyStrayImpactDamage(int x, int y, int dmg, String source) {
        if (world == null || !world.inBounds(x,y)) return "The stray impact vanishes outside the tactical grid.";
        NpcEntity n = world.npcAt(x,y);
        if (n != null && !(x == playerX && y == playerY)) {
            if (n.isUntargetableAnchor()) return "Stray impact is refused by protected civic continuity: " + n.name + " cannot be harmed or targeted.";
            int armor = factionArmor(n);
            int ndmg = Math.max(1, dmg - armor);
            n.hp -= ndmg;
            addSoundPing(x,y,"impact");
            if (n.hp <= 0) {
                String corpseLine = createCorpseLootContainer(n, "stray round from " + source);
                world.npcs.remove(n);
                return "Stray impact hits " + n.name + " for " + ndmg + " and drops them. " + corpseLine;
            }
            return "Stray impact clips " + n.name + " for " + ndmg + " damage.";
        }
        BaseObject obj = baseObjectAt(x,y);
        if (obj != null) {
            obj.integrity = Math.max(0, obj.integrity - Math.max(1,dmg));
            addSoundPing(x,y,"impact");
            return "Stray impact hits object " + obj.name + " at " + x + "," + y + "; integrity now " + obj.integrity + ".";
        }
        char ch = world.tiles[x][y];
        if (isProtectedTransitionTile(ch)) return "Stray impact sparks off protected transit infrastructure; level-transition anchors are not destructible.";
        if (ch == InterstitialInfrastructureApi.VOID_SPACE) return "The shot disappears into the intrahive abyss.";
        String key = terrainKey(x,y);
        int hp = terrainIntegrity.getOrDefault(key, defaultTerrainIntegrity(ch));
        hp = Math.max(0, hp - Math.max(1,dmg));
        terrainIntegrity.put(key, hp);
        addSoundPing(x,y,"impact");
        if (hp <= 0 && tileCanBeDamagedIntoDebris(ch)) {
            world.tiles[x][y] = '*';
            return "Stray impact destroys " + describeTile(ch) + " at " + x + "," + y + ", leaving debris.";
        }
        return "Stray impact scars " + describeTile(ch) + " at " + x + "," + y + "; integrity now " + hp + ".";
    }

    boolean isProtectedTransitionTile(char ch) {
        return ch == 'D' || ch == 'S' || ch == 'v' || ch == 'E';
    }

    boolean tileCanBeDamagedIntoDebris(char ch) {
        if (ch == '.' || ch == '+' || ch == ',' || ch == ';' || ch == ':' || ch == '~' || ch == '=' || ch == '-' || ch == '`') return false;
        if (ch == InterstitialInfrastructureApi.VOID_SPACE) return false;
        return !isProtectedTransitionTile(ch);
    }

    void performCombatAttackAt(TargetingSolution ts) {
        if (ts == null || world == null || active == null) return;
        noteCombatMusicActivity(ts.targetNpc != null && isGovernorCombatNpc(ts.targetNpc), "player targeted combat");
        if (!consumeLoadedShotsForAttack(ts)) return;
        sounds.playWeaponEffect(ts.weaponName, ts.ranged, options);
        int time = fireModeTimeCost();
        int roll = rng.nextInt(100) + 1;
        String label = ts.targetNpc == null ? ("terrain " + describeTile(world.tiles[ts.targetX][ts.targetY])) : ts.targetNpc.name;
        if (roll > ts.hitPercent) {
            Point missPoint = ts.ranged ? strayImpactPoint(ts) : new Point(ts.targetX, ts.targetY);
            startCombatAnimation(ts.weaponName, playerX, playerY, missPoint.x, missPoint.y, false);
            int terrainDmg = Math.max(1, ItemCatalog.priceFor(ts.weaponName) / 10 + fireModeDamageModifier());
            String impact = ts.ranged ? applyStrayImpactDamage(missPoint.x, missPoint.y, terrainDmg, "missed " + ts.weaponName) : "";
            String miss = ts.coverPenalty > 0 ? " Shot defeated by obstruction/cover: " + ts.coverSummary + "." : "";
            logEvent("ATTACK MISSED: " + active.name + " fires/strikes at " + label + " with " + ts.weaponName + " [" + ts.fireMode + "]. Chance " + ts.hitPercent + "%, roll " + roll + "." + miss + (impact.isBlank()?"":" " + impact));
            gainXp(ts.ranged ? "Firearms" : "Melee", 1, "missed targeted attack");
            for (int i=0;i<time;i++) advanceTurn("recovers from a targeted attack.");
            return;
        }
        startCombatAnimation(ts.weaponName, playerX, playerY, ts.targetX, ts.targetY, true);
        if (ts.targetNpc != null && ts.targetNpc.isUntargetableAnchor()) {
            logEvent("ATTACK BLOCKED: " + ts.targetNpc.name + " is a protected continuity anchor and cannot be targeted.");
            return;
        }
        if (ts.targetNpc == null) {
            int terrainDmg = Math.max(1, ItemCatalog.priceFor(ts.weaponName) / 8 + fireModeDamageModifier() + 1);
            String impact = applyStrayImpactDamage(ts.targetX, ts.targetY, terrainDmg, "direct " + ts.weaponName);
            logEvent("ATTACK IMPACT: " + active.name + " hits " + label + " at " + ts.targetX + "," + ts.targetY + ". " + impact);
            gainXp(ts.ranged ? "Firearms" : "Melee", 1, "terrain shot");
            for (int i=0;i<time;i++) advanceTurn("fires at terrain/cover.");
            return;
        }
        String weapon = ts.weaponName;
        boolean armed = weapon != null;
        int skill = Math.max(1, (ts.ranged ? stat("Firearms",5) : stat("Melee",5)) + playerAttackMotionSkillModifier());
        int base = armed ? Math.max(2, ItemCatalog.priceFor(weapon) / 4) : 2;
        int raw = base + Math.max(0, skill/2) + fireModeDamageModifier() + rng.nextInt(4);
        int armor = factionArmor(ts.targetNpc);
        int shieldCut = ts.shieldReduction <= 0 ? 0 : Math.max(1, (raw * ts.shieldReduction) / 100);
        int dmg = Math.max(1, raw - armor - shieldCut);
        ts.targetNpc.hp -= dmg;
        String zone = ts.ranged ? randomBodyHitLocation() : randomTargetedCombatReward(ts.targetNpc);
        logEvent("ATTACK HIT: " + ts.targetNpc.name + " struck in " + zone + " for " + dmg + " damage after armor " + armor + (shieldCut > 0 ? " and shield mitigation " + shieldCut : "") + ". HP now " + Math.max(0,ts.targetNpc.hp) + ". Chance " + ts.hitPercent + "%, roll " + roll + ", line " + ts.coverSummary + ".");
        gainXp(ts.ranged ? "Firearms" : "Melee", ts.targetNpc.hp <= 0 ? 5 : 3, "targeted combat hit");
        if (ts.targetNpc.hp <= 0) {
            runKills++;
            String corpseLine = createCorpseLootContainer(ts.targetNpc, "targeted combat");
            PersonnelReplacementRequest req = PersonnelProvenanceApi.recordDeathAndScheduleReplacement(world, ts.targetNpc, turn, rng, "targeted combat");
            recordProminentKillForNews(ts.targetNpc, "targeted combat");
            world.npcs.remove(ts.targetNpc);
            String repl = req == null ? "No replacement source was scheduled." : " Replacement due from " + req.source + " at turn " + req.dueTurn + ".";
            logEvent("ENEMY DOWN: " + label + " is incapacitated. " + repl + " " + corpseLine);
        }
        for (int i=0;i<time;i++) advanceTurn(ts.ranged ? "uses ranged fire mode " + ts.fireMode.toLowerCase(Locale.ROOT) + "." : "makes a melee attack.");
    }

    void performCombatAttack(boolean targeted) {
        if (world == null || active == null) { logEvent("Combat failed: no active world or character is loaded."); return; }
        NpcEntity target = nearestAttackTarget();
        if (target == null) { logEvent("Combat blocked: no adjacent hostile target. Move beside a hostile or wait for it to close."); return; }
        if (target.isUntargetableAnchor()) { logEvent("Combat blocked: " + target.name + " is protected and non-targetable."); return; }
        noteCombatMusicActivity(isGovernorCombatNpc(target), "player combat attack");
        String weapon = bestWeaponName();
        boolean adjacentRanged = weapon != null && ItemCatalog.isFirearmLike(weapon);
        if (adjacentRanged) {
            TargetingSolution ts = targetingSolutionAt(target.x, target.y);
            if (!consumeLoadedShotsForAttack(ts)) return;
        }
        sounds.playWeaponEffect(weapon, adjacentRanged, options);
        startCombatAnimation(weapon, playerX, playerY, target.x, target.y, true);
        boolean armed = weapon != null;
        if (armed && !hasUsableWeaponLimb()) { logEvent("Combat failed: your arms/hands are too damaged to keep a weapon working. Use medical care or fight bare-handed later."); return; }
        int skill = Math.max(stat("Melee", 5), armed && ItemCatalog.isFirearmLike(weapon) ? stat("Firearms",5) : stat("Melee",5));
        int armPenalty = crippledArmCount() * 2;
        int painPenalty = pain >= 6 ? 2 : 0;
        int motionPenalty = playerAttackMotionSkillModifier();
        skill = Math.max(1, skill - armPenalty - painPenalty + motionPenalty);
        int defense = 8 + Math.max(0, target.idleBias) + factionArmor(target);
        int roll = rng.nextInt(20) + 1 + skill;
        int hitTarget = defense + (targeted ? 5 : 0);
        int time = targeted ? 2 : 1;
        if (roll < hitTarget) {
            if (adjacentRanged) { Point missPoint = new Point(Math.max(0, Math.min(world.w-1, target.x + rng.nextInt(3)-1)), Math.max(0, Math.min(world.h-1, target.y + rng.nextInt(3)-1))); startCombatAnimation(weapon, playerX, playerY, missPoint.x, missPoint.y, false); applyStrayImpactDamage(missPoint.x, missPoint.y, Math.max(1, ItemCatalog.priceFor(weapon)/10), "adjacent miss " + weapon); }
            logEvent("ATTACK MISSED: " + active.name + " attacks " + target.name + (armed ? " with " + weapon : " bare-handed") + ". Roll " + roll + " < " + hitTarget + ". Active motion " + activeMotionStateLabel() + " modifies aim by " + playerAttackMotionSkillModifier() + ".");
            gainXp(armed && ItemCatalog.isFirearmLike(weapon) ? "Firearms" : "Melee", 1, "missed combat attack");
            for (int i=0;i<time;i++) advanceTurn("attacks and recovers balance.");
            return;
        }
        int base = armed ? Math.max(2, ItemCatalog.priceFor(weapon) / 4) : 2;
        int raw = base + Math.max(0, skill/2) + rng.nextInt(4);
        int armor = factionArmor(target);
        int dmg = Math.max(1, raw - armor);
        target.hp -= dmg;
        String zone = targeted ? randomTargetedCombatReward(target) : randomBodyHitLocation();
        logEvent("ATTACK HIT: " + target.name + " struck in " + zone + " for " + dmg + " damage after armor " + armor + ". HP now " + Math.max(0,target.hp) + ". Active motion " + activeMotionStateLabel() + " modifies aim by " + playerAttackMotionSkillModifier() + ".");
        gainXp(armed && ItemCatalog.isFirearmLike(weapon) ? "Firearms" : "Melee", target.hp <= 0 ? 5 : 3, "combat hit");
        if (target.hp <= 0) {
            runKills++;
            String corpseLine = createCorpseLootContainer(target, "player combat");
            PersonnelReplacementRequest req = PersonnelProvenanceApi.recordDeathAndScheduleReplacement(world, target, turn, rng, "player combat");
            recordProminentKillForNews(target, "adjacent combat");
            world.npcs.remove(target);
            String repl = req == null ? "No replacement source was scheduled." : " Replacement due from " + req.source + " at turn " + req.dueTurn + ".";
            logEvent("ENEMY DOWN: " + target.name + " is incapacitated. " + target.originSummary() + repl + " " + corpseLine);
        }
        for (int i=0;i<time;i++) advanceTurn(targeted ? "makes a careful targeted attack." : "makes a combat attack.");
    }

    String randomBodyHitLocation() {
        String[] parts = {"head", "torso", "left arm", "right arm", "left leg", "right leg"};
        return parts[rng.nextInt(parts.length)];
    }

    String randomTargetedCombatReward(NpcEntity target) {
        String[] parts = {"weapon arm", "knee joint", "breathing line", "tool hand", "center mass"};
        String p = parts[rng.nextInt(parts.length)];
        if (p.contains("arm")) logEvent("TARGETING EFFECT: a clean arm hit stresses weapon handling and equipment control.");
        if (p.contains("knee")) logEvent("TARGETING EFFECT: leg damage is flagged for speed and pursuit impairment.");

    }

    Point strayImpactPointFrom(int sx, int sy, int tx, int ty, int distance) {
        if (world == null) return new Point(tx, ty);
        int dx = Integer.compare(tx - sx, 0);
        int dy = Integer.compare(ty - sy, 0);
        int sideX = dy;
        int sideY = -dx;
        int range = Math.max(1, distance);
        int scatter = 1 + rng.nextInt(Math.max(1, Math.min(3, 1 + range / 4)));
        if (rng.nextBoolean()) scatter = -scatter;
        int forward = rng.nextInt(3) - 1;
        int ix = tx + sideX * scatter + dx * forward;
        int iy = ty + sideY * scatter + dy * forward;
        ix = Math.max(0, Math.min(world.w-1, ix));
        iy = Math.max(0, Math.min(world.h-1, iy));
        return new Point(ix, iy);
    }

    boolean maybeNpcRangedAttack(NpcEntity npc, int dist) {
        if (npc == null || world == null || active == null || !isHostileNpc(npc)) return false;
        ensureNpcCombatLoadout(npc);
        String weapon = npc.equippedRangedWeapon;
        if (weapon == null || weapon.isBlank() || !ItemCatalog.isFirearmLike(weapon)) return false;
        if (dist <= 1 || dist > weaponRange(weapon)) return false;
        if (turn % 2 != Math.floorMod(npc.numericId, 2)) return false;
        if (!hasLineOfSight(npc.x, npc.y, playerX, playerY)) return false;
        int cap = magazineCapacityForWeapon(weapon);
        if (cap <= 0) return false;
        if (npc.loadedShots <= 0) {
            if (npc.ammoReloadsRemaining > 0) {
                npc.ammoReloadsRemaining--;
                npc.loadedShots = cap;
                npc.activeMotionStateIndex = MOTION_STATIONARY;
                sounds.playReloadEffect(options);
                if (isVisible(npc.x, npc.y) || dist <= hearingRange()) logEvent("ENEMY RELOAD: " + npc.name + " reloads " + weapon + " from faction-carried reserve. Loaded " + npc.loadedShots + "/" + cap + ", reserve reloads " + npc.ammoReloadsRemaining + ".");
                addSoundPing(npc.x, npc.y, "reload");
                DebugLog.audit("NPC_RELOAD", "npc=" + npc.id + " weapon=" + weapon + " loaded=" + npc.loadedShots + " reserve=" + npc.ammoReloadsRemaining + " dist=" + dist + " turn=" + turn);
                return true;
            }
            if (isVisible(npc.x, npc.y) || dist <= hearingRange()) logEvent("ENEMY AMMO DRY: " + npc.name + " tries to bring " + weapon + " up, but the reserve is spent and they close with " + npc.equippedMeleeWeapon + ".");
            return false;
        }
        npc.loadedShots--;
        npc.activeMotionStateIndex = npc.selectedMovementModeIndex;
        sounds.playWeaponEffect(weapon, true, options);
        int coverPenalty = rangedCoverPenalty(npc.x, npc.y, playerX, playerY, true, null);
        int darknessPenalty = rangedDarknessPenalty(npc.x, npc.y, playerX, playerY);
        int skill = 35 + npc.intellect * 3 + npc.equipmentTier * 4;
        if (npc.activeMotionStateIndex == MOTION_RUN) skill -= 8;
        if (npc.activeMotionStateIndex == MOTION_SPRINT) skill -= 14;
        if (npc.activeMotionStateIndex == MOTION_STATIONARY) skill += 8;
        int evade = Math.max(0, playerDefenseMotionModifier() * 3);
        int hit = Math.max(3, Math.min(92, skill - dist * 4 - coverPenalty - darknessPenalty - evade));
        int roll = rng.nextInt(100) + 1;
        if (roll > hit) {
            Point miss = strayImpactPointFrom(npc.x, npc.y, playerX, playerY, dist);
            startCombatAnimation(weapon, npc.x, npc.y, miss.x, miss.y, false);
            String impact = applyStrayImpactDamage(miss.x, miss.y, Math.max(1, ItemCatalog.priceFor(weapon)/12), "enemy missed " + weapon);
            logEvent("ENEMY RANGED MISSED: " + npc.name + " fires " + weapon + " at you. Hit " + hit + "%, roll " + roll + "; loaded " + npc.loadedShots + "/" + cap + ", reserve reloads " + npc.ammoReloadsRemaining + ". " + impact);
            DebugLog.audit("NPC_RANGED_MISS", "npc=" + npc.id + " weapon=" + weapon + " hit=" + hit + " roll=" + roll + " cover=" + coverPenalty + " lightPenalty=" + darknessPenalty + " loaded=" + npc.loadedShots + " reserve=" + npc.ammoReloadsRemaining + " turn=" + turn);
            return true;
        }
        startCombatAnimation(weapon, npc.x, npc.y, playerX, playerY, true);
        int raw = Math.max(2, ItemCatalog.priceFor(weapon) / 8) + rng.nextInt(4) + Math.max(0, npc.equipmentTier);
        int shield = shieldMitigationPercentAt(playerX, playerY, equippedClothing == null ? Faction.NONE : equippedClothing.alignedFaction);
        int armor = (equippedClothing != null && equippedClothing.name.toLowerCase(Locale.ROOT).contains("armor") ? 2 : 0) + playerEquippedArmorBonus();
        int dmg = Math.max(1, ((raw - armor) * Math.max(20, 100 - shield)) / 100);
        logEvent("ENEMY RANGED HIT: " + npc.name + " fires " + weapon + " and hits. Hit " + hit + "%, roll " + roll + "; shield mitigation " + shield + "%; loaded " + npc.loadedShots + "/" + cap + ", reserve reloads " + npc.ammoReloadsRemaining + ".");
        applyPlayerBodyDamage(dmg, randomBodyHitLocation(), npc.name, weapon);
        DebugLog.audit("NPC_RANGED_HIT", "npc=" + npc.id + " weapon=" + weapon + " hit=" + hit + " roll=" + roll + " cover=" + coverPenalty + " lightPenalty=" + darknessPenalty + " shield=" + shield + " dmg=" + dmg + " loaded=" + npc.loadedShots + " reserve=" + npc.ammoReloadsRemaining + " turn=" + turn);
        return true;
    }

    void hostileAdjacentAttack(NpcEntity npc) {
        if (npc == null || active == null || Math.abs(npc.x-playerX)+Math.abs(npc.y-playerY) > 1) return;
        noteCombatMusicActivity(isGovernorCombatNpc(npc), "hostile adjacent attack");
        ensureNpcCombatLoadout(npc);
        String npcWeapon = npcAttackWeapon(npc);
        sounds.playWeaponEffect(npcWeapon, ItemCatalog.isFirearmLike(npcWeapon), options);
        if (activeMotionIsSneak() && !npc.hostileBarked) {
            logEvent("STEALTH HELD: " + npc.name + " is close enough to be dangerous, but has not fixed on you through the crouched movement and bad light.");
            maybeGrantSneakXp("undetected adjacent threat");
            return;
        }
        int motionDefense = playerDefenseMotionModifier();
        int defense = 8 + stat("Agility",5) + stat("Nerve",5)/2 + motionDefense;
        int roll = rng.nextInt(20) + 1 + Math.max(4, npc.idleBias + 5 + npc.equipmentTier);
        if (roll < defense) { logEvent("ENEMY ATTACK MISSED: " + npc.name + " lunges but you avoid it. Roll " + roll + " < defense " + defense + " (motion " + activeMotionStateLabel() + " +" + motionDefense + ")."); return; }
        int raw = 2 + rng.nextInt(5) + Math.max(0, ItemCatalog.priceFor(npcWeapon) / 18);
        int armor = (equippedClothing != null && equippedClothing.name.toLowerCase(Locale.ROOT).contains("armor") ? 2 : 0) + playerEquippedArmorBonus();
        int dmg = Math.max(1, raw - armor);
        applyPlayerBodyDamage(dmg, randomBodyHitLocation(), npc.name, npcWeapon);
    }

    boolean weaponAddsSleepPressure(String weapon) {
        if (weapon == null) return false;
        String l = weapon.toLowerCase(Locale.ROOT);
        return l.contains("shock") || l.contains("taser") || l.contains("stun") || l.contains("webber") || l.contains("sleep gas") || l.contains("gas") || l.contains("tranq") || l.contains("compliance") || l.contains("arc");
    }

    void applyTraumaSleepPressure(int dmg, String weapon, int oldBleeding) {
        int add = 0;
        if (bleeding > oldBleeding) add += (bleeding - oldBleeding) * (TURNS_PER_HOUR / 5);
        if (bleeding > 0) add += Math.max(1, dmg) * 2;
        if (weaponAddsSleepPressure(weapon)) add += TURNS_PER_HOUR / 2 + Math.max(0, dmg) * 8;
        if (add <= 0) return;
        int old = sleepNeed;
        sleepNeed = Math.min(MAX_SLEEP_NEED, sleepNeed + add);
        fatigue = Math.min(30, fatigue + (weaponAddsSleepPressure(weapon) ? 2 : 0));
        if (sleepNeed != old) logEvent("TRAUMA FATIGUE: bleeding, shock, stun, or gas trauma drives sleep debt " + old + "->" + sleepNeed + "/" + MAX_SLEEP_NEED + ".");
    }

    void applyPlayerBodyDamage(int dmg, String location, String source, String weapon) {
        int oldBleeding = bleeding;
        wounds += Math.max(1, dmg / 3);
        pain = Math.min(20, pain + Math.max(1, dmg));
        if (dmg >= 3) bleeding = Math.min(12, bleeding + 1 + dmg/3);
        if (dmg >= 4) infectionRisk = Math.min(12, infectionRisk + 1);
        applyTraumaSleepPressure(dmg, weapon, oldBleeding);
        BodyPart bp = bodyPartForLocation(location);
        if (bp == null && active != null && active.body != null && !active.body.isEmpty()) {
            ArrayList<BodyPart> parts = new ArrayList<>(active.body.values());
            bp = parts.get(rng.nextInt(parts.size()));
        }
        double before = bp == null ? 0 : bp.currentHealth();
        if (bp != null) bp.health = Math.max(0, bp.health - dmg * 35.0);
        String part = bp == null ? location : bp.name;
        logEvent("DAMAGE TAKEN: " + source + " hits your " + part + " for " + dmg + ". Wounds " + wounds + ", bleeding " + bleeding + ", pain " + pain + ".");
        if (bp != null) {
            DebugLog.audit("BODY_DAMAGE", "part=" + bp.name + " hp=" + String.format(Locale.US,"%.0f", before) + "->" + String.format(Locale.US,"%.0f", bp.currentHealth()) + " bleeding=" + bleeding + " pain=" + pain + " infectionRisk=" + infectionRisk + " state=" + stateSummary());
            if (bp.currentHealth() <= bp.maxHealth()*0.25 && before > bp.maxHealth()*0.25) {
                String low = bp.name.toLowerCase(Locale.ROOT);
                if (low.contains("leg") || low.contains("foot")) logEvent("BODY DAMAGE: " + bp.name + " crippled. Movement now costs extra time until treated.");
                else if (low.contains("arm") || low.contains("hand") || low.contains("finger")) logEvent("BODY DAMAGE: " + bp.name + " crippled. Weapon handling and tool work are penalized.");
                else if (low.contains("head")) logEvent("BODY DAMAGE: head trauma. Pain and stun checks become dangerous.");
                else logEvent("BODY DAMAGE: " + bp.name + " critically damaged. Medical treatment is advised, which is underhive for 'stop leaking'.");
            }
            if (bp.destroyed()) logEvent("BODY DAMAGE: " + bp.name + " disabled. Severe consequences are active in movement/combat penalties.");
        }
        evaluatePlayerDefeatState("combat trauma to " + part, source, weapon, part);
    }

    void applyMedicalProgressionTick() {
        if (turn <= 0 || turn % TURNS_PER_HOUR != 0) return;
        if (bleeding > 0) {
            wounds += 1; fatigue = Math.min(25, fatigue + 1); sleepNeed = Math.min(MAX_SLEEP_NEED, sleepNeed + TURNS_PER_HOUR / 3); bleeding = Math.max(0, bleeding - 1);
            logEvent("MEDICAL WARNING: bleeding costs one wound and drives sleep debt this hour. Bleeding now " + bleeding + ", sleep need " + sleepNeed + "/" + MAX_SLEEP_NEED + ". Use bandages, dressings, or a medkit.");
        }
        if (infectionRisk > 0 && turn % (8*TURNS_PER_HOUR) == 0) {
            if (rng.nextInt(12) < infectionRisk) { wounds++; fatigue = Math.min(25, fatigue + 2); logEvent("MEDICAL WARNING: untreated infection risk worsens. Wounds " + wounds + ", fatigue " + fatigue + "."); }
            infectionRisk = Math.max(0, infectionRisk - 1);
        }
        if (pain > 0 && turn % (2*TURNS_PER_HOUR) == 0) pain = Math.max(0, pain - 1);
    }

    void rebuildLightLevelCache() {
        if (world == null) return;
        if (lightLevelTiles == null || lightLevelCacheWorld != world || lightLevelTiles.length != world.w || lightLevelTiles[0].length != world.h) {
            lightLevelTiles = new int[world.w][world.h];
        }
        int ambient = ambientLightLevelForWorld();
        for (int x=0; x<world.w; x++) for (int y=0; y<world.h; y++) {
            int base = ambient;
            char ch = world.tiles[x][y];
            if (blocksLineOfSight(ch) || InterstitialInfrastructureApi.isInterstitialSolid(ch) || ch == InterstitialInfrastructureApi.VOID_SPACE) base = Math.max(0, ambient/3);
            lightLevelTiles[x][y] = base;
        }
        int activeSources = 0;
        int r = activePortableLightRadius();
        if (r > 0) { addLightLevelSource(playerX, playerY, r, 78, new Color(255,230,150)); activeSources++; }
        for (PortableLightInstance l : portableLights) {
            if (l == null || l.expiresTurn <= turn || !sameWorldLocation(l.worldKey)) continue;
            PortableLightProfile p = PortableLightProfile.profile(l.itemName);
            int rr = p == null ? l.radius : p.radius;
            addLightLevelSource(l.x, l.y, rr, portableLightIntensity(l.itemName), new Color(250,230,130));
            activeSources++;
        }
        if (world.lightSources != null) for (ZoneLightSourceRecord l : world.lightSources) {
            if (!lightSourceCurrentlyEmitting(l)) continue;
            addLightLevelSource(l.x, l.y, Math.max(1, l.radius), Math.max(8, l.intensity), l.color());
            activeSources++;
        }
        addDoorAndMachineFallbackLight();
        lightLevelCacheWorld = world;
        lightLevelCacheTurn = turn;
        profilingLightCacheRebuilds++;
        localDirtyTracker.clearLight(world);
        lastLightRenderReport = "Light render cache: ambient " + ambient + ", active sources " + activeSources + ", flicker turn " + turn + ", rebuilds " + profilingLightCacheRebuilds + ".";
        lastDirtyRegionReport = localDirtyTracker.summary();
    }

    void addLightLevelSource(int sx, int sy, int radius, int intensity, Color color) {
        if (world == null || lightLevelTiles == null || !world.inBounds(sx, sy)) return;
        int r = Math.max(1, Math.min(12, radius));
        int power = Math.max(1, Math.min(100, intensity));
        for (int x=Math.max(0,sx-r); x<=Math.min(world.w-1,sx+r); x++) for (int y=Math.max(0,sy-r); y<=Math.min(world.h-1,sy+r); y++) {
            double dist = Math.hypot(x - sx, y - sy);
            if (dist > r + 0.001) continue;
            // End-point walls may be lit, but any tile behind a blocking wall remains dark.
            if (!hasLineOfSight(sx, sy, x, y)) continue;
            double norm = Math.max(0.0, Math.min(1.0, dist / Math.max(1.0, r)));
            double smooth = norm * norm * (3.0 - 2.0 * norm);
            double falloff = Math.max(0.0, (1.0 - smooth) * 0.82 + 0.18 / (1.0 + 2.0 * norm * norm));
            int add = (int)Math.round(power * falloff);
            if (add <= 0) continue;
            lightLevelTiles[x][y] = Math.max(lightLevelTiles[x][y], Math.min(100, add));
        }
    }

    boolean inPlayerVisionShape(int x, int y) {
        if (x == playerX && y == playerY) return true;
        int dx = x - playerX, dy = y - playerY;
        int fx = facingDx(), fy = facingDy();
        double forward = dx * fx + dy * fy;
        double side = Math.abs(dx * fy - dy * fx);
        if (forward >= 0) {
            double a = Math.max(1.0, visionForwardRadius());
            double b = Math.max(1.0, visionSideRadius());
            return (forward*forward)/(a*a) + (side*side)/(b*b) <= 1.0;
        } else {
            double a = Math.max(1.0, visionRearRadius());
            double b = Math.max(1.0, visionSideRadius() * 0.78);
            double rear = -forward;
            return (rear*rear)/(a*a) + (side*side)/(b*b) <= 1.0;
        }
    }

    int applyLightCircle(int sx, int sy, int radius, boolean carried) {
        int added = 0;
        int r = Math.max(1, Math.min(10, radius));
        for (int x=Math.max(0,sx-r); x<=Math.min(world.w-1,sx+r); x++) for (int y=Math.max(0,sy-r); y<=Math.min(world.h-1,sy+r); y++) {
            int dist = Math.abs(x-sx) + Math.abs(y-sy);
            if (dist > r) continue;
            if (!hasLineOfSight(sx, sy, x, y)) continue;
            if (!visibleTiles[x][y]) { visibleTiles[x][y] = true; added++; }
            rememberedTiles[x][y] = true;
        }
        return added;
    }

    void updateSensoryModel(String reason) {
        if (world == null || active == null || visibleTiles == null || rememberedTiles == null) return;
        int vr = visionRange();
        ensureLightLevelCache();
        rebuildNoiseHearingFieldCache(reason);
        int fx = facingDx(), fy = facingDy();
        boolean fullRecalcNeeded = sensoryCacheWorld != world || sensoryCacheX != playerX || sensoryCacheY != playerY || sensoryCacheVision != vr || sensoryCacheFacingDx != fx || sensoryCacheFacingDy != fy || sensoryCacheMotion != activeMotionStateIndex || portableLightCacheDirty() || !"paint".equals(reason);
        if (!fullRecalcNeeded) return;
        int visible = 0;
        int maxSpan = Math.max(vr + 2, (int)Math.ceil(Math.max(visionForwardRadius(), Math.max(visionSideRadius(), visionRearRadius()))) + 2);
        Rectangle clearRect = localDirtyTracker.visibilityClearRect(world, playerX, playerY, maxSpan, lastVisibilityDirtyRect);
        clearVisibleTiles(clearRect);
        lastVisibilityDirtyRect = clearRect;
        for (int dx=-maxSpan; dx<=maxSpan; dx++) for (int dy=-maxSpan; dy<=maxSpan; dy++) {
            int x = playerX + dx, y = playerY + dy;
            if (x < 0 || y < 0 || x >= world.w || y >= world.h) continue;
            if (!inPlayerVisionShape(x, y)) continue;
            int dist = Math.abs(dx) + Math.abs(dy);
            if (!lightAllowsPlayerVision(x, y, dist)) continue;
            if (hasLineOfSight(playerX, playerY, x, y)) {
                visibleTiles[x][y] = true;
                rememberedTiles[x][y] = true;
                visible++;
            }
        }
        // 0.9.10jh: light is a visibility gate, not remote clairvoyance.
        // Zone fixtures and thrown lights no longer reveal distant rooms by themselves;
        // the player still needs immediate vision-shape range and line of sight.
        lightingCacheTurn = turn;
        lastVisibleCount = visible;
        sensoryCacheWorld = world;
        sensoryCacheX = playerX;
        sensoryCacheY = playerY;
        sensoryCacheVision = vr;
        sensoryCacheFacingDx = fx;
        sensoryCacheFacingDy = fy;
        sensoryCacheMotion = activeMotionStateIndex;
        decaySoundPings();
        if (!"paint".equals(reason)) scanForDistantSounds(reason);
        localDirtyTracker.clearVision(world);
        lastDirtyRegionReport = localDirtyTracker.summary();
    }

    void clearVisibleTiles(Rectangle rect) {
        if (world == null || visibleTiles == null) return;
        if (rect == null || rect.width >= world.w || rect.height >= world.h) {
            for (int x=0; x<world.w; x++) for (int y=0; y<world.h; y++) visibleTiles[x][y] = false;
            profilingVisibilityFullClears++;
            return;
        }
        int x0 = Math.max(0, rect.x), y0 = Math.max(0, rect.y);
        int x1 = Math.min(world.w-1, rect.x + rect.width - 1), y1 = Math.min(world.h-1, rect.y + rect.height - 1);
        for (int x=x0; x<=x1; x++) for (int y=y0; y<=y1; y++) visibleTiles[x][y] = false;
        profilingVisibilityLocalClears++;
    }

    void decaySoundPings() {
        for (int i=sensePings.size()-1; i>=0; i--) {
            SensePing p = sensePings.get(i);
            p.ttl--;
            if (p.ttl <= 0) sensePings.remove(i);
        }
    }

    void scanForDistantSounds(String reason) {
        int hr = hearingRange();
        int heard = 0;
        ArrayList<String> heardLines = new ArrayList<>();
        for (int x=Math.max(0, playerX-hr); x<=Math.min(world.w-1, playerX+hr); x++) {
            for (int y=Math.max(0, playerY-hr); y<=Math.min(world.h-1, playerY+hr); y++) {
                if (isVisible(x,y)) continue;
                char ch = world.tiles[x][y];
                if (!makesSound(ch)) continue;
                int dist = Math.abs(playerX-x) + Math.abs(playerY-y);
                int chance = soundChance(ch, x, y, dist);
                int roll = rng.nextInt(100) + 1;
                DebugLog.audit("HEARING_CHECK", "reason=" + reason + " tile=" + ch + " at=" + x + "," + y + " dist=" + dist + " roll=" + roll + " chance=" + chance + " noisePenalty=" + noisePenaltyAt(x,y) + " state=" + stateSummary());
                if (roll <= chance) {
                    heard++;
                    String label = soundLabel(ch);
                    addSoundPing(x, y, label);
                    maybePlayHeardSoundCue(label, dist);
                    if (heardLines.size() < 3) heardLines.add(directionAndDistanceLabel(x, y) + " " + label);
                }
            }
        }
        if (world.noiseSources != null) {
            for (ZoneNoiseSourceRecord ns : world.noiseSources) {
                if (ns == null || !ns.active || !world.inBounds(ns.x, ns.y)) continue;
                int dist = Math.abs(playerX-ns.x) + Math.abs(playerY-ns.y);
                if (dist <= 0 || dist > Math.min(hr + 6, ns.radius + hr/2)) continue;
                if (isVisible(ns.x, ns.y) && dist > detailVisionRange()) continue;
                int chance = Math.max(4, Math.min(92, 18 + stat("Hearing",6)*5 + ns.intensity - dist*4 - noisePenaltyAt(playerX, playerY)/3));
                if (ns.intermittent && Math.floorMod(turn + ns.phase, Math.max(2, ns.period)) == 0) chance /= 3;
                int roll = rng.nextInt(100) + 1;
                DebugLog.audit("HEARING_NOISE_SOURCE", "reason=" + reason + " source=" + ns.summary() + " dist=" + dist + " roll=" + roll + " chance=" + chance + " state=" + stateSummary());
                if (roll <= chance) {
                    heard++;
                    addSoundPing(ns.x, ns.y, ns.label);
                    maybePlayHeardSoundCue(ns.label, dist);
                    if (heardLines.size() < 3) heardLines.add(directionAndDistanceLabel(ns.x, ns.y) + " " + ns.label);
                }
            }
        }
        if (world.npcs != null) {
            for (NpcEntity npc : world.npcs) {
                if (npc == null) continue;
                int dist = Math.abs(playerX-npc.x) + Math.abs(playerY-npc.y);
                if (dist <= 0 || dist > hr) continue;
                if (isVisible(npc.x, npc.y) && dist > Math.max(2, detailVisionRange())) continue;
                int chance = npcSoundChance(npc, dist);
                int roll = rng.nextInt(100) + 1;
                DebugLog.audit("HEARING_NPC_CHECK", "reason=" + reason + " npc=" + npc.id + " at=" + npc.x + "," + npc.y + " state=" + npc.state + " dist=" + dist + " roll=" + roll + " chance=" + chance + " active=" + activeMotionStateLabel());
                if (roll <= chance) {
                    heard++;
                    String label = npcSoundLabel(npc);
                    addSoundPing(npc.x, npc.y, label);
                    maybePlayHeardSoundCue(label, dist);
                    if (heardLines.size() < 3) heardLines.add(directionAndDistanceLabel(npc.x, npc.y) + " " + label + (npc.activeMotionStateIndex == MOTION_SNEAK ? " — possible sneaking entity" : ""));
                }
            }
        }
        lastHeardCount = heard;
        if (heard > 0) {
            String detail = heardLines.isEmpty() ? "" : " — " + String.join("; ", heardLines);
            lastSensesReport = "Heard " + heard + " disturbance(s) within " + hr + " tiles on " + reason + detail + ".";
            if (turn != lastHearingLogTurn || heard >= 2) {
                logEvent(lastSensesReport);
                lastHearingLogTurn = turn;
            }
        } else {
            lastSensesReport = "No clear distant sounds on " + reason + ".";
        }
    }

    boolean makesSound(char ch) { return ch == 'g' || ch == 'm' || ch == 'R' || ch == 'A' || ch == 'N' || ch == 'p' || ch == 'b' || ch == 'M'; }
    String soundLabel(char ch) {
        switch(ch) {
            case 'g': case 'b': return "voices/boots";
            case 'm': return "wet movement";
            case 'R': case 'N': return "machinery";
            case 'A': case 'p': return "law boots";
            case 'M': return "drilled movement";
            default: return "unknown noise";
        }
    }

    int soundChance(char ch, int x, int y, int dist) {
        int base = 15 + stat("Hearing", 6) * 6 - dist * 4 - noisePenaltyAt(x,y);
        if (ch == 'N' || ch == 'R') base += 18;
        if (ch == 'A' || ch == 'p' || ch == 'M') base += 8;
        return Math.max(3, Math.min(85, base));
    }

    int npcSoundChance(NpcEntity npc, int dist) {
        if (npc == null) return 0;
        int base = 10 + stat("Hearing", 6) * 5 - dist * 3;
        if ("Hostile".equalsIgnoreCase(npc.state)) base += 18;
        if ("Patrol".equalsIgnoreCase(npc.state) || "Guard".equalsIgnoreCase(npc.state)) base += 10;
        if ("Trade".equalsIgnoreCase(npc.state) || "Prayer".equalsIgnoreCase(npc.state) || "Pilgrim Service".equalsIgnoreCase(npc.state)) base += 5;
        if (npc.faction == Faction.ROGUE_MACHINE || npc.faction == Faction.MECHANIST_COLLEGIA || (npc.faction != null && npc.faction.name().startsWith("MECHANIST COLLEGIA"))) base += 4;
        if (activeMotionStateIndex == MOTION_STATIONARY) base += 8;
        if (activeMotionStateIndex == MOTION_SNEAK) base += 4;
        if (activeMotionStateIndex == MOTION_RUN) base -= 10;
        if (activeMotionStateIndex == MOTION_SPRINT) base -= 18;
        base += npcMotionSoundModifier(npc);
        base += doorwayExposureBonus(npc.x, npc.y);
        base -= detectionObstructionPenalty(playerX, playerY, npc.x, npc.y, false) / 2;
        return Math.max(4, Math.min(95, base));
    }

    int npcMotionSoundModifier(NpcEntity npc) {
        if (npc == null) return 0;
        switch(npc.activeMotionStateIndex) {
            case MOTION_SNEAK: return -18;
            case MOTION_WALK: return 0;
            case MOTION_RUN: return 12;
            case MOTION_SPRINT: return 22;
            default: return -6;
        }
    }

    int npcMotionVisionModifier(NpcEntity npc) {
        if (npc == null) return 0;
        switch(npc.activeMotionStateIndex) {
            case MOTION_SNEAK: return -1;
            case MOTION_RUN: return -1;
            case MOTION_SPRINT: return -2;
            case MOTION_WALK: return 1;
            default: return 0;
        }
    }

    int npcMotionHearingModifier(NpcEntity npc) {
        if (npc == null) return 0;
        switch(npc.activeMotionStateIndex) {
            case MOTION_SNEAK: return 1;
            case MOTION_RUN: return -2;
            case MOTION_SPRINT: return -4;
            case MOTION_WALK: return 1;
            default: return 2;
        }
    }

    int doorwayExposureBonus(int x, int y) {
        if (world == null || !world.inBounds(x,y)) return 0;
        if (isDoorTile(world.tiles[x][y])) return 24;
        int bonus = 0;
        int[][] dirs={{1,0},{-1,0},{0,1},{0,-1}};
        for (int[] d: dirs) {
            int nx=x+d[0], ny=y+d[1];
            if (world.inBounds(nx,ny) && isDoorTile(world.tiles[nx][ny])) bonus = Math.max(bonus, 10);
        }
        return bonus;
    }

    int detectionObstructionPenalty(int x0, int y0, int x1, int y1, boolean includeEntities) {
        TargetingSolution scratch = new TargetingSolution();
        int raw = rangedCoverPenalty(x0,y0,x1,y1,includeEntities,scratch);
        if (scratch.coverHardBlocked) return 60;
        return Math.max(0, Math.min(45, raw / 2));
    }

    String npcSoundLabel(NpcEntity npc) {
        if (npc == null) return "unidentified movement";
        if (npc.activeMotionStateIndex == MOTION_SNEAK) return doorwayExposureBonus(npc.x, npc.y) > 0 ? "careful movement crossing a doorway" : "careful hidden movement";
        if ("Hostile".equalsIgnoreCase(npc.state)) return "urgent movement";
        if ("Guard".equalsIgnoreCase(npc.state) || "Patrol".equalsIgnoreCase(npc.state)) return "boots and patrol kit";
        if (npc.faction == Faction.MUTANT) return "wet shuffling";
        if (npc.faction == Faction.CULTIST || npc.faction == Faction.HERETIC) return "low muttering";
        if (npc.faction == Faction.ROGUE_MACHINE) return "servo movement";
        if (npc.faction == Faction.MINISTORUM) return "prayers and footfalls";
        return "movement";
    }

    String directionAndDistanceLabel(int x, int y) {
        int dx = x - playerX, dy = y - playerY;
        String ns = dy < 0 ? "north" : (dy > 0 ? "south" : "");
        String ew = dx < 0 ? "west" : (dx > 0 ? "east" : "");
        String dir = ns.isEmpty() ? ew : (ew.isEmpty() ? ns : ns + "-" + ew);
        if (dir.isEmpty()) dir = "nearby";
        int dist = Math.abs(dx) + Math.abs(dy);
        return dir + " ~" + dist + "t:";
    }

    void rebuildNoiseHearingFieldCache(String reason) {
        if (world == null) return;
        if (world.noiseFieldTurn == turn && world.noiseField != null && !localDirtyTracker.noiseDirty(world)) return;
        NoiseHearingFieldApi.Result nr = NoiseHearingFieldApi.rebuild(world, turn);
        profilingNoiseCacheRebuilds++;
        localDirtyTracker.clearNoise(world);
        if (nr != null && nr.changedSummary) DebugLog.audit("NOISE_HEARING_FIELD", "reason=" + reason + " " + nr.summary() + " rebuilds=" + profilingNoiseCacheRebuilds);
    }

    int noisePenaltyAt(int tx, int ty) {
        if (world == null) return 0;
        if (world.noiseField == null || world.noiseFieldTurn != turn || localDirtyTracker.noiseDirty(world)) rebuildNoiseHearingFieldCache("noise-query");
        if (world.noiseField != null && world.inBounds(tx,ty)) return Math.max(0, Math.min(70, world.noiseField[tx][ty]));
        return 0;
    }

    int localNoiseFieldAt(int tx, int ty) {
        return noisePenaltyAt(tx, ty);
    }

    void addSoundPing(int x, int y, String label) {
        for (SensePing p : sensePings) {
            if (Math.abs(p.x-x) <= 1 && Math.abs(p.y-y) <= 1 && p.label.equals(label)) { p.x=x; p.y=y; p.ttl=4; return; }
        }
        sensePings.add(new SensePing(x,y,label,4));
        while (sensePings.size() > 10) sensePings.remove(0);
    }

    boolean hasLineOfSight(int x0, int y0, int x1, int y1) {
        int dx = Math.abs(x1 - x0), dy = Math.abs(y1 - y0);
        int sx = x0 < x1 ? 1 : -1;
        int sy = y0 < y1 ? 1 : -1;
        int err = dx - dy;
        int x = x0, y = y0;
        while (true) {
            if (!(x == x0 && y == y0) && blocksLineOfSight(world.tiles[x][y])) return x == x1 && y == y1;
            if (x == x1 && y == y1) return true;
            int e2 = 2 * err;
            if (e2 > -dy) { err -= dy; x += sx; }
            if (e2 < dx) { err += dx; y += sy; }
            if (x < 0 || y < 0 || x >= world.w || y >= world.h) return false;
        }
    }

    boolean blocksLineOfSight(char ch) {
        // Walls and closed/secured doors block sight. Open archways/floor/corridors do not.
        return ch == '#' || InterstitialInfrastructureApi.isInterstitialSolid(ch) || ch == InterstitialInfrastructureApi.VOID_SPACE || ch == '|' || ch == 'L' || ch == 'X' || ch == 'V';
    }

    boolean isDoorTile(char ch) { return ch == '/' || ch == '|' || ch == 'L' || ch == 'X' || ch == 'V'; }

    void evaluateFactionPressure() {
        if (world == null || active == null) return;
        LinkedHashSet<Faction> witnesses = nearbyDisguiseWitnessFactions(6);
        if (witnesses.isEmpty()) return;
        char here = world.tiles[playerX][playerY];
        Faction roomFaction = factionForTile(here);
        Faction profileFactionOnly = factionOnlyRoomFactionAt(playerX, playerY);
        if (roomFaction == Faction.NONE && profileFactionOnly != Faction.NONE) roomFaction = profileFactionOnly;
        if ((tileIsFactionOnly(here) || profileFactionOnly != Faction.NONE) && roomFaction != Faction.NONE
                && witnesses.contains(roomFaction)
                && activeFaction() != roomFaction && equippedClothing.alignedFaction != roomFaction) {
            temporaryHostileTurns.put(roomFaction, 8);
            lastDisguiseFailure = roomFaction;
            lastInterrogation = "Trespass observed in " + roomFaction.label + " controlled ground at " + playerX + "," + playerY + ".";
            DebugLog.audit("FACTION_ZONE", lastInterrogation + " clothing=" + equippedClothing.name + " witnesses=" + witnesses + " state=" + stateSummary());
            logEvent("ALERT: " + roomFaction.label + " witness catches you in controlled ground. Hostility triggered.");
            return;
        }
        for (Faction witness : witnesses) {
            if (witness == activeFaction()) {
                DebugLog.audit("FACTION_AFFILIATION_PASS", "witness=" + witness + " jobFaction=" + activeFaction() + " no disguise check required state=" + stateSummary());
                continue;
            }
            if (witness == equippedClothing.alignedFaction) {
                int roll = rng.nextInt(100) + 1;
                int chance = disguiseChance(witness);
                DebugLog.audit("DISGUISE_CHECK", "witness=" + witness + " roll=" + roll + " chance=" + chance + " clothing=" + equippedClothing.name + " state=" + stateSummary());
                if (roll > chance) {
                    handleDisguiseFailure(witness, "masquerade check failed");
                    return;
                }
            } else if (factionImmediatelyObjects(witness)) {
                if (factionOnlyRoomFactionAt(playerX, playerY) == witness || factionForTile(here) == witness) {
                    handleInterrogation(witness, "suspicious outsider clothing " + equippedClothing.name);
                    return;
                }
                DebugLog.audit("DISGUISE_NON_TERRITORY_PASS", "witness=" + witness + " clothing=" + equippedClothing.name + " no faction-only territory hostility state=" + stateSummary());
            }
        }
    }

    LinkedHashSet<Faction> nearbyDisguiseWitnessFactions(int radius) {
        LinkedHashSet<Faction> witnesses = new LinkedHashSet<>();
        if (world == null || world.npcs == null) return witnesses;
        int r = Math.max(1, radius);
        for (NpcEntity npc : world.npcs) {
            if (npc == null || npc.hp <= 0 || npc.faction == null || npc.faction == Faction.NONE) continue;
            int dist = Math.max(Math.abs(npc.x - playerX), Math.abs(npc.y - playerY));
            if (dist > r) continue;
            if (dist > 1 && !hasLineOfSight(npc.x, npc.y, playerX, playerY)) continue;
            witnesses.add(npc.faction);
        }
        return witnesses;
    }

    void handleDisguiseFailure(Faction witness, String reason) {
        lastDisguiseFailure = witness;
        Faction territorial = factionOnlyRoomFactionAt(playerX, playerY);
        char here = (world != null && world.inBounds(playerX, playerY)) ? world.tiles[playerX][playerY] : '.';
        if (territorial != witness && factionForTile(here) != witness) {
            logEvent("DISGUISE READ: " + witness.label + " notices the false cover, but this is not their faction-only territory. No hostility follows.");
            DebugLog.audit("DISGUISE_NON_TERRITORY_FAILURE", "witness=" + witness + " reason=" + reason + " state=" + stateSummary());
            return;
        }
        logEvent("ALERT: " + witness.label + " recognizes your disguise is wrong inside controlled territory. Interrogation begins.");
        handleInterrogation(witness, reason);
    }

    void handleInterrogation(Faction witness, String reason) {
        int roll = rng.nextInt(100) + 1;
        int chance = interrogationChance(witness);
        DebugLog.audit("INTERROGATION", "witness=" + witness + " reason=" + reason + " roll=" + roll + " chance=" + chance + " clothing=" + equippedClothing.name + " state=" + stateSummary());
        if (witness == Faction.CIVIC_WARDENS && turn < arbitesInspectionCooldownUntilTurn) {
            lastInterrogation = "Civic Wardens scrutiny delayed: recent inspection cooldown until turn " + arbitesInspectionCooldownUntilTurn + ".";
            lastArbitesPatrolReport = lastInterrogation;
            return;
        }
        if (roll <= chance) {
            lastInterrogation = "Talked past " + witness.label + " scrutiny. Roll " + roll + " <= " + chance + ".";
            logEvent("Interrogation survived: " + witness.label + " accepts your explanation for now.");
            if (witness == Faction.CIVIC_WARDENS) {
                arbitesInspectionCooldownUntilTurn = Math.max(arbitesInspectionCooldownUntilTurn, turn + ARBITES_INSPECTION_COOLDOWN_TURNS);
                lastArbitesPatrolReport = lastInterrogation + " Next street inspection no sooner than " + TimeSurfaceApi.timeTextAtTurn(arbitesInspectionCooldownUntilTurn) + ".";
            }
        } else {
            factionStanding.put(witness, factionStanding.getOrDefault(witness, 0) - 1);
            if (witness == Faction.CIVIC_WARDENS) {
                suspicion += 3;
                arbitesInspectionCooldownUntilTurn = Math.max(arbitesInspectionCooldownUntilTurn, turn + ARBITES_INSPECTION_COOLDOWN_TURNS);
                boolean captured = roll >= chance + 12 || suspicion >= 10 || reason.toLowerCase(Locale.ROOT).contains("trespass");
                if (captured) {
                    if (turn < arbitesCaptureCooldownUntilTurn) {
                        lastInterrogation = "Failed Civic Wardens interrogation, but recent custody paperwork blocks a fresh jail processing until " + TimeSurfaceApi.timeTextAtTurn(arbitesCaptureCooldownUntilTurn) + ". They rough you up, record your face, and move you along.";
                        lastArbitesPatrolReport = lastInterrogation;
                        pain = Math.min(24, pain + 2);
                        fatigue = Math.min(30, fatigue + 1);
                        logEvent("CIVIC WARDENS INSPECTION: " + lastInterrogation);
                        return;
                    }
                    lastInterrogation = "Failed Civic Wardens interrogation hard enough to be processed. Roll " + roll + " > " + chance + ".";
                    handleArbitesCapture("failed interrogation: " + reason, Math.max(1, (roll - chance) / 10 + suspicion / 6));
                    return;
                }
            }
            if (witness == Faction.BANDIT) gangHeat += 2;
            lastInterrogation = "Failed " + witness.label + " interrogation. Roll " + roll + " > " + chance + ". Temporary hostility active.";
            if (witness == Faction.CIVIC_WARDENS) lastArbitesPatrolReport = lastInterrogation + " Inspection cooldown until " + TimeSurfaceApi.timeTextAtTurn(arbitesInspectionCooldownUntilTurn) + ".";
            temporaryHostileTurns.put(witness, 8);
            logEvent("FAILED INTERROGATION: " + witness.label + " turns hostile while nearby.");
        }
    }

    static final int ARBITES_INSPECTION_COOLDOWN_TURNS = 4 * TURNS_PER_HOUR;
    static final int ARBITES_CAPTURE_COOLDOWN_TURNS = 3 * HOURS_PER_DAY * TURNS_PER_HOUR;

    boolean arbitesPatrolEligibleZone() {
        if (world == null || world.zoneType == null) return false;
        switch (world.zoneType) {
            case NEUTRAL_CIVILIAN_FLOOR:
            case HAB_STACK:
            case SUMP_MARKET:
            case TRAIN_SERVICE_YARD:
            case ADMINISTRATUM_ARCHIVE:
            case IMPERIAL_GUARD_BILLET:
            case SECTOR_GOVERNORS_MANSION:
            case NOBLE_SERVICE_SPINE:
            case IMPERIAL_NEWS_NETWORK:
            case NEUTRAL_RAIL_DEPOT:
            case ARBITES_PRECINCT_EDGE:
                return true;
            default:
                return false;
        }
    }

    boolean arbitesNobleOrCivilZone() {
        if (world == null || world.zoneType == null) return false;
        return arbitesPatrolEligibleZone();
    }

    void applyArbitesPatrolSweep() {
        if (screen == Screen.LOST || screen == Screen.CAPTURE || active == null || world == null) return;
        if (!arbitesNobleOrCivilZone()) return;
        if (turn < arbitesInspectionCooldownUntilTurn) return;
        int pressure = Math.max(0, suspicion) + Math.max(0, gangHeat / 2);
        if (world.zoneType == ZoneType.ARBITES_PRECINCT_EDGE) pressure += 5;
        if (world.zoneType == ZoneType.SECTOR_GOVERNORS_MANSION || world.zoneType == ZoneType.NOBLE_SERVICE_SPINE) pressure += 3;
        if (pressure < 6) return;
        if (turn % 10 != 0) return;
        int chance = Math.min(70, 10 + pressure * 5);
        if (rng.nextInt(100) >= chance) return;
        String reason = "civil/noble-sector patrol sweep in " + world.zoneType.label + " with heat " + gangHeat + " and suspicion " + suspicion;
        lastArbitesPatrolReport = "Civic Wardens patrol initiates inspection: " + reason + ".";
        logEvent("CIVIC WARDENS PATROL: " + lastArbitesPatrolReport);
        handleInterrogation(Faction.CIVIC_WARDENS, reason);
    }

boolean arbitesAuthorityText(String text) {
        if (text == null) return false;
        String l = text.toLowerCase(Locale.ROOT);
        return l.contains("civic Wardens") || l.contains("law enforcement") || l.contains("precinct") || l.contains("suppression") || l.contains("riot");
    }

    void handleArbitesCapture(String reason, int severity) {
        if (screen == Screen.LOST || active == null) return;
        severity = Math.max(1, Math.min(8, severity));
        int startTurn = turn;
        int startCarried = carriedScript;
        int startBanked = totalBankedCash();
        int startBaseStash = baseStashedScript;
        int baseFine = 35 + suspicion * 18 + gangHeat * 7 + severity * 45 + rng.nextInt(90 + severity * 30);
        int finePaid = payArbitesFine(baseFine);
        int unpaidFine = Math.max(0, baseFine - finePaid);
        int jailHours = Math.max(3, Math.min(72, 4 + severity * 3 + rng.nextInt(8 + severity * 4)));
        int statLosses = 1 + (severity >= 4 && rng.nextBoolean() ? 1 : 0);
        ArrayList<String> losses = new ArrayList<>();
        boolean fatalCustodyStatLoss = false;
        for (int i=0; i<statLosses; i++) {
            String loss = applyRandomArbitesStatLoss();
            losses.add(loss);
            if (loss.contains("FATAL CUSTODY TRAUMA")) fatalCustodyStatLoss = true;
        }
        int itemLoss = arbitesEvidenceItemLoss(severity);
        advancePlayerAndWorldTime(jailHours * TURNS_PER_HOUR, "Civic Wardens custody time");
        int fatigueBefore = fatigue, sleepBefore = sleepNeed, painBefore = pain, woundsBefore = wounds;
        fatigue = Math.min(30, fatigue + 4 + severity);
        sleepNeed = Math.min(MAX_SLEEP_NEED, sleepNeed + jailHours * TURNS_PER_HOUR / 2);
        pain = Math.min(24, pain + 2 + severity);
        wounds = Math.min(19, wounds + Math.max(1, severity / 2));
        suspicion = Math.max(0, suspicion - (4 + severity));
        temporaryHostileTurns.remove(Faction.CIVIC_WARDENS);
        arbitesCaptureCooldownUntilTurn = Math.max(arbitesCaptureCooldownUntilTurn, turn + ARBITES_CAPTURE_COOLDOWN_TURNS);
        arbitesInspectionCooldownUntilTurn = Math.max(arbitesInspectionCooldownUntilTurn, turn + ARBITES_INSPECTION_COOLDOWN_TURNS);
        int endBanked = totalBankedCash();
        lastCrimePunishmentReport = "Captured by Civic Wardens for " + reason + ". Hard time " + jailHours + "h, fine assessed " + baseFine + ", collected " + finePaid + ", stat loss " + String.join(", ", losses) + ", evidence items lost " + itemLoss + ".";
        ArrayList<String> detail = new ArrayList<>();
        detail.add("ADEPTUS CIVIC WARDENS CUSTODY REPORT");
        detail.add("Cause: " + reason + ".");
        detail.add("Severity: " + severity + " / 8.");
        detail.add("Time taken: " + jailHours + " local hour(s). Turn " + startTurn + " -> " + turn + ".");
        detail.add("Fine assessed: " + baseFine + " Concord Script. Paid/seized: " + finePaid + ". Unpaid docket remainder: " + unpaidFine + ".");
        detail.add("Carried script: " + startCarried + " -> " + carriedScript + ". Banked script: " + startBanked + " -> " + endBanked + ". Base stash: " + startBaseStash + " -> " + baseStashedScript + ".");
        detail.add("Items confiscated into evidence: " + itemLoss + ".");
        detail.add("Stat damage: " + (losses.isEmpty() ? "none" : String.join(", ", losses)) + ".");
        detail.add("Trauma: fatigue " + fatigueBefore + " -> " + fatigue + ", sleep debt " + sleepBefore + " -> " + sleepNeed + ", pain " + painBefore + " -> " + pain + ", wounds " + woundsBefore + " -> " + wounds + ".");
        detail.add("Street inspection cooldown until: " + TimeSurfaceApi.timeTextAtTurn(arbitesInspectionCooldownUntilTurn) + ".");
        detail.add("Fresh jail-processing cooldown until: " + TimeSurfaceApi.timeTextAtTurn(arbitesCaptureCooldownUntilTurn) + ".");
        if (fatalCustodyStatLoss) detail.add("Outcome: fatal custody trauma. A core stat reached zero.");
        else detail.add("Outcome: released from holding near an Civic Wardens authority point. The paperwork is not apology; it is inventory.");
        lastCustodyReportDetailed = String.join("\n", detail);
        lastInterrogation = lastCrimePunishmentReport;
        if (fatalCustodyStatLoss) {
            lastCrimePunishmentReport += " The beating reduced a core stat to zero; death in police custody recorded.";
            lastInterrogation = lastCrimePunishmentReport;
            lastCustodyReportDetailed += "\nFinal status: death in police custody.";
            logEvent("CIVIC WARDENS CUSTODY DEATH: " + lastCrimePunishmentReport);
            DebugLog.audit("ARBITES_CUSTODY_DEATH", lastCustodyReportDetailed.replace('\n',' ') + " state=" + stateSummary());
            triggerPlayerDeath("death in Civic Wardens police custody after repeated beatings", "Adeptus Civic Wardens custody", "institutional brutality", "precinct holding cell");
            return;
        }
        Point release = nearestArbitesReleasePoint();
        if (release != null) { playerX = release.x; playerY = release.y; lookX = playerX; lookY = playerY; }
        updateSensoryModel("Civic Wardens release");
        logEvent("CIVIC WARDENS CAPTURE: " + lastCrimePunishmentReport);
        DebugLog.audit("ARBITES_CAPTURE", lastCustodyReportDetailed.replace('\n',' ') + " state=" + stateSummary());
        maybeAutosave("Civic Wardens release");
        setScreen(Screen.CAPTURE);
        repaint();
    }

    int payArbitesFine(int amount) {
        int remaining = Math.max(0, amount);
        int paid = 0;
        int fromPocket = Math.min(carriedScript, remaining);
        carriedScript -= fromPocket; paid += fromPocket; remaining -= fromPocket;
        if (remaining > 0) {
            for (BankProfile b : BankProfile.ALL) {
                if (remaining <= 0) break;
                if (b.arbitesFineProtected) continue;
                int bal = bankBalance(b.id);
                if (bal <= 0) continue;
                int take = Math.min(bal, remaining);
                bankBalances.put(b.id, bal - take);
                paid += take; remaining -= take;
            }
        }
        if (remaining > 0 && baseStashedScript > 0 && rng.nextInt(100) < 20) {
            int take = Math.min(baseStashedScript, Math.max(1, remaining / 2));
            baseStashedScript -= take; paid += take; remaining -= take;
        }
        return paid;
    }

    String applyRandomArbitesStatLoss() {
        String[] candidates = {"Strength","Agility","Endurance","Nerve","Charm","Vision","Hearing","Melee","Firearms"};
        String statName = candidates[rng.nextInt(candidates.length)];
        int old = active.stats.getOrDefault(statName, Candidate.statMin(statName));
        if (old <= 0) {
            active.stats.put(statName, 0);
            return statName + " 0->0 FATAL CUSTODY TRAUMA";
        }
        int neu = Math.max(0, old - 1);
        active.stats.put(statName, neu);
        if (neu <= 0) return statName + " " + old + "->0 FATAL CUSTODY TRAUMA";
        return statName + " " + old + "->" + neu;
    }

    int arbitesEvidenceItemLoss(int severity) {
        if (inventory.isEmpty()) return 0;
        int target = Math.min(inventory.size(), Math.max(1, severity / 2));
        String evidence = CONTAINER_FACTION_STOCK_PREFIX + Faction.CIVIC_WARDENS.name();
        ensureContainer(evidence, "Civic Wardens evidence lockup");
        int moved = 0;
        for (int i=0; i<target && !inventory.isEmpty(); i++) {
            int idx = rng.nextInt(inventory.size());
            ItemActionResult r = transferContainerItemResultAt(CONTAINER_PLAYER_INVENTORY, inventory, idx, evidence, "Civic Wardens evidence lockup", null, "confiscated during Civic Wardens capture");
            if (r.success) moved++;
        }
        return moved;
    }

    Point nearestArbitesReleasePoint() {
        if (world == null) return new Point(playerX, playerY);
        Point best = null; int bestD = Integer.MAX_VALUE;
        for (int x=1; x<world.w-1; x++) for (int y=1; y<world.h-1; y++) {
            char ch = world.tiles[x][y];
            boolean precinct = ch == 'p' || ch == 'A' || (world.zoneType == ZoneType.ARBITES_PRECINCT_EDGE && world.walkable(x,y));
            if (!precinct || !world.walkable(x,y) || world.npcAt(x,y) != null) continue;
            int d = Math.abs(x-playerX) + Math.abs(y-playerY);
            if (d < bestD) { bestD = d; best = new Point(x,y); }
        }
        if (best != null) return best;
        Point sp = world.startPoint();
        return new Point(Math.max(1, Math.min(world.w-2, sp.x)), Math.max(1, Math.min(world.h-2, sp.y)));
    }

    void scavengeFor(String target) {
        selectedScavengeTarget = target;
        if (world == null) { logEvent("Cannot scavenge: no world loaded."); return; }
        int roomId = world.roomIdAt(playerX, playerY);
        if (roomId < 0) { logEvent("Scavenge failed: corridors and walls do not have a stable room loot table yet."); return; }
        if (roomId == claimedRoomId) { logEvent("Scavenge blocked: this is your claimed base room. Use storage/workshop systems instead."); return; }
        RoomProfile rp = world.roomProfile(roomId);
        ZoneType zt = world.zoneType;
        int key = world.locationKey() * 1000 + roomId;
        int readyTurn = scavengeCooldownUntilTurn.getOrDefault(key, -1);
        if (readyTurn > turn && !target.equals("Beg for money")) {
            int remain = readyTurn - turn;
            lastScavengeReport = "This room was already picked clean. It will need roughly " + remain + " turns (~" + Math.max(1, remain/6) + " hours) before meaningful scavenge returns.";
            logEvent(lastScavengeReport);
            showTimedAlert("SCAVENGE RESULT", lastScavengeReport, 3600);
            DebugLog.audit("SCAVENGE_COOLDOWN", "target="+target+" room=" + roomId + " readyTurn=" + readyTurn + " now=" + turn + " state=" + stateSummary());
            return;
        }
        if (target.equals("Beg for money")) {
            begForImperialScript(rp, zt);
            return;
        }
        int targetBonus = target.equals("General supplies") ? 0 : 10;
        int chance = rp.scavengeChance + zt.scavengeBonus + stat("Vision", 5) + stat("Mechanics", 5) / 2 + targetBonus;
        chance = Math.max(5, Math.min(94, chance));
        int roll = rng.nextInt(100) + 1;
        int turns = Math.max(2, 8 - stat("Agility", 5) / 2);
        DebugLog.audit("SCAVENGE_TARGET", "target="+target+" room=" + roomId + " type=" + rp.name + " zone=" + zt.label + " roll=" + roll + " chance=" + chance + " turns=" + turns + " state=" + stateSummary());
        if (roll <= chance) {
            ItemActionResult found = scavengeFromRoomCache(target, roomId, rp, zt);
            String loot = found.success ? found.itemName : targetedLoot(target, rp, zt);
            if (!found.success) addInventoryItem(loot, ItemProvenanceRecord.found(loot, world, turn, "scavenged fallback " + rp.name + " in " + zt.label));
            if (loot.toLowerCase(Locale.ROOT).contains("scrap") || loot.toLowerCase(Locale.ROOT).contains("machine") || loot.toLowerCase(Locale.ROOT).contains("wire")) machineParts++;
            if (loot.toLowerCase(Locale.ROOT).contains("ration") || loot.toLowerCase(Locale.ROOT).contains("food")) food = Math.min(MAX_FOOD_WATER, food + 1);
            if (loot.toLowerCase(Locale.ROOT).contains("water") || loot.toLowerCase(Locale.ROOT).contains("canteen")) water = Math.min(MAX_FOOD_WATER, water + 1);
            lastScavengeReport = "ITEM FOUND: Scavenged for " + target + " in " + rp.name + ": found " + loot + " (" + roll + " <= " + chance + ").";
            gainXp("Scavenging", 2, "successful room scavenge");
        } else {
            lastScavengeReport = "Scavenged for " + target + " in " + rp.name + ": nothing useful (" + roll + " > " + chance + ").";
            gainXp("Scavenging", 1, "failed room scavenge");
        }
        logEvent(lastScavengeReport);
        showTimedAlert("SCAVENGE RESULT", lastScavengeReport, 3600);
        scavengeCooldownUntilTurn.put(key, turn + SCAVENGE_COOLDOWN_TURNS);
        for (int i=0;i<turns;i++) advanceTurn("scavenges for " + target.toLowerCase(Locale.ROOT) + ".");
        updatePendingInteractionSummary();
        repaint();
    }

    void ensureRoomCacheSeeded(int roomId, RoomProfile rp, ZoneType zt) {
        String id = roomCacheContainerId(roomId);
        if (itemContainers.containsKey(id)) return;
        ensureContainer(id, roomCacheContainerLabel(roomId, rp, zt));
        boolean hivewallCache = HivewallRoomCacheApi.isHivewallRoom(rp);
        ZoneFacilityLedgerEntry linkedFacility = hivewallCache ? null : FacilityLinkedRoomCacheApi.matchFacilityForRoom(world, rp, zt);
        ArrayList<String> seeded = hivewallCache
                ? RoomContainerApi.seedCacheItems(rp, zt, rng)
                : FacilityLinkedRoomCacheApi.seedCacheItems(rp, zt, rng, linkedFacility, world);
        if (!hivewallCache) ProductionFacilityOutputSimulationApi.addMatchingOutputsToCache(seeded, world, linkedFacility, rng);
        if (!hivewallCache) ProductionDistributionApi.addMatchingMovementSamplesToCache(seeded, world, linkedFacility, rp, rng);
        if (!hivewallCache) HistoricalItemMaterializationApi.addMaterializedSamplesToCache(seeded, world, linkedFacility, rp, rng);

            ItemProvenanceRecord prov = hivewallCache
                    ? HivewallRoomCacheApi.cacheProvenance(item, world, roomId, rp, turn)
                    : FacilityLinkedRoomCacheApi.cacheProvenance(item, world, roomId, rp, zt, linkedFacility, turn);
            addItemToContainerResult(id, roomCacheContainerLabel(roomId, rp, zt), item,
                    prov, null, hivewallCache ? "seed hivewall/interwall persistent cache" : "seed facility-linked persistent room cache");
        }
        DebugLog.audit(hivewallCache ? "HIVEWALL_ROOM_CACHE" : "FACILITY_LINKED_ROOM_CACHE", "seeded roomCache=" + id + " room=" + roomId + " zone=" + (zt==null?"none":zt.label) + " profile=" + (rp==null?"none":rp.name) + " facility=" + (linkedFacility==null?"none":linkedFacility.summary()) + " items=" + seeded);
