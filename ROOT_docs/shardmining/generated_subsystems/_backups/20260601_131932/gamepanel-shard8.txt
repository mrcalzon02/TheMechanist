    void chooseGeneratedAssetPayloadRoot() {
        try {
            JFileChooser chooser = new JFileChooser(options.hasGeneratedAssetPayloadRoot() ? new File(options.generatedAssetPayloadRoot()) : new File("."));
            chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
            chooser.setDialogTitle("Select The Mechanist generated-art payload root");
            int result = chooser.showOpenDialog(this);
            if (result != JFileChooser.APPROVE_OPTION || chooser.getSelectedFile() == null) {
                logEvent("Generated-art payload root unchanged.");
                return;

            File selected = chooser.getSelectedFile();
            options.generatedAssetPayloadRoot = selected.getAbsolutePath();
            options.applyGeneratedAssetRuntimeProperties();

    boolean isGraphicsDropdownButton(ButtonBox b) {
        if (b == null || b.label == null) return false;
        String label = b.label.trim();
        if (label.startsWith("> ")) label = label.substring(2).trim();
        if (graphicsDropdown == 0) {
            return label.equals("Windowed") || label.equals("Borderless Windowed") || label.equals("Exclusive Fullscreen");
        }
        if (graphicsDropdown == 1) {
            for (int i=0;i<DisplayResolutionAuthority.choiceCount();i++) if (label.equals(DisplayResolutionAuthority.modeLabel(i))) return true;
            return false;
        }
        if (graphicsDropdown == 2) {
            for (String name : GameOptions.PALETTE_NAMES) if (label.equals(name)) return true;
            return false;
        }
        if (graphicsDropdown == 3) {
            for (String name : GameOptions.DOWNSCALE_LABELS) if (label.equals(name)) return true;
            return false;
        }
        if (graphicsDropdown == 4) {
            for (String name : GameOptions.TARGET_FPS_LABELS) if (label.equals(name)) return true;
            return false;
        }
        if (graphicsDropdown == 5) {
            for (String name : GameOptions.RENDER_QUALITY_LABELS) if (label.equals(name)) return true;
            return false;
        }
        return false;
    }

    void addGraphicsDropdownButtons(int x, int y, int bw, int gap) {
        if (graphicsDropdown < 0) return;
        Rectangle inner = graphicsDropdownInnerRect();
        if (inner.width <= 0 || inner.height <= 0) return;
        int rowH = Math.max(20, Math.min(30, scaled(27)));
        int yy = inner.y;
        if (graphicsDropdown == 0) {
            String[] modes = {"Windowed", "Borderless Windowed", "Exclusive Fullscreen"};
            for (int i=0;i<modes.length;i++) { final int idx=i; buttons.add(new ButtonBox(modes[i], inner.x, yy+i*rowH, inner.width, rowH-2, "Select " + modes[i] + " window mode.", () -> setWindowMode(idx))); }
        } else if (graphicsDropdown == 1) {
            int available = DisplayResolutionAuthority.choiceCount();
            int visible = Math.max(1, Math.min(available, inner.height / Math.max(1, rowH)));
            int start = Math.max(0, Math.min(Math.max(0, available - visible), options.resolutionIndex - visible/2));
            for (int row=0; row<visible; row++) {
                final int idx=start+row;
                String prefix = idx==options.resolutionIndex ? "> " : "";
                buttons.add(new ButtonBox(prefix + DisplayResolutionAuthority.modeLabel(idx), inner.x, yy+row*rowH, inner.width, rowH-2, "Select detected/safe resolution " + DisplayResolutionAuthority.modeLabel(idx) + ".", () -> setResolutionIndex(idx)));
            }
        } else if (graphicsDropdown == 2) {
            for (int i=0;i<GameOptions.PALETTE_NAMES.length;i++) { final int idx=i; buttons.add(new ButtonBox(GameOptions.PALETTE_NAMES[i], inner.x, yy+i*rowH, inner.width, rowH-2, "Apply palette globally: " + GameOptions.PALETTE_NAMES[i] + ".", () -> setColorPreset(idx))); }
        } else if (graphicsDropdown == 3) {
            for (int i=0;i<GameOptions.DOWNSCALE_LABELS.length;i++) { final int idx=i; buttons.add(new ButtonBox(GameOptions.DOWNSCALE_LABELS[i], inner.x, yy+i*rowH, inner.width, rowH-2, "Use internal render scale " + GameOptions.DOWNSCALE_LABELS[i] + ".", () -> setDownscaleIndex(idx))); }
        } else if (graphicsDropdown == 4) {
            for (int i=0;i<GameOptions.TARGET_FPS_LABELS.length;i++) { final int idx=i; buttons.add(new ButtonBox(GameOptions.TARGET_FPS_LABELS[i], inner.x, yy+i*rowH, inner.width, rowH-2, "Use target frame pacing " + GameOptions.TARGET_FPS_LABELS[i] + ".", () -> setTargetFpsIndex(idx))); }
        } else if (graphicsDropdown == 5) {
            for (int i=0;i<GameOptions.RENDER_QUALITY_LABELS.length;i++) { final int idx=i; buttons.add(new ButtonBox(GameOptions.RENDER_QUALITY_LABELS[i], inner.x, yy+i*rowH, inner.width, rowH-2, "Use render quality profile " + GameOptions.RENDER_QUALITY_LABELS[i] + ".", () -> setRenderQualityIndex(idx))); }
        }
    }

    void acceptJvmSettingsAndRestart() {
        JvmRuntimeProfileAuthority.save(jvmRuntimeProfile);
        java.util.List<String> command = new ArrayList<>();
        String javaExe = Paths.get(System.getProperty("java.home", ""), "bin", isWindowsHost() ? "java.exe" : "java").toString();
        command.add(javaExe);
        command.addAll(jvmRuntimeProfile.buildJvmArgs());
        command.add("-cp");
        command.add(System.getProperty("java.class.path"));
        command.add(TheMechanist.class.getName());
        DebugLog.audit("JVM_RESTART", "Restarting with command=" + command);
        logEvent("Restarting with accepted JVM profile: " + jvmRuntimeProfile.targetLabel() + ".");
        try {
            new ProcessBuilder(command).inheritIO().start();
            shutdownRuntime();
            Window win = SwingUtilities.getWindowAncestor(this);
            if (win != null) win.dispose();
            System.exit(0);
        } catch (Throwable t) {
            DebugLog.error("JVM_RESTART", "Failed to restart with selected JVM profile.", t);
            jvmRuntimeNotice = "Restart failed: " + t.getMessage();
            logEvent(jvmRuntimeNotice);
            repaint();
        }
    }

    ArrayList<String> controlsReferenceLines() {
        ArrayList<String> lines = new ArrayList<>();
        lines.add(controlProfileTitle());
        lines.add("Action names are shared across keyboard and controller paths; context decides whether a prompt moves, examines, interacts, confirms, or backs out.");
        lines.add("Required recovery actions: Move, Confirm, Cancel/Back, Pause/Menu, and a keyboard/mouse fallback.");
        for (InputAction action : InputAction.values()) {
            lines.add(inputActionLabel(action) + " | " + inputActionContext(action) + " | "
                    + keyboardPromptFor(action) + " | " + controllerPromptFor(action)
                    + (requiredInputAction(action) ? " | required" : ""));
        }
        lines.add("Context overlap: E examines while the Look panel is open and interacts from the game surface; A/Cross confirms in menus and examines/interacts only where that panel owns the prompt.");
        lines.add("Safe fallback: Escape/B/Circle backs out, and keyboard movement remains available if a controller is missing or reconnecting.");
        if (rebindingTarget != null && !rebindingTarget.isEmpty()) lines.add(rebindingTarget);
        return lines;
    }

    void drawClippedWrappedLines(Graphics2D g, java.util.List<String> lines, int x, int y, int w, int h, Font font, boolean centered) {
        if (lines == null || w <= 4 || h <= 4) return;
        Font oldFont = g.getFont();
        Shape oldClip = g.getClip();
        g.setFont(font == null ? smallFont : font);
        FontMetrics fm = g.getFontMetrics();
        int lineH = Math.max(12, fm.getHeight() + 1);
        g.setClip(x, y, Math.max(1, w), Math.max(1, h));
        int yy = y + fm.getAscent();
        for (String line : TextLayoutAuthority.wrapAllPixels(g, g.getFont(), lines, Math.max(8, w - 4))) {
            if (yy > y + h) { g.setClip(oldClip); g.setFont(oldFont); return; }
            g.setColor(optionColor(GameOptions.TEXT_MAIN));
            if (centered) g.drawString(line, x + (w - fm.stringWidth(line)) / 2, yy); else g.drawString(line, x, yy);
            yy += lineH;
        }
        g.setClip(oldClip);
        g.setFont(oldFont);
    }

    void drawFrameWrappedCenteredText(Graphics2D g, String text, int x, int y, int w, int h, Font font) {
        if (text == null) text = "";
        Font oldFont = g.getFont();
        g.setFont(font == null ? uiFont : font);
        FontMetrics fm = g.getFontMetrics();
        java.util.List<String> lines = TextLayoutAuthority.wrapPixels(g, g.getFont(), text, Math.max(8, w - 8));
        int lineH = Math.max(fm.getHeight() + 6, 22);
        int totalH = lines.size() * lineH;
        int yy = y + Math.max(fm.getAscent(), (h - totalH) / 2 + fm.getAscent());
        for (String line : lines) {
            if (yy > y + h) break;
            g.setColor(optionColor(GameOptions.TEXT_MAIN));
            center(g, line, x + w / 2, yy);
            yy += lineH;
        }
        g.setFont(oldFont);
    }

    void drawTextPanel(Graphics2D g, int x, int y, int w, int h, java.util.List<String> lines, boolean centered) {

    }

    void drawHoverHelp(Graphics2D g, int hoverButtonIndex) {
        if (!options.hoverHelp) return;
        String help = null;
        if (hoverButtonIndex >= 0 && hoverButtonIndex < buttons.size()) help = buttons.get(hoverButtonIndex).tip;
        else if (mouseX < 0 && selectedButton >= 0 && selectedButton < buttons.size() && buttonIsModalInteractive(buttons.get(selectedButton))) help = buttons.get(selectedButton).tip;
        if (help == null || mouseX < 0 || mouseY < 0) return;
        g.setFont(smallFont);
        int maxChars = 34;
        java.util.List<String> lines = TextSurfaceApi.wrap(help, maxChars);
        FontMetrics fm = g.getFontMetrics();
        int tw = 0;
        for (String line : lines) tw = Math.max(tw, fm.stringWidth(line));
        int boxW = Math.min(340, Math.max(190, tw + 22));
        int lineH = fm.getHeight();
        int boxH = Math.max(34, lines.size()*lineH + 14);
        int x = mouseX + 18;
        int y = mouseY + 18;
        if (x + boxW > getWidth() - 12) x = mouseX - boxW - 18;
        if (y + boxH > getHeight() - 12) y = mouseY - boxH - 18;
        x = Math.max(8, x); y = Math.max(8, y);
        g.setColor(new Color(8,8,9,238)); g.fillRect(x, y, boxW, boxH);
        g.setColor(new Color(165,138,82)); g.drawRect(x, y, boxW, boxH);
        g.setColor(new Color(220,205,160));
        int ty = y + 18;
        for (String line : lines) { drawUiTextLine(g, line, x+11, ty); ty += lineH; }
    }

    void drawUiTextLine(Graphics2D g, String s, int x, int y) {
        if (s == null) return;
        FontMetrics fm = g.getFontMetrics();
        int tw = fm.stringWidth(s);
        int th = fm.getHeight();
        Color old = g.getColor();
        g.setColor(new Color(0, 0, 0, 196));
        g.fillRoundRect(x - 5, y - fm.getAscent() - 3, tw + 10, th + 5, 7, 7);
        g.setColor(new Color(128, 105, 58, 135));
        g.drawRoundRect(x - 5, y - fm.getAscent() - 3, tw + 10, th + 5, 7, 7);
        g.setColor(old);
        g.drawString(s, x, y);
    }

    void center(Graphics2D g, String s, int x, int y) {
        FontMetrics fm = g.getFontMetrics();
        drawUiTextLine(g, s, x - fm.stringWidth(s)/2, y);
    }

    boolean buttonIsModalInteractive(ButtonBox b) {
        if (b == null) return false;
        if (screen == Screen.OPTIONS && graphicsDropdown >= 0) return isGraphicsDropdownButton(b);
        if (screen == Screen.SECTOR_AUDIT && auditZoneDropdownOpen) return isZoneAuditDropdownButton(b) || (b.label != null && b.label.startsWith("ZONE:"));
        return true;
    }

    int activeHoverButtonIndex() {
        if (mouseX < 0 || mouseY < 0) return -1;
        // Topmost buttons win. When a dropdown is open it is modal for hover and
        // help text as well as clicks, so obscured controls underneath cannot glow.
        for (int i=buttons.size()-1; i>=0; i--) {
            ButtonBox b = buttons.get(i);
            if (!buttonIsModalInteractive(b)) continue;
            if (b.contains(mouseX, mouseY)) return i;
        }
        return -1;
    }

    void activateSelectedButton() {
        if (!buttons.isEmpty() && selectedButton >=0 && selectedButton < buttons.size()) {
            ButtonBox b = buttons.get(selectedButton);
            if (!buttonIsModalInteractive(b)) { sounds.play("panelClose", options); return; }
            sounds.play("button", options);
            runGuarded("BUTTON", "activate selected button " + b.label, b.action);
        }
    }


    void targetPanelCursorFromMouse(int tx, int ty) {
        if (world == null || !world.inBounds(tx, ty)) return;
        if (panelMode == PanelMode.LOOK) {
            lookCursorActive = true;
            lookX = tx; lookY = ty;
            lookStackIndex = 0; lookStackScroll = 0;
            activeScrollTag = "look-stack";
        } else if (panelMode == PanelMode.COMBAT) {
            combatCursorActive = true;
            combatX = tx; combatY = ty;
            lastTargetingReport = targetingSolutionAt(tx, ty).summary;
        } else if (panelMode == PanelMode.INTERACT) {
            interactCursorActive = true;
            lookX = tx; lookY = ty;
            clampInteractCursorToAdjacent();
            lookStackIndex = 0;
            updatePendingInteractionSummary();
        }
        DebugLog.audit("MOUSE_TARGETING", "panel=" + panelMode + " tile=" + tx + "," + ty + " source=virtual-map");
    }

    @Override public void mouseClicked(MouseEvent e) {
        lastInputMillis=System.currentTimeMillis();
        if (screen == Screen.INTRO_CRAWL) { continueFromIntroCrawl(); return; }
        Point mp = mapInputPoint(e);
        if (mp == null) return;
        int mx = mp.x, my = mp.y;
        if (screen==Screen.ZONE_SPLASH) { continueFromZoneSplash(); repaint(); requestFocusInWindow(); return; }
        if (screen == Screen.EDITOR && (SwingUtilities.isLeftMouseButton(e) || SwingUtilities.isRightMouseButton(e)) && handleInGameEditorGridPaint(mx, my, SwingUtilities.isRightMouseButton(e))) { requestFocusInWindow(); return; }
        if (screen == Screen.SECTOR_AUDIT && auditZoneDropdownOpen) {
            for (int i=buttons.size()-1;i>=0;i--) {
                ButtonBox b = buttons.get(i);
                boolean zoneSelectorControl = isZoneAuditDropdownButton(b) || (b != null && b.label != null && b.label.startsWith("ZONE:"));
                if (zoneSelectorControl && b.contains(mx, my)) {
                    selectedButton=i;
                    sounds.play("button", options);
                    runGuarded("MOUSE", "click zone audit selector " + b.label, b.action);
                    repaint();
                    requestFocusInWindow();
                    return;
                }
            }
            Rectangle dd = zoneAuditDropdownOuterRect(uiLayout());
            if (dd.contains(mx, my)) { repaint(); requestFocusInWindow(); return; }
            auditZoneDropdownOpen = false;
            repaint();
            requestFocusInWindow();
            return;
        }
        if (screen == Screen.SECTOR_AUDIT && auditWorld != null) {
            Point tile = screenPointToAuditTile(mx, my);
            if (tile != null) { auditCursorX = tile.x; auditCursorY = tile.y; repaint(); requestFocusInWindow(); return; }
        }
        if (screen == Screen.GAME && world != null && firstPersonRenderViewport.handleMouseClicked(this, e, mx, my)) {
            requestFocusInWindow();
            return;
        }
        if (screen == Screen.GAME && world != null) {
            Point tile = screenPointToWorldTile(mx, my);
            if (SwingUtilities.isRightMouseButton(e) && tile != null) {
                if (tile.x == playerX && tile.y == playerY) {
                    clearMouseMovementPreview("right-click self opens look targeting");
                    interactCursorActive=false; lookX=playerX; lookY=playerY; lookCursorActive=true; openPanel(PanelMode.LOOK);
                    repaint(); requestFocusInWindow(); return;
                }
                if (mouseMovePreviewActive) clearMouseMovementPreview("right-click toggle");
                else updateMouseMovementPreviewTo(tile.x, tile.y);
                repaint(); requestFocusInWindow(); return;
            }
            if (SwingUtilities.isLeftMouseButton(e) && mouseMovePreviewActive) {
                executeMouseMovementPreview();
                requestFocusInWindow(); return;
            }
        }
        if (screen == Screen.PANEL && world != null && (panelMode == PanelMode.LOOK || panelMode == PanelMode.COMBAT || panelMode == PanelMode.INTERACT)) {
            Point tile = screenPointToWorldTile(mx, my);
            if (tile != null && (SwingUtilities.isLeftMouseButton(e) || SwingUtilities.isRightMouseButton(e))) {
                targetPanelCursorFromMouse(tile.x, tile.y);
                repaint(); requestFocusInWindow(); return;
            }
        }
        if (screen == Screen.PANEL && panelMode == PanelMode.INVENTORY && handleInventoryStackPanelClick(mx, my)) { requestFocusInWindow(); return; }
        if (findScrollRegion(mx,my) != null) { handleScrollbarClick(mx,my); repaint(); requestFocusInWindow(); return; }
        if (screen == Screen.KNOWLEDGE && handleKnowledgeTreeClick(mx, my)) { requestFocusInWindow(); return; }
        if (screen==Screen.OPTIONS && graphicsDropdown>=0) {
            // Open dropdowns are modal blockers. They must consume the click before
            // any button underneath them can receive it. This prevents selecting a
            // dropdown value while also triggering an obscured option button.
            for (int i=buttons.size()-1;i>=0;i--) {
                ButtonBox b = buttons.get(i);
                if (isGraphicsDropdownButton(b) && b.contains(mx, my)) {
                    selectedButton=i;
                    sounds.play("button", options);
                    runGuarded("MOUSE", "click dropdown " + b.label, b.action);
                    repaint();
                    return;
                }
            }
            // Clicked outside the dropdown: close it and swallow the click.
            graphicsDropdown=-1;
            repaint();
            requestFocusInWindow();
            return;
        }
        if (screen == Screen.CHARACTER) {
            if (characterNameEditRect.contains(mx, my)) {
                characterNameEditActive = true;
                requestFocusInWindow();
                repaint();
                return;
            } else if (characterNameEditActive) {
                Candidate c = candidates.isEmpty() ? null : candidates.get(candidateIndex);
                if (c != null) { c.name = CharacterCreationAuthority.sanitizePlayerName(c.name, rng); refreshNameLockedCandidateState(c); }
                characterNameEditActive = false;
            }
        }
        for (int i=buttons.size()-1;i>=0;i--) if (buttonIsModalInteractive(buttons.get(i)) && buttons.get(i).contains(mx, my)) { selectedButton=i; sounds.play("button", options); runGuarded("MOUSE", "click button " + buttons.get(i).label, buttons.get(i).action); repaint(); return; }
        requestFocusInWindow();
    }

    }

    void handleKeyPressed(KeyEvent e) {
        lastInputMillis=System.currentTimeMillis();
        int code=e.getKeyCode();
        if (eulaGateActive) {
            if (code == KeyEvent.VK_ENTER || code == KeyEvent.VK_A) { acceptEulaGate(); repaint(); return; }
            if (code == KeyEvent.VK_ESCAPE || code == KeyEvent.VK_Q || code == KeyEvent.VK_X) { requestApplicationExit("EULA declined by key"); return; }

            if (code == KeyEvent.VK_PAGE_UP) { scrollEulaGate(-1, true); return; }
            if (code == KeyEvent.VK_DOWN || code == KeyEvent.VK_S) { scrollEulaGate(1, false); return; }
            if (code == KeyEvent.VK_UP || code == KeyEvent.VK_W) { scrollEulaGate(-1, false); return; }
            if (code == KeyEvent.VK_HOME) { eulaScroll = 0; repaint(); return; }
            if (code == KeyEvent.VK_END) { eulaScroll = Math.max(0, eulaMaxScroll); repaint(); return; }
            return;
        }
        if (code == KeyEvent.VK_F3) { togglePerformanceDiagnostics(); return; }
        if (code == KeyEvent.VK_F1 && (screen == Screen.GAME || screen == Screen.PANEL)) { toggleTacticalSlate(); return; }
        if (code == KeyEvent.VK_Y && screen != Screen.EDITOR) { openChatWindow(); return; }
        if (screen == Screen.GAME && firstPersonRenderViewport.handleKeyPressed(this, code)) { repaint(); return; }
        if (worldZoomControlActive() && (code == KeyEvent.VK_HOME || code == KeyEvent.VK_END)) {
            changeWorldZoom(code == KeyEvent.VK_HOME ? 1 : -1, code == KeyEvent.VK_HOME ? "HOME" : "END");
            return;
        }
        if (screen==Screen.INTRO_CRAWL) {
            if (code==KeyEvent.VK_ENTER || code==KeyEvent.VK_SPACE || code==KeyEvent.VK_ESCAPE) { continueFromIntroCrawl(); repaint(); }
            return;
        }
        if (screen==Screen.ZONE_SPLASH) {
            if (code==KeyEvent.VK_ENTER || code==KeyEvent.VK_SPACE || code==KeyEvent.VK_ESCAPE) { continueFromZoneSplash(); repaint(); }
            return;
        }
        if (screen==Screen.CAPTURE) {
            if (code==KeyEvent.VK_ENTER || code==KeyEvent.VK_SPACE || code==KeyEvent.VK_ESCAPE || code==KeyEvent.VK_E) { setScreen(Screen.GAME); repaint(); }
            return;
        }
        if (screen==Screen.BOOT) {
            if (code==KeyEvent.VK_ESCAPE || code==KeyEvent.VK_ENTER || code==KeyEvent.VK_SPACE) { sounds.play("button", options); finishBootSequence("key"); }
            return;
        }
        if (screen == Screen.CHARACTER && characterNameEditActive) {
            Candidate c = candidates.isEmpty() ? null : candidates.get(candidateIndex);
            if (code == KeyEvent.VK_ESCAPE || code == KeyEvent.VK_ENTER) {
                if (c != null) { c.name = CharacterCreationAuthority.sanitizePlayerName(c.name, rng); refreshNameLockedCandidateState(c); }
                characterNameEditActive = false;
                repaint();
                return;
            }
            if (code == KeyEvent.VK_BACK_SPACE) {
                if (c != null && c.name != null && !c.name.isEmpty()) { c.name = c.name.substring(0, c.name.length()-1); refreshNameLockedCandidateState(c); }
                repaint();
                return;
            }
            if (code == KeyEvent.VK_DELETE) {
                if (c != null) { c.name = ""; refreshNameLockedCandidateState(c); }
                repaint();
                return;
            }
            return;

    void processUnifiedInputRegistry(long now) {
        // Keyboard retains the existing direct KeyEvent path. This timer consumer only
        // translates GAMEPAD source states so a controller release cannot erase a held key.
        if (inputRegistry.consumePressed(InputAction.PAUSE, InputSource.GAMEPAD) || inputRegistry.consumePressed(InputAction.CANCEL, InputSource.GAMEPAD)) {
            if (screen==Screen.GAME && manualMovementPlanActive) cancelManualMovementPlan("gamepad cancel");
            else if (screen==Screen.GAME) setScreen(Screen.PAUSE);
            else if (screen==Screen.PANEL) closePanel();
            else if (screen==Screen.PAUSE) setScreen(Screen.GAME);
            else setScreen(Screen.MENU);
            repaint();
            return;
        }
        if (inputRegistry.consumePressed(InputAction.CONFIRM, InputSource.GAMEPAD)) {
            if (screen == Screen.GAME && manualMovementPlanActive) confirmManualMovementPlan();
            else if (screen == Screen.INTRO_CRAWL) continueFromIntroCrawl();
            else if (screen == Screen.ZONE_SPLASH) continueFromZoneSplash();
            else activateSelectedButtonUniversal();
            repaint();
        }
        if (inputRegistry.consumePressed(InputAction.INVENTORY, InputSource.GAMEPAD) && screen == Screen.GAME) { openPanel(PanelMode.INVENTORY); return; }
        if (inputRegistry.consumePressed(InputAction.CHARACTER, InputSource.GAMEPAD) && screen == Screen.GAME) { openPanel(PanelMode.CHARACTER); return; }
        if (inputRegistry.consumePressed(InputAction.LOOK, InputSource.GAMEPAD) && screen == Screen.GAME) { interactCursorActive=false; lookX=playerX; lookY=playerY; lookCursorActive=true; openPanel(PanelMode.LOOK); return; }
        if (inputRegistry.consumePressed(InputAction.PLAN_MOVE, InputSource.GAMEPAD) && screen == Screen.GAME) { beginManualMovementPlan(); return; }
        if (inputRegistry.consumePressed(InputAction.ZOOM_IN, InputSource.GAMEPAD) && worldZoomControlActive()) changeWorldZoom(1, "gamepad bumper");
        if (inputRegistry.consumePressed(InputAction.ZOOM_OUT, InputSource.GAMEPAD) && worldZoomControlActive()) changeWorldZoom(-1, "gamepad bumper");
        if (now < nextUnifiedInputRepeatMillis) return;
        int dx = 0, dy = 0;
        if (inputRegistry.isActiveFromSource(InputAction.MOVE_LEFT, InputSource.GAMEPAD)) dx--;
        if (inputRegistry.isActiveFromSource(InputAction.MOVE_RIGHT, InputSource.GAMEPAD)) dx++;
        if (inputRegistry.isActiveFromSource(InputAction.MOVE_UP, InputSource.GAMEPAD)) dy--;
        if (inputRegistry.isActiveFromSource(InputAction.MOVE_DOWN, InputSource.GAMEPAD)) dy++;
        if (dx == 0 && dy == 0) return;
        nextUnifiedInputRepeatMillis = now + 125L;
        if (screen == Screen.GAME && world != null && manualMovementPlanActive) {
            nudgeManualMovementPlan(Integer.compare(dx, 0), Integer.compare(dy, 0));
        } else if (screen == Screen.GAME && world != null) {
            queueOrExecuteMovementInput(Integer.compare(dx, 0), Integer.compare(dy, 0));
        } else if (screen == Screen.PANEL && panelMode == PanelMode.LOOK && world != null) {
            lookCursorActive = true;
            lookX = Math.max(0, Math.min(world.w-1, lookX+Integer.compare(dx, 0)));
            lookY = Math.max(0, Math.min(world.h-1, lookY+Integer.compare(dy, 0)));
            lookStackIndex = 0;
        } else if (screen == Screen.PANEL && panelMode == PanelMode.INTERACT && world != null) {
            moveInteractCursor(Integer.compare(dx, 0), Integer.compare(dy, 0));
        } else if (screen == Screen.PANEL && panelMode == PanelMode.COMBAT && world != null) {
            moveCombatCursor(Integer.compare(dx, 0), Integer.compare(dy, 0));
        } else if (screen == Screen.PANEL && (panelMode == PanelMode.BUILD || panelMode == PanelMode.WORKBENCH) && buildPlacementActive && world != null) {
            moveBuildCursor(Integer.compare(dx, 0), Integer.compare(dy, 0));
        } else if (!buttons.isEmpty()) {
            moveSelectedButton(dy < 0 || dx < 0 ? -1 : 1);
        }
        repaint();
    }

    void auditLoadingWatchdog() {
        long now = System.currentTimeMillis();
        long sinceStart = now - loadingStartMillis;
        long sincePulse = now - lastLoadingPulseMillis;
        if (loadingStep == 0 && sinceStart > 1200L && !loadingWatchdogReported) {
            loadingWatchdogReported = true;
            loadingWatchdogRepeats++;
            DebugLog.warn("LOADING_WATCHDOG_ZERO", "Loader remained at step zero for " + sinceStart + "ms. Forcing an audited progress tick. status=" + loadingStatus + " detail=" + loadingDetail);
            loadingStatus = "Watchdog forced loader heartbeat.";
            loadingDetail = "The loader had not advanced from step zero; an audited progress tick was forced so the UI cannot silently stall.";
            try { progressLoading(); } catch (Throwable t) { DebugLog.error("LOADING_WATCHDOG_FORCE_FAILED", "Forced loading heartbeat failed.", t); }
        } else if (loadingFinalizationRunning && now - loadingFinalizationStartMillis > 3000L && sincePulse > 2200L) {
            loadingWatchdogRepeats++;
            DebugLog.warn("LOADING_FINALIZER_WATCHDOG", "Background finalizer still running kind=" + loadingFinalizationKind + " elapsedMs=" + (now - loadingFinalizationStartMillis) + " step=" + loadingStep + " progress=" + loadingProgress + " repeat=" + loadingWatchdogRepeats);
            loadingStatus = "Still finalizing " + loadingFinalizationKind + "...";
            loadingDetail = "The background finalizer is alive. If this repeats for too long, diagnostic details will show the stuck phase rather than silently freezing.";
            lastLoadingPulseMillis = now;
        } else if (sincePulse > 2200L && sinceStart > 2200L && loadingProgress < 100) {
            loadingWatchdogRepeats++;
            DebugLog.warn("LOADING_WATCHDOG_STALL", "Loader pulse stale for " + sincePulse + "ms at step=" + loadingStep + " progress=" + loadingProgress + ". repeat=" + loadingWatchdogRepeats);
            lastLoadingPulseMillis = now;
        }
    }

    @Override public void actionPerformed(ActionEvent e) {
        long framePulseNanos = System.nanoTime();
        if (!frameLimiter.shouldRunPulse(framePulseNanos)) return;
        if (eulaGateActive) { repaint(); return; }
        if (screen == Screen.LOADING) {
            try { progressLoading(); }
            catch (Throwable t) {
                DebugLog.error("LOADING_PROGRESS_FAILED", "Loader progress tick failed at step=" + loadingStep + " progress=" + loadingProgress + " status=" + loadingStatus, t);
                loadingStatus = "Loader fault caught; returning to menu.";
                loadingDetail = "Diagnostic details were recorded. The watchdog caught the failure instead of silently freezing at zero.";
                setScreen(Screen.MENU);
            }
        }
        if (screen == Screen.LOADING) auditLoadingWatchdog();
        long now = System.currentTimeMillis();
        if (now - lastMusicDirectiveMillis > 1000L) { lastMusicDirectiveMillis = now; updateDynamicMusic("timer"); }
        if (screen == Screen.BOOT && options.bootSound && ((System.currentTimeMillis() - bootStartMillis) / 950) != bootStage) { bootStage = (int)((System.currentTimeMillis() - bootStartMillis) / 950); sounds.play("type", options); }
        if (screen == Screen.BOOT && System.currentTimeMillis() - bootStartMillis > 15800) finishBootSequence("timer");
        processUnifiedInputRegistry(now);
        firstPersonRenderViewport.updateContinuousMotion(this);
        processPendingMovementInput();
        singlePlayerSectorBridge.maybeRunContinuousTick(this);
        spendDeferredSimulationBudget((screen == Screen.GAME ? "active-timer" : "idle-ui-timer"), screen == Screen.GAME ? ACTIVE_SIMULATION_BUDGET_MS : IDLE_SIMULATION_BUDGET_MS);
        maybePlayCoreAmbientCue(now);
        pruneTileMotions();
        pruneCombatAnimations(now);
        repaint();
    }
    @Override public void mousePressed(MouseEvent e) {
        requestFocusInWindow();
        Point mp = mapInputPoint(e);
        if (mp == null) return;
        if (screen == Screen.EDITOR && (SwingUtilities.isLeftMouseButton(e) || SwingUtilities.isRightMouseButton(e)) && handleInGameEditorGridPaint(mp.x, mp.y, SwingUtilities.isRightMouseButton(e))) { return; }
        if (screen == Screen.KNOWLEDGE && knowledgeCanvasRect().contains(mp)) { knowledgeTreeLastDrag = mp; knowledgeTreeDragging = true; return; }
        if (findScrollRegion(mp.x, mp.y) != null) { handleScrollbarClick(mp.x, mp.y); repaint(); }
    }
    @Override public void mouseReleased(MouseEvent e) { scrollbarDragActive=false; knowledgeTreeDragging=false; knowledgeTreeLastDrag=null; }
    @Override public void mouseEntered(MouseEvent e) {}
    @Override public void mouseExited(MouseEvent e) { mouseX=-1; mouseY=-1; repaint(); }
    @Override public void mouseMoved(MouseEvent e) {
        lastInputMillis=System.currentTimeMillis();
        Point mp = mapInputPoint(e);
        if (mp == null) { mouseX=-1; mouseY=-1; repaint(); return; }
        mouseX=mp.x; mouseY=mp.y; if (firstPersonRenderViewport.handleMouseMoved(this, mouseX, mouseY)) { repaint(); return; } if (mouseMovePreviewActive && screen == Screen.GAME && world != null) { Point t = screenPointToWorldTile(mouseX, mouseY); if (t != null) updateMouseMovementPreviewTo(t.x, t.y); } repaint();
    }
    @Override public void mouseDragged(MouseEvent e) {
        lastInputMillis=System.currentTimeMillis();
        Point mp = mapInputPoint(e);
        if (mp == null) return;
        mouseX=mp.x; mouseY=mp.y; if (screen == Screen.EDITOR && (SwingUtilities.isLeftMouseButton(e) || SwingUtilities.isRightMouseButton(e)) && handleInGameEditorGridPaint(mouseX, mouseY, SwingUtilities.isRightMouseButton(e))) { return; } if (screen == Screen.KNOWLEDGE && knowledgeTreeDragging && knowledgeTreeLastDrag != null) { panKnowledgeTree(mp.x - knowledgeTreeLastDrag.x, mp.y - knowledgeTreeLastDrag.y); knowledgeTreeLastDrag = mp; repaint(); return; } if (firstPersonRenderViewport.handleMouseDragged(this, mouseX, mouseY)) { repaint(); return; } handleScrollbarDrag(mouseX, mouseY); repaint();
    }
    @Override public void mouseWheelMoved(MouseWheelEvent e) {
        lastInputMillis=System.currentTimeMillis();
        if (eulaGateActive) { scrollEulaGate(e.getWheelRotation(), false); return; }
        if (screen == Screen.KNOWLEDGE) { Point mp = mapInputPoint(e); if (mp != null) zoomKnowledgeTreeAt(mp.x, mp.y, e.getPreciseWheelRotation()); return; }
        if (worldZoomControlActive() && !e.isShiftDown()) {
            int rot = e.getWheelRotation();
            if (rot != 0) changeWorldZoom(rot < 0 ? 1 : -1, "mouse wheel");
            return;
        }
        if (screen == Screen.PANEL && panelMode == PanelMode.MAP && e.isShiftDown()) {
            mapLayerView = Math.max(0, Math.min(19, mapLayerView + e.getWheelRotation()));
            DebugLog.audit("UI_SCROLL", "map layer=" + mapLayerView);
            repaint();
            return;
        }
        scrollActivePanel(e.getWheelRotation(), false);
    }
    @Override public void keyTyped(KeyEvent e) {
        if (screen == Screen.PANEL && panelMode == PanelMode.INFOPEDIA && isAssetInfopediaTab(infopediaTab) && "infopedia-asset-filter".equals(activeScrollTag)) {
            char ch = e.getKeyChar();
            if (ch != KeyEvent.CHAR_UNDEFINED && ch != '\b' && ch != '\n' && ch != '\r' && ch != '\u001b') { appendInfopediaAssetFilter(ch); }
            return;
        }
        if (screen == Screen.MULTIPLAYER) { multiplayerMenu.keyTyped(e.getKeyChar()); repaint(); return; }
        if (screen == Screen.CHARACTER && characterNameEditActive && !candidates.isEmpty()) {
            char ch = e.getKeyChar();
            Candidate c = candidates.get(candidateIndex);
            if (CharacterCreationAuthority.acceptNameChar(ch) && c.name.length() < CharacterCreationAuthority.MAX_PLAYER_NAME_LENGTH) {
                c.name = CharacterCreationAuthority.normalizeSpacing(c.name + ch);
                refreshNameLockedCandidateState(c);
                repaint();
            }
        }
    } @Override public void keyReleased(KeyEvent e) {
        keyboardInputBridge.keyReleased(e);
        int code = e.getKeyCode();
        if (firstPersonRenderViewport.handleKeyReleased(this, code)) { repaint(); return; }
        if (code==KeyEvent.VK_LEFT || code==KeyEvent.VK_RIGHT || code==KeyEvent.VK_UP || code==KeyEvent.VK_DOWN || code==KeyEvent.VK_A || code==KeyEvent.VK_D || code==KeyEvent.VK_W || code==KeyEvent.VK_S) {
            pendingMoveDx = 0;
            pendingMoveDy = 0;
            pendingHeldMove = false;
            DebugLog.audit("INPUT_PACING_RELEASE", "release preserved movement ghost armed=" + movementGhostArmed + " armed=" + armedMoveDx + "," + armedMoveDy + " turn=" + turn);
        }
    }
