
    String currentScrollableTag(){
        if(screen==Screen.CHARACTER) return jobDossierTab==0?"job-dossier":"job-list";
        if(screen==Screen.PANEL){
            if(panelMode==PanelMode.INVENTORY) return activeScrollTag != null && activeScrollTag.equals("inventory-info") ? "inventory-info" : "inventory";
            if(panelMode==PanelMode.LOOK) return "look-stack";
            if(panelMode==PanelMode.CHARACTER) return "character";
            if(panelMode==PanelMode.FACTIONS) return "factions";
            if(panelMode==PanelMode.BASES) return "bases";
            if(panelMode==PanelMode.NEWS) return "infopedia-detail";
            if(panelMode==PanelMode.MAP && mapTab==1) return "map-info";
            if(panelMode==PanelMode.INFOPEDIA) return activeScrollTag!=null && activeScrollTag.equals("infopedia-detail") ? "infopedia-detail" : "infopedia-list";
        }
        return "";
    }
    boolean scrollActivePanel(int delta, boolean page){
        String tag=currentScrollableTag();
        if(tag.isEmpty()) return false;
        int step=page ? ("look-stack".equals(tag) ? 3 : 5) : 1;
        applyScrollValue(tag, getScrollValue(tag)+delta*step, true);
        repaint();
        return true;
    }
    ScrollRegion findScrollRegion(int mx,int my){
        for(int i=scrollRegions.size()-1;i>=0;i--){ ScrollRegion r=scrollRegions.get(i); if(r.track.contains(mx,my) || r.thumb.contains(mx,my)) return r; }
        return null;
    }
    void handleScrollbarClick(int mx,int my){
        ScrollRegion r=findScrollRegion(mx,my);
        if(r==null) return;
        activeScrollTag=r.tag;
        if(r.maxScroll<=0){ applyScrollValue(r.tag,0,true); return; }
        if(r.thumb.contains(mx,my)){ scrollbarDragActive=true; scrollbarDragOffsetY=my-r.thumb.y; return; }
        int dir = my < r.thumb.y ? -1 : 1;
        applyScrollValue(r.tag, r.value + dir * Math.max(1,r.visibleLines), true);
    }
    void handleScrollbarDrag(int mx,int my){
        if(!scrollbarDragActive || activeScrollTag==null || activeScrollTag.isEmpty()) return;
        ScrollRegion target=null; for(ScrollRegion r:scrollRegions) if(r.tag.equals(activeScrollTag)){ target=r; break; }
        if(target==null || target.maxScroll<=0) return;
        int usable=Math.max(1,target.track.height-target.thumb.height);
        int rel=Math.max(0, Math.min(usable, my - scrollbarDragOffsetY - target.track.y));
        int value=(int)Math.round((rel/(double)usable)*target.maxScroll);
        applyScrollValue(target.tag, value, true); repaint();
    }

int countContainerItems(String containerId, ArrayList<String> legacyList, String item) {
        if (item == null) return 0;
        ContainerRecord c = itemContainers.get(containerId);
        if (c != null && !c.itemInstanceIds.isEmpty()) {
            int n = 0;
            for (String id : c.itemInstanceIds) {
                ItemInstance inst = itemInstances.get(id);
                if (inst != null && ItemQuality.namesMatch(inst.displayName, item)) n++;
            }
            return n;
        }
        int n=0; for (String s: legacyList) if (ItemQuality.namesMatch(s, item)) n++; return n;
    }

    ContainerRecord ensureContainer(String id, String label) {
        String safeId = (id == null || id.isBlank()) ? "unknown.container" : id;
        ContainerRecord c = itemContainers.get(safeId);
        if (c == null) {
            c = new ContainerRecord(safeId, label == null || label.isBlank() ? safeId : label);
            itemContainers.put(safeId, c);
        }
        return c;
    }

String traderShelfContainerId(TraderSession trader) {
        if (trader == null) return CONTAINER_TRADER_SHELF_PREFIX + "unknown";
        String site = trader.sourceSite == null ? "nosite" : trader.sourceSite.name + "." + trader.sourceSite.locationKey();
        return CONTAINER_TRADER_SHELF_PREFIX + ContainerIdentityApi.safeToken(trader.name) + "." + ContainerIdentityApi.safeToken(site);
    }

    ItemActionResult buyTraderOfferIntoInventory(TraderSession trader, TradeOffer offer) {
        if (trader == null || offer == null) return ItemActionResult.fail("No trader offer was available.", "buy trader offer failed: missing trader/offer");
        ItemInstance shelfItem = ensureTraderOfferShelfInstance(trader, offer);
        if (shelfItem == null) return ItemActionResult.fail("Trader shelf stock could not be itemized.", "buy trader offer failed: no shelf instance item=" + (offer==null?"null":offer.name));
        ItemActionResult moved = transferContainerItemResultByInstanceId(shelfItem.id, CONTAINER_PLAYER_INVENTORY, "Player carried inventory", inventory, "sold by " + trader.name + " to player");
        if (moved.success) {
            offer.provenance = ItemProvenanceRecord.transferred(offer.provenance == null ? moved.provenance : offer.provenance, offer.name, world, turn, "remaining trader shelf stock after one unit sold by " + trader.name);
            offer.itemInstanceId = "";
            rememberItemProvenance(offer.name, moved.provenance);
            verifyItemOperationalParity("after trader purchase");
        }
        return moved;
    }

    ItemInstance ensureTraderOfferShelfInstance(TraderSession trader, TradeOffer offer) {
        if (trader == null || offer == null || offer.name == null || offer.name.isBlank()) return null;
        String shelfId = traderShelfContainerId(trader);
        ensureContainer(shelfId, trader.name + " trader shelf");
        if (offer.itemInstanceId != null && !offer.itemInstanceId.isBlank()) {
            ItemInstance existing = itemInstances.get(offer.itemInstanceId);
            if (existing != null && ItemQuality.namesMatch(existing.displayName, offer.name)) {
                if (shelfId.equals(existing.containerId)) return existing;
                ItemActionResult moved = transferContainerItemResultByInstanceId(existing.id, shelfId, trader.name + " trader shelf", null, "loaded persistent faction stock onto trader shelf");
                if (moved.success) return itemInstances.get(existing.id);
            }
        }
        ItemProvenanceRecord shelfTrace = offer.provenance;
        if (shelfTrace == null) {
            shelfTrace = provenanceFromNpcSite(offer.name, trader.sourceSite, "stocked on trader shelf for " + trader.name);
        }
        ItemActionResult added = addItemToContainerResult(shelfId, trader.name + " trader shelf", offer.name, shelfTrace, null, "itemized trader shelf offer");
        offer.itemInstanceId = added.itemInstanceId;
        offer.provenance = added.provenance;
        return itemInstances.get(added.itemInstanceId);
    }

    String nextItemInstanceId() {
        while (itemInstances.containsKey("I" + nextItemInstanceSeq)) nextItemInstanceSeq++;
        return "I" + (nextItemInstanceSeq++);
    }

    ItemActionResult transferContainerItemResultByInstanceId(String instanceId, String toContainerId, String toLabel, ArrayList<String> toLegacy, String route) {
        if (instanceId == null || instanceId.isBlank()) return ItemActionResult.fail("No item instance ID was supplied.", "container transfer by instance failed: blank id to=" + toContainerId);
        ItemInstance inst = itemInstances.get(instanceId);
        if (inst == null) return ItemActionResult.fail("That item record is no longer available.", "container transfer by instance failed: missing id=" + instanceId + " to=" + toContainerId);
        String fromContainerId = inst.containerId;
        String accessProblem = ActorAccessAuthority.containerTransferProblem(this, fromContainerId, toContainerId, route);
        if (accessProblem != null) {
            DebugLog.audit("ACTOR_ACCESS_BLOCK", "container transfer blocked from=" + fromContainerId + " to=" + toContainerId + " route=" + route + " reason=" + accessProblem + " state=" + stateSummary());
            return ItemActionResult.fail("Access denied: " + accessProblem, "actor access blocked from=" + fromContainerId + " to=" + toContainerId + " route=" + route);
        }
        ContainerRecord from = itemContainers.get(fromContainerId);
        if (from != null) from.itemInstanceIds.remove(instanceId);
        ItemProvenanceRecord prior = inst.provenance == null ? takeProvenanceForItem(inst.displayName) : inst.provenance;
        ItemProvenanceRecord moved = ItemProvenanceRecord.transferred(prior, inst.displayName, world, turn, route == null ? "moved between containers" : route);
        ContainerRecord dest = ensureContainer(toContainerId, toLabel);
        inst.containerId = dest.id;
        inst.provenance = moved;
        inst.provenanceUnitId = moved == null ? inst.provenanceUnitId : moved.unitId;
        itemInstances.put(inst.id, inst);
        if (!dest.itemInstanceIds.contains(inst.id)) dest.itemInstanceIds.add(inst.id);
        if (toLegacy != null) toLegacy.add(inst.displayName);
        rememberItemProvenance(inst.displayName, moved);
        DebugLog.audit("CONTAINER_TRANSFER", "item=" + inst.displayName + " from=" + fromContainerId + " to=" + toContainerId + " route=" + route + " instance=" + inst.id + " state=" + stateSummary());
        return ItemActionResult.ok(inst.displayName, inst.id, toContainerId, moved, inst.displayName + " moved to " + toLabel + ".", "container transfer by instance item=" + inst.displayName + " instance=" + inst.id + " from=" + fromContainerId + " to=" + toContainerId + " route=" + route);
    }

    ItemInstance transferContainerItemAt(String fromContainerId, ArrayList<String> fromLegacy, int idx,
                                         String toContainerId, String toLabel, ArrayList<String> toLegacy,
                                         String route) {
        if (idx < 0 || idx >= fromLegacy.size()) return null;
        String accessProblem = ActorAccessAuthority.containerTransferProblem(this, fromContainerId, toContainerId, route);
        if (accessProblem != null) {
            DebugLog.audit("ACTOR_ACCESS_BLOCK", "legacy container transfer blocked from=" + fromContainerId + " to=" + toContainerId + " route=" + route + " reason=" + accessProblem + " state=" + stateSummary());
            return null;
        }
        String item = fromLegacy.remove(idx);
        ItemInstance inst = removeContainerInstanceAt(fromContainerId, idx, item);
        ItemProvenanceRecord prior = inst != null && inst.provenance != null ? inst.provenance : takeProvenanceForItem(item);
        ItemProvenanceRecord moved = ItemProvenanceRecord.transferred(prior, item, world, turn, route == null ? "moved between containers" : route);
        ContainerRecord dest = ensureContainer(toContainerId, toLabel);
        if (inst == null) {
            String id = nextItemInstanceId();
            inst = new ItemInstance(id, item, dest.id, moved == null ? "" : moved.unitId, moved);
            itemInstances.put(id, inst);
            dest.itemInstanceIds.add(id);
        } else {
            inst.displayName = item;
            inst.containerId = dest.id;
            inst.provenance = moved;
            inst.provenanceUnitId = moved == null ? inst.provenanceUnitId : moved.unitId;
            itemInstances.put(inst.id, inst);
            dest.itemInstanceIds.add(inst.id);
        }
        if (toLegacy != null) toLegacy.add(item);
        rememberItemProvenance(item, moved);
        DebugLog.audit("CONTAINER_TRANSFER", "item=" + item + " from=" + fromContainerId + " to=" + toContainerId + " route=" + route + " instance=" + (inst==null?"none":inst.id) + " state=" + stateSummary());
        return inst;
    }

    ItemInstance transferContainerItemByName(String fromContainerId, ArrayList<String> fromLegacy, String item,

        return null;
    }

    void rememberItemProvenance(String item, ItemProvenanceRecord provenance) {
        if (item == null || provenance == null) return;
        ArrayDeque<ItemProvenanceRecord> q = itemProvenance.computeIfAbsent(item, k -> new ArrayDeque<>());
        q.addLast(provenance);
        while (q.size() > 24) q.removeFirst();
    }

    ItemProvenanceRecord provenanceForContainerIndex(String containerId, int idx, String fallbackItem) {
        ContainerRecord c = itemContainers.get(containerId);
        if (c != null && idx >= 0 && idx < c.itemInstanceIds.size()) {
            ItemInstance inst = itemInstances.get(c.itemInstanceIds.get(idx));
            if (inst != null && inst.provenance != null) return inst.provenance;
        }
        return provenanceForItem(fallbackItem);
    }

    ItemProvenanceRecord provenanceForItem(String item) {
        if (item == null) return null;
        for (String containerId : new String[]{CONTAINER_PLAYER_INVENTORY, CONTAINER_BASE_STORAGE}) {
            ContainerRecord c = itemContainers.get(containerId);
            if (c == null) continue;
            for (int i=c.itemInstanceIds.size()-1; i>=0; i--) {
                ItemInstance inst = itemInstances.get(c.itemInstanceIds.get(i));
                if (inst != null && inst.displayName != null && (inst.displayName.equals(item) || ItemQuality.stripManufacturingIdentity(inst.displayName).equalsIgnoreCase(ItemQuality.stripManufacturingIdentity(item)))) {
                    if (inst.provenance != null) return inst.provenance;
                }
            }
        }
        ArrayDeque<ItemProvenanceRecord> q = itemProvenance.get(item);
        if (q == null || q.isEmpty()) q = itemProvenance.get(ItemQuality.stripManufacturingIdentity(item));
        return q == null || q.isEmpty() ? null : q.peekLast();
    }

    String provenanceSummaryForItem(String item) {
        ItemProvenanceRecord pr = provenanceForItem(item);
        return pr == null ? "Origin: untraced legacy item. Reserved full-world generation will assign a factory, worker, input ledger, and trade path." : pr.summary();
    }

    ItemProvenanceRecord takeProvenanceForItem(String item) {
        if (item == null) return null;
        ArrayDeque<ItemProvenanceRecord> q = itemProvenance.get(item);
        if (q == null || q.isEmpty()) q = itemProvenance.get(ItemQuality.stripManufacturingIdentity(item));
        if (q == null || q.isEmpty()) q = itemProvenance.get(ItemQuality.stripQuality(item));
        return q == null || q.isEmpty() ? null : q.removeLast();
    }

    ItemInstance removeContainerInstanceByName(String containerId, String item) {
        ContainerRecord c = itemContainers.get(containerId);
        if (c == null || item == null) return null;
        for (int i=0; i<c.itemInstanceIds.size(); i++) {
            String id = c.itemInstanceIds.get(i);
            ItemInstance inst = itemInstances.get(id);
            if (inst != null && item.equals(inst.displayName)) {
                c.itemInstanceIds.remove(i);
                itemInstances.remove(id);
                return inst;
            }
        }
        return null;
    }

    ItemInstance removeContainerInstanceAt(String containerId, int idx, String item) {
        ContainerRecord c = itemContainers.get(containerId);
        if (c != null && idx >= 0 && idx < c.itemInstanceIds.size()) {
            String id = c.itemInstanceIds.remove(idx);
            ItemInstance inst = itemInstances.remove(id);
            if (inst != null) return inst;
        }
        return removeContainerInstanceByName(containerId, item);
    }

    void rebuildItemContainersFromLegacyLists() {
        itemInstances.clear();
        itemContainers.clear();
        ensureContainer(CONTAINER_PLAYER_INVENTORY, "Player carried inventory");
        ensureContainer(CONTAINER_BASE_STORAGE, "Claimed base storage");
        int oldNext = nextItemInstanceSeq;
        nextItemInstanceSeq = 1;
        for (String item : new ArrayList<String>(inventory)) {
            ItemProvenanceRecord pr = provenanceForItem(item);
            addItemToContainer(CONTAINER_PLAYER_INVENTORY, "Player carried inventory", item, pr == null ? ItemProvenanceRecord.unknown(item, this) : pr, new ArrayList<String>());

        for (String item : new ArrayList<String>(baseStorage)) {

            addItemToContainer(CONTAINER_BASE_STORAGE, "Claimed base storage", item, pr == null ? ItemProvenanceRecord.unknown(item, this) : pr, new ArrayList<String>());
        }
        nextItemInstanceSeq = Math.max(nextItemInstanceSeq, oldNext);
    }

    LinkedHashMap<String,Integer> stackInventory(List<String> items) {
        LinkedHashMap<String,Integer> stacks = new LinkedHashMap<>();
        for (String raw : items) {
            String key = raw == null ? "Unknown item" : raw.trim();
            if (key.isEmpty()) key = "Unknown item";
            stacks.put(key, stacks.getOrDefault(key, 0) + 1);
        }
        return stacks;
    }

    ArrayList<String> stackedInventoryNames() {
        LinkedHashMap<String,Integer> stacks = stackInventory(inventory);
        if (supplies>0) stacks.put("Construction supplies", stacks.getOrDefault("Construction supplies",0)+supplies);
        if (food>0) stacks.put("Food reserve", stacks.getOrDefault("Food reserve",0)+food);
        if (water>0) stacks.put("Water reserve", stacks.getOrDefault("Water reserve",0)+water);
        return new ArrayList<>(stacks.keySet());
    }

    void equipInventoryClothing(String item) {
        int idx = inventory.indexOf(item);
        if (idx < 0) { logEvent("Cannot equip " + item + ": no carried copy found."); return; }
        Clothing old = equippedClothing;
        String oldItem = old == null || old.name == null ? "" : old.name + (old.damaged && !old.name.toLowerCase(Locale.ROOT).contains("damaged") ? " (damaged)" : "");
        if (oldItem != null && !oldItem.isBlank()) {
            ItemActionResult returned = transferFirstEquipmentItemBackToInventory(2, oldItem, "replaced equipped clothing");
            if (!returned.success && old != null && old.name != null && !old.name.isEmpty()) addInventoryItem(oldItem, ItemProvenanceRecord.of(old.name, old.alignedFaction, "previously equipped clothing", world, turn, "existing equipped garment", "returned to carried inventory after clothing change"));
        }
        boolean damaged = item.toLowerCase(Locale.ROOT).contains("damaged");
        Faction f = factionForClothingItem(item);
        int base = damaged ? 26 : 46;
        int defense = damaged ? 1 : 2;
        if (f == Faction.CIVIC_WARDENS || f == Faction.IMPERIAL_GUARD) { defense += damaged ? 1 : 2; base -= 4; }
        if (f == Faction.MECHANICUS_CLOISTER_RED) { defense += 1; base += 2; }
        ItemActionResult moved = transferContainerItemResultAt(CONTAINER_PLAYER_INVENTORY, inventory, idx, equipmentContainerIdForSlot(2), "Player equipment slot: Clothing", null, "equipped by player as clothing");
        if (!moved.success) { logEvent("Cannot equip " + item + ": " + moved.playerText); return; }
        equippedClothing = new Clothing(item, f, Math.max(8, Math.min(70, base)), defense, damaged);
        selectedEquipmentSlotIndex = 2;
        logEvent("Equipped " + item + ". Identity=" + equippedClothing.alignedFaction.label + ", cover=" + equippedClothing.disguiseBase + "%, defense=" + equippedClothing.defense + ". Previous clothing returned to inventory.");
        DebugLog.audit("INVENTORY_EQUIP_CLOTHING", moved.auditText + " faction=" + f + " old=" + (old!=null?old.name:"none") + " state=" + stateSummary());
        advanceTurn("changes clothing"); repaint();
    }

    void applyMedicalItemEffect(String item, String low, int oldWounds, int oldFatigue) {
        int oldBleed = bleeding, oldInfection = infectionRisk, oldPain = pain;
        if (low.contains("bandage") || low.contains("field dressing")) {
            bleeding = Math.max(0, bleeding - 4);
            wounds = Math.max(0, wounds - 1);
        } else if (low.contains("antiseptic")) {
            infectionRisk = Math.max(0, infectionRisk - 6);
        } else if (low.contains("splint")) {
            pain = Math.max(0, pain - 2);
            stabilizeWorstLimb(35.0);
        } else if (low.contains("medkit") || low.contains("medicae")) {
            bleeding = Math.max(0, bleeding - 6);
            infectionRisk = Math.max(0, infectionRisk - 4);
            wounds = Math.max(0, wounds - 2);
            fatigue = Math.max(0, fatigue - 2);
            pain = Math.max(0, pain - 4);
            healWorstBodyPart(55.0);
        } else if (low.contains("injector")) {
            wounds = Math.max(0, wounds - 1);
            fatigue = Math.max(0, fatigue - 2);
            pain = Math.max(0, pain - 3);
        }
        logEvent("MEDICAL USE: " + item + " applied. Wounds " + oldWounds + "->" + wounds + ", bleeding " + oldBleed + "->" + bleeding + ", infection risk " + oldInfection + "->" + infectionRisk + ", pain " + oldPain + "->" + pain + ".");
        DebugLog.audit("MEDICAL_ITEM", "item=" + item + " wounds=" + oldWounds + "->" + wounds + " bleeding=" + oldBleed + "->" + bleeding + " infection=" + oldInfection + "->" + infectionRisk + " pain=" + oldPain + "->" + pain + " state=" + stateSummary());
    }

    void stabilizeWorstLimb(double minHealth) {
        if (active == null || active.body == null) return;
        BodyPart worst = null;
        for (BodyPart bp : active.body.values()) {
            String low = bp.name.toLowerCase(Locale.ROOT);
            if (!(low.contains("leg") || low.contains("arm") || low.contains("hand") || low.contains("foot"))) continue;
            if (worst == null || bp.currentHealth() < worst.currentHealth()) worst = bp;
        }
        if (worst != null && worst.currentHealth() < minHealth) worst.health = Math.min(worst.maxHealth(), minHealth);
    }

    void healWorstBodyPart(double amount) {
        if (active == null || active.body == null) return;
        BodyPart worst = null;
        for (BodyPart bp : active.body.values()) if (worst == null || bp.currentHealth()/bp.maxHealth() < worst.currentHealth()/worst.maxHealth()) worst = bp;
        if (worst != null) {
            double before = worst.currentHealth();
            worst.health = Math.min(worst.maxHealth(), worst.currentHealth() + amount);
            logEvent("BODY RECOVERY: " + worst.name + " HP " + String.format(Locale.US,"%.0f", before) + "->" + String.format(Locale.US,"%.0f", worst.currentHealth()) + ".");
        }
    }

    void dropSelectedInventoryItem() {
        String item = selectedCarriedInventoryActionItem();
        if (item == null) { logEvent("Inventory empty. Nothing to drop."); return; }
        if (item.equals("Construction supplies") && supplies > 0) { supplies--; logEvent("Dropped one Construction supply onto the floor. It may or may not still be there later."); }
        else if (item.equals("Food reserve") && food > 0) { food--; addInventoryItem("Dropped food ration", ItemProvenanceRecord.found("Dropped food ration", world, turn, "converted from abstract food reserve")); logEvent("Converted one abstract food reserve into a carried dropped food ration item."); }
        else if (item.equals("Water reserve") && water > 0) { water--; addInventoryItem("Dropped water ration", ItemProvenanceRecord.found("Dropped water ration", world, turn, "converted from abstract water reserve")); logEvent("Converted one abstract water reserve into a carried dropped water ration item."); }
        else if (PortableLightProfile.isLightItem(item)) {
            dropPortableLightItem(item);
        }
        else {
            int idx = inventory.indexOf(item);
            if (idx >= 0) {
                ItemActionResult dropped = transferContainerItemResultAt(CONTAINER_PLAYER_INVENTORY, inventory, idx, CONTAINER_VOID_DISCARDED, "discarded item ledger", null, "dropped by player in active sector");
                logEvent("ITEM DROPPED: Dropped one " + item + ". The item leaves your active carried stack. " + (dropped.provenance == null ? "Origin: untraced legacy item." : dropped.provenance.summary()));
                DebugLog.audit("INVENTORY_DROP", dropped.auditText + " state=" + stateSummary());
            } else logEvent("Drop failed: selected stack is not currently removable.");
        }
        selectedInventoryIndex = Math.max(0, Math.min(selectedInventoryIndex, Math.max(0, stackedInventoryNames().size()-1)));
        advanceTurn("drops an inventory item."); repaint();
    }

    void useSelectedInventoryItemBody() {
        String item = selectedCarriedInventoryActionItem();
        if (item == null) { logEvent("Inventory empty. Nothing to use."); return; }
        String low = item.toLowerCase(Locale.ROOT);
        if (item.equals("Food reserve") && food > 0) { logEvent("Food reserve already internalized. Use carried ration items to restore reserves later."); return; }
        if (item.equals("Water reserve") && water > 0) { logEvent("Water reserve already internalized. Use canteens/water ration items to restore reserves later."); return; }
        if (low.contains("coffee") || low.contains("stim") || low.contains("narcotic")) { consumeWakefulnessItem(item); return; }
        if (isMedicalUseItem(low)) { useUtilityInventoryItem(item); return; }
        if (PortableLightProfile.isLightItem(item)) { usePortableLightItem(item); return; }
        if (low.contains("ration") || low.contains("canteen") || low.contains("water") || low.contains("food")) { consumeSpecificInventoryItem(item); return; }
        if (equipInventoryItemToPaperDoll(item)) return;
        if (item.equals("Construction supplies") && supplies > 0) { logEvent("Construction supplies are already pooled for BUILD/WORKBENCH actions. Current supplies=" + supplies + "."); return; }
        if (low.contains("construction supplies")) { int idx = inventory.indexOf(item); if (idx >= 0) { removeInventoryItem(item); supplies++; logEvent("Packed " + item + " into construction supplies pool. Supplies now " + supplies + "."); advanceTurn("sorts construction supplies"); repaint(); return; } }
        useUtilityInventoryItem(item);
    }

    void usePortableLightItem(String item) {

        activePortableLightItem = p.name;
        activePortableLightWorn = p.wearable;
        activePortableLightExpiresTurn = turn + Math.max(1, p.durationTurns);
        lastPortableLightReport = (p.wearable ? "WORN LIGHT: " : "CARRIED LIGHT: ") + p.name + " radius " + p.radius + " for about " + Math.max(1, p.durationTurns / TURNS_PER_HOUR) + " hour(s). " + p.note;
        logEvent(lastPortableLightReport);
        DebugLog.audit("PORTABLE_LIGHT_USE", "item=" + p.name + " radius=" + p.radius + " expires=" + activePortableLightExpiresTurn + " worn=" + p.wearable + " state=" + stateSummary());

    }

    void dropPortableLightItem(String item) {
        int idx = inventory.indexOf(item);

    }

    void throwSelectedPortableLight() {
        String item = selectedInventoryItemName();
        if (item == null) { logEvent("No selected item to throw."); return; }
        if (!PortableLightProfile.isLightItem(item)) { logEvent("THROW LIGHT blocked: selected stack is not a portable light source."); return; }
        int idx = inventory.indexOf(item);
        if (idx < 0) { logEvent("THROW LIGHT blocked: selected light is not removable from inventory."); return; }
        PortableLightProfile p = PortableLightProfile.profile(item);
        if (p == null) { logEvent("THROW LIGHT blocked: no light profile registered for " + item + "."); return; }
        int fx = facingDx(), fy = facingDy();
        int range = Math.max(1, p.throwRange);
        int tx = playerX, ty = playerY;
        for (int step=0; step<range; step++) {
            int nx = tx + fx, ny = ty + fy;
            if (world == null || !world.inBounds(nx,ny) || !world.walkable(nx,ny) || world.npcAt(nx,ny) != null) break;
            tx = nx; ty = ny;
        }
        ItemActionResult thrown = transferContainerItemResultAt(CONTAINER_PLAYER_INVENTORY, inventory, idx, CONTAINER_VOID_DISCARDED, "portable light throw ledger", null, "thrown as active light source");
        int dur = p.remainingDuration(turn, activePortableLightItem, activePortableLightExpiresTurn);
        portableLights.add(new PortableLightInstance(p.name, tx, ty, p.radius, turn + Math.max(1, dur), portableLightWorldKey(), "thrown"));
        while (portableLights.size() > 48) portableLights.remove(0);
        if (p.name.equals(activePortableLightItem)) { activePortableLightItem=""; activePortableLightExpiresTurn=0; activePortableLightWorn=false; }
        lastPortableLightReport = "THROWN LIGHT: " + p.name + " lands at " + tx + "," + ty + ", radius " + p.radius + ", burn remaining about " + Math.max(1,dur/TURNS_PER_HOUR) + "h.";
        logEvent(lastPortableLightReport + " " + (thrown.provenance == null ? "" : thrown.provenance.summary()));
        sensoryCacheWorld = null;
        advanceTurn("throws a light source."); repaint();
    }

    void processPortableLightTick() {
        if (activePortableLightExpiresTurn > 0 && activePortableLightExpiresTurn <= turn) {
            logEvent("LIGHT OUT: " + activePortableLightItem + " gutters out or dies. The dark is professionally unimpressed.");
            activePortableLightItem = "";
            activePortableLightExpiresTurn = 0;
            activePortableLightWorn = false;
            sensoryCacheWorld = null;
        }
        int before = portableLights.size();
        portableLights.removeIf(l -> l == null || l.expiresTurn <= turn);
        if (before != portableLights.size()) sensoryCacheWorld = null;
    }

        PortableLightProfile p = PortableLightProfile.profile(activePortableLightItem);

    }

    void consumeWakefulnessItem(String item) {
        int idx = inventory.indexOf(item);
        if (idx < 0) { logEvent("Cannot use " + item + ": no removable carried item found."); return; }
        ItemActionResult consumed = transferContainerItemResultAt(CONTAINER_PLAYER_INVENTORY, inventory, idx, CONTAINER_VOID_CONSUMED, "consumed item ledger", null, "consumed by player for wakefulness");
        if (!consumed.success) { logEvent("Cannot use " + item + ": " + consumed.playerText); return; }
        String low = item.toLowerCase(Locale.ROOT);
        int reduce = low.contains("narcotic") ? 12 : (low.contains("stim") ? 9 : 6);
        int strain = low.contains("narcotic") ? 3 : (low.contains("stim") ? 2 : 1);
        int oldSleep = sleepNeed, oldWounds = wounds, oldStrain = stimulantStrain;
        sleepNeed = Math.max(0, sleepNeed - reduce);
        stimulantStrain += strain;
        if (stimulantStrain >= 6) { wounds++; stimulantStrain -= 6; }
        DebugLog.audit("WAKEFULNESS_ITEM", "item=" + item + " sleepNeed " + oldSleep + "->" + sleepNeed + " strain " + oldStrain + "->" + stimulantStrain + " wounds " + oldWounds + "->" + wounds + " state=" + stateSummary());
        logEvent("Used " + item + ". Sleep need " + oldSleep + "->" + sleepNeed + "/" + MAX_SLEEP_NEED + ". Stimulant strain rises; cumulative abuse damages endurance.");
        advanceTurn("uses a wakefulness drug."); repaint();
    }

    void consumeSpecificInventoryItem(String item) {
        int idx = inventory.indexOf(item);
        if (idx < 0) { logEvent("Cannot consume " + item + ": no removable carried item found."); return; }
        ItemActionResult consumed = transferContainerItemResultAt(CONTAINER_PLAYER_INVENTORY, inventory, idx, CONTAINER_VOID_CONSUMED, "consumed item ledger", null, "consumed by player as food/water");
        if (!consumed.success) { logEvent("Cannot consume " + item + ": " + consumed.playerText); return; }
        int oldFood = food, oldWater = water;
        String low = item.toLowerCase(Locale.ROOT);
        if (low.contains("water") || low.contains("canteen")) water = Math.min(MAX_FOOD_WATER, water + 4);
        else if (low.contains("food")) food = Math.min(MAX_FOOD_WATER, food + 4);
        else { food = Math.min(MAX_FOOD_WATER, food + 4); water = Math.min(MAX_FOOD_WATER, water + 2); }
        DebugLog.audit("CONSUME_SELECTED_ITEM", "item=" + item + " food " + oldFood + "->" + food + " water " + oldWater + "->" + water + " state=" + stateSummary());
        logEvent("Consumed " + item + ". Food " + oldFood + "->" + food + ", water " + oldWater + "->" + water + ".");
        advanceTurn("uses an inventory item."); repaint();
    }

    ArrayList<String> factionMemberEquipmentCandidates(List<String> items, int limit) {
        ArrayList<String> out = new ArrayList<>();
        if (items == null) return out;
        LinkedHashMap<String,Integer> stacks = stackInventory(items);

            if (out.size() >= Math.max(1, limit)) break;
            if (factionMemberGearUseLine(item) != null) out.add(item + " x" + stacks.get(item));
        }
        return out;
    }

    String heatPressureBand(int heat, int suspicionValue, int pressure) {
        int total = Math.max(0, heat) + Math.max(0, suspicionValue) + Math.max(0, pressure / 2);

    String baseStoredItemSafetyLine(String item) {
        if (item == null || item.isBlank()) return "unknown stored item";
        String low = item.toLowerCase(Locale.ROOT);
        if (isQuestOrEvidenceItem(item)) return "quest/evidence item; protect it until the right turn-in";
        if (ItemCatalog.isFirearmLike(item) || low.contains("knife") || low.contains("pistol") || low.contains("shotgun") || low.contains("grenade")) return "weapon-linked; raids, law, or rivals may care";
        if (low.contains("data spike") || low.contains("lockpick") || low.contains("keycard") || low.contains("stolen") || low.contains("profane") || low.contains("contraband")) return "restricted or suspicious if searched";
        if (low.contains("permit") || low.contains("stock certificate") || low.contains("gold") || low.contains("trade chit")) return "commerce-useful; consider banking, permits, or contracts";
        if (low.contains("med") || low.contains("bandage") || low.contains("field dressing") || low.contains("antiseptic")) return "medical reserve; keep near clinics or dangerous work";
        if (low.contains("ration") || low.contains("food") || low.contains("water")) return "survival reserve; keep some carried before travel";
        return "ordinary stored goods";
    }

    boolean isQuestOrEvidenceItem(String item) {
        if (item == null) return false;
        String low = item.toLowerCase(Locale.ROOT);
        if (low.contains("ident chip") || low.contains("sealed object") || low.contains("lockbox") || low.contains("leader journal") || low.contains("evidence")) return true;
        for (FactionContract c : factionContracts) if (c != null && !c.completed && c.requiredTurnInItem != null && ItemQuality.namesMatch(item, c.requiredTurnInItem)) return true;
        return false;
    }

            if (ControlledProductionJobAuthority.isGeneratedAssignment(recipeName)) {

"ready" : "blocked: " + block) + " | " + manual);

        return lines;
    }

            if (obj.symbol == 'p') rating += 3 + Math.max(0, obj.cover);
            if (obj.symbol == 'g') rating += 4 + Math.max(0, obj.capacity/2);
            if (obj.symbol == 'D') rating += 4 + Math.max(0, obj.cover);
            if (obj.symbol == 'T') rating += 8 + Math.max(0, obj.cover) + Math.max(0, obj.charges);
            if (obj.symbol == 'H') rating += 6 + Math.max(0, obj.cover) + Math.max(0, obj.charges);

    void applyBusinessPressureTick(boolean sleeping) {
        if (turn <= 0 || turn % TURNS_PER_HOUR != 0 || !baseClaimed) return;
        int legal = 0, illegal = 0, security = baseSecurityRating();
        for (BaseObject obj : baseObjects) {
            if (!obj.isBusinessAsset() || !obj.businessOpen) continue;
            if (obj.permittedBusiness) legal++; else illegal++;
        }
        if (legal == 0 && illegal == 0) return;
        int heatGain = Math.max(0, illegal * 2 + legal - security / 4);
        if (heatGain > 0) {
            int oldH = gangHeat, oldS = suspicion;
            gangHeat += heatGain;
            suspicion += Math.max(0, illegal - security / 8);
            logEvent("BUSINESS HEAT: open commerce draws attention. Heat " + oldH + "->" + gangHeat + ", suspicion " + oldS + "->" + suspicion + ". Security rating " + security + ".");
        } else if (security > 0 && (gangHeat > 0 || suspicion > 0)) {
            int oldH = gangHeat, oldS = suspicion;
            gangHeat = Math.max(0, gangHeat - 1);
            if (security >= 6) suspicion = Math.max(0, suspicion - 1);
            if (oldH != gangHeat || oldS != suspicion) logEvent("BASE SECURITY: patrols, locks, and alarms bleed off attention. Heat " + oldH + "->" + gangHeat + ", suspicion " + oldS + "->" + suspicion + ".");
        }
        maybeTriggerBaseRaid(security);
    }

    void maybeTriggerBaseRaid(int security) {
        int pressure = baseRaidPressure(security);
        if (pressure < 12) return;
        int chance = baseRaidChance(security);
        if (rng.nextInt(100) >= chance) return;
        String attacker = zoneRaidThreatName();
        int shield = 0;
        for (BaseObject obj : baseObjects) if ((obj.symbol == 'H' || obj.symbol == 'T') && obj.armed && obj.charges > 0) { shield += obj.cover; obj.charges = Math.max(0, obj.charges - 1); }
        int damage = Math.max(1, 2 + rng.nextInt(4) - Math.max(0, security/5) - Math.max(0, shield/4));
        ArrayList<BaseObject> vulnerable = new ArrayList<>();
        for (BaseObject obj : baseObjects) if (obj.integrity > 0) vulnerable.add(obj);
        if (!vulnerable.isEmpty()) {
            BaseObject hit = vulnerable.get(rng.nextInt(vulnerable.size()));
            int old = hit.integrity;
            hit.integrity = Math.max(0, hit.integrity - damage);
            logEvent("BASE RAID: " + attacker + " hits " + hit.name + ". Integrity " + old + "->" + hit.integrity + ". Defense rating " + security + ", powered shield " + shield + ".");
        } else {
            logEvent("BASE RAID: " + attacker + " sweeps the claimed room, but no intact assets remain to smash. A grim efficiency.");
        }
        if (security >= 8) {
            gangHeat = Math.max(0, gangHeat - 3);
            suspicion = Math.max(0, suspicion - 2);
            logEvent("DEFENSE HOLDS: alarms, barricades, and crew response blunt the raid. Heat and suspicion drop as the attackers reassess.");
        } else {
            if (!baseStorage.isEmpty()) {
                String stolen = removeBaseStorageAtWithTrace(rng.nextInt(baseStorage.size()));
                logEvent("ITEM LOST: raiders seize " + stolen + " from base storage.");
            }
            fatigue = Math.min(25, fatigue + 2);
            wounds += 1;
            logEvent("DEFENSE FAILED: the raid leaves you battered. Wounds +1, fatigue +2. Build defenses or buy lawful cover.");
        }
        DebugLog.audit("BASE_RAID", "attacker="+attacker+" pressure="+pressure+" chance="+chance+" security="+security+" heat="+gangHeat+" suspicion="+suspicion+" state="+stateSummary());
    }


    int playerFactionPower() {
        int power = 0;
        power += Math.max(0, factionRecruits.size() * 2);
        power += Math.max(0, openBusinessCount() * 4);
        power += Math.max(0, baseStorage.size() / 3);
        power += Math.max(0, baseObjects.size() * 2);
        power += Math.max(0, businessSellPowerBonus() * 2);
        power += Math.max(0, cargoNetworkCapacity() / 4);
        power += Math.max(0, xp / 20);
        for (BaseObject obj : baseObjects) {
            if (obj == null) continue;
            if (obj.symbol == 'T' || obj.symbol == 'H') power += 8;
            if (obj.symbol == 'g' || obj.symbol == 'p') power += 4;
            power += Math.max(0, ItemQuality.tierIndex((obj.qualityName == null ? "Common" : obj.qualityName) + " base asset") / 2);
        }
        return power;
    }

    Faction rivalFactionForBusiness(BaseObject obj) {
        if (obj == null) return Faction.BANDIT;
        switch (obj.symbol) {
            case 'B': return Faction.HIVER;
            case 'M': return Faction.CIVIC_WARDENS;
            case 'w': return Faction.MECHANIST_COLLEGIA;
            case 'f': return Faction.MECHANIST_COLLEGIA;
            case 'l': return Faction.CIVIC_LEDGER_OFFICE;
            case 'u': case 'e': return Faction.HIVER_BLOCK_SUMPLEDGER;
            case 's': return Faction.BANDIT;
            default: return Faction.BANDIT;
        }
    }

    Faction rivalFactionForRecipe(CraftingRecipe r, ProductionRecipe pr) {
        String text = ((r == null ? "" : (r.name + " " + r.outputBaseItem + " " + r.requiredKnowledge)) + " " + (pr == null ? "" : pr.outputItemName())).toLowerCase(Locale.ROOT);
        if (text.contains("water") || text.contains("ration") || text.contains("food")) return Faction.HIVER_BLOCK_SUMPLEDGER;
        if (text.contains("med") || text.contains("bandage") || text.contains("antiseptic") || text.contains("splint")) return Faction.CIVIC_WARDENS;
        if (text.contains("weapon") || text.contains("ammo") || text.contains("stub") || text.contains("blade")) return Faction.BANDIT;
        if (text.contains("construction") || text.contains("machine") || text.contains("forge") || text.contains("logic Engine")) return Faction.MECHANIST_COLLEGIA;
        if (text.contains("knowledge") || text.contains("doctrine") || text.contains("permit")) return Faction.CIVIC_LEDGER_OFFICE;
        return Faction.BANDIT;
    }

    void addFactionMarketPressure(Faction faction, int amount, String reason) {
        if (faction == null || faction == Faction.NONE || amount <= 0) return;
        int security = baseSecurityRating();
        int mitigated = Math.max(0, amount - Math.max(0, security / 10));
        if (mitigated <= 0) return;
        int old = factionMarketPressure.getOrDefault(faction, 0);
        int next = Math.min(100, old + mitigated);
        factionMarketPressure.put(faction, next);
        if (next >= old + 3 || next >= 20 && old < 20 || next >= 40 && old < 40 || next >= 70 && old < 70) {
            logEvent("FACTION PRESSURE: " + faction.label + " resents your " + reason + ". Pressure " + old + "->" + next + ".");
        }
        DebugLog.audit("FACTION_MARKET_PRESSURE", "faction=" + faction + " amount=" + amount + " mitigated=" + mitigated + " reason=" + reason + " pressure=" + old + "->" + next + " power=" + playerFactionPower() + " state=" + stateSummary());
    }

        if (zone == ZoneType.MECHANICUS_FORGE_CLOISTER) for (NpcFactionSite site : npcFactionSites) if (site.faction == Faction.MECHANIST_COLLEGIA) return site;
        if (zone == ZoneType.IMPERIAL_GUARD_BILLET) for (NpcFactionSite site : npcFactionSites) if (site.faction == Faction.IMPERIAL_GUARD) return site;
        if (zone == ZoneType.NOBLE_SERVICE_SPINE) for (NpcFactionSite site : npcFactionSites) if (site.faction == Faction.NOBLE) return site;

        if (!obj.permittedBusiness) pressure += 2;
        pressure += Math.max(0, businessSellPowerBonus() / 2);
        addFactionMarketPressure(rivalFactionForBusiness(obj), pressure, obj.businessName() + " taking market share");
    }

    void applyFactionMarketPressureTick(boolean sleeping) {
        if (turn <= 0 || turn % TURNS_PER_HOUR != 0 || !baseClaimed) return;
        int power = playerFactionPower();
        if (power <= 0) return;
        int open = openBusinessCount();
        int production = 0;
        for (BaseObject obj : baseObjects) if (obj != null && obj.assignedRecipe != null && !obj.assignedRecipe.isBlank()) production++;
        int slowPressure = Math.max(0, (power / 25) + Math.max(0, open - 1) + Math.max(0, production - 1));
        if (slowPressure > 0) {
            Faction ambient = ambientRivalFaction();
            addFactionMarketPressure(ambient, Math.max(1, slowPressure - (sleeping ? 1 : 0)), "growing base power");
        }
        decayFactionMarketPressure();
        maybeTriggerFactionStrategicRaid();
    }

    void decayFactionMarketPressure() {
        int security = baseSecurityRating();
        for (Faction f : Faction.values()) {
            int v = factionMarketPressure.getOrDefault(f, 0);
            if (v <= 0) continue;
            int decay = security >= 18 ? 2 : (security >= 8 ? 1 : 0);
            if (decay > 0) factionMarketPressure.put(f, Math.max(0, v - decay));
        }
    }

    Faction highestFactionMarketPressureFaction() {
        Faction best = Faction.NONE;
        int bv = 0;
        for (Map.Entry<Faction,Integer> e : factionMarketPressure.entrySet()) {
            if (e.getKey() != null && e.getKey() != Faction.NONE && e.getValue() != null && e.getValue() > bv) { best = e.getKey(); bv = e.getValue(); }
        }
        return best;
    }

    String highestFactionMarketPressureLine() {
        Faction f = highestFactionMarketPressureFaction();
        int v = f == Faction.NONE ? 0 : factionMarketPressure.getOrDefault(f, 0);
        if (v <= 0) return "no focused faction resentment yet";
        String stage = v < 20 ? "watching" : (v < 40 ? "irritated" : (v < 70 ? "hostile planning" : "openly preparing violence"));
        return f.label + " " + stage + " (" + v + ")";
    }

    int strategicRaidChance() {
        Faction f = highestFactionMarketPressureFaction();
        int pressure = f == Faction.NONE ? 0 : factionMarketPressure.getOrDefault(f, 0);
        if (pressure < 25) return 0;
        int power = playerFactionPower();
        int security = baseSecurityRating();
        return Math.max(1, Math.min(55, pressure / 3 + power / 8 - security / 3));
    }

    String contestedMarketSummaryLine() {
        ArrayList<String> parts = new ArrayList<>();
        for (Map.Entry<Faction,Integer> e : factionMarketPressure.entrySet()) {
            if (e.getKey() == null || e.getKey() == Faction.NONE || e.getValue() == null || e.getValue() <= 0) continue;
            parts.add(e.getKey().label + " " + e.getValue());
        }
        if (parts.isEmpty()) return "no rival ledger entries yet";
        parts.sort((a,b) -> Integer.compare(parseTrailingInt(b), parseTrailingInt(a)));
        return String.join("; ", parts.subList(0, Math.min(4, parts.size())));
    }

    int parseTrailingInt(String text) {
        try { return Integer.parseInt(text.replaceAll("^.* ", "")); } catch(Exception e) { return 0; }
    }

    void maybeTriggerFactionStrategicRaid() {
        Faction faction = highestFactionMarketPressureFaction();
        if (faction == Faction.NONE) return;
        int chance = strategicRaidChance();
        if (chance <= 0 || rng.nextInt(100) >= chance) return;
        int pressure = factionMarketPressure.getOrDefault(faction, 0);
        int security = baseSecurityRating();
        int power = playerFactionPower();
        String attacker = faction.label + " reprisal crew";
        int shield = 0;
        for (BaseObject obj : baseObjects) if ((obj.symbol == 'H' || obj.symbol == 'T') && obj.armed && obj.charges > 0) { shield += obj.cover; obj.charges = Math.max(0, obj.charges - 1); }
        int severity = Math.max(1, pressure / 18 + power / 18 - security / 10 - shield / 5);
        ArrayList<BaseObject> targets = new ArrayList<>();
        for (BaseObject obj : baseObjects) if (obj.integrity > 0 && (obj.isBusinessAsset() || obj.assignedRecipe != null && !obj.assignedRecipe.isBlank() || obj.symbol == 'k' || obj.symbol == 'q' || obj.symbol == 'G')) targets.add(obj);
        if (targets.isEmpty()) for (BaseObject obj : baseObjects) if (obj.integrity > 0) targets.add(obj);
        if (!targets.isEmpty()) {
            BaseObject hit = targets.get(rng.nextInt(targets.size()));
            int old = hit.integrity;
            hit.integrity = Math.max(0, hit.integrity - severity);
            logEvent("FACTION RAID: " + attacker + " attacks your market footprint and damages " + hit.name + " " + old + "->" + hit.integrity + ".");
        } else {
            logEvent("FACTION RAID: " + attacker + " searches the base, finds little to break, and leaves threats written in someone else's blood.");
        }
        if (!baseStorage.isEmpty() && severity > 1 && security < pressure / 6) {
            String stolen = removeBaseStorageAtWithTrace(rng.nextInt(baseStorage.size()));
            logEvent("ITEM LOST: " + faction.label + " raiders seize " + stolen + " to remind you who owns the market.");
        }
        int oldPressure = pressure;
        factionMarketPressure.put(faction, Math.max(0, pressure - Math.max(8, security / 2 + severity * 2)));
        if (security >= pressure / 6) {
            factionStanding.put(faction, factionStanding.getOrDefault(faction, 0) - 1);
            logEvent("DEFENSE HOLDS: your base survives the " + faction.label + " reprisal. Their pressure drops " + oldPressure + "->" + factionMarketPressure.getOrDefault(faction,0) + ", but relations sour.");
        } else {
            wounds += 1;
            fatigue = Math.min(25, fatigue + severity);
            logEvent("DEFENSE STAGGERS: the reprisal hurts. Wounds +1, fatigue +" + severity + ". More guards and powered defenses are advised.");
        }
        DebugLog.audit("FACTION_STRATEGIC_RAID", "faction=" + faction + " chance=" + chance + " pressure=" + oldPressure + "->" + factionMarketPressure.getOrDefault(faction,0) + " power=" + power + " security=" + security + " severity=" + severity + " state=" + stateSummary());
    }

    void collectBusinessIncome() {
        if (!baseClaimed) { logEvent("Business income blocked: no claimed base exists."); return; }
        int total = 0, open = 0;
        ArrayList<String> reports = new ArrayList<>();
        for (BaseObject obj : baseObjects) {
            if (!obj.isBusinessAsset() || !obj.businessOpen) continue;
            open++;
            int q = ItemQuality.tierIndex((obj.qualityName==null?"Common":obj.qualityName)+" business");
            int income = Math.max(1, 2 + q + obj.charges/3);
            if (obj.symbol == 's') income += Math.min(5, baseStorage.size()/3);
            if (obj.symbol == 'u' || obj.symbol == 'e') income += Math.max(0, Math.min(4, water/3));
            if (obj.symbol == 'M') income += Math.max(1, medicalStockCount());
            income += businessSellPowerBonus();
            if (!obj.permittedBusiness) { income = Math.max(1, income + 2); obj.businessHeat += 2; gangHeat += 1; suspicion += 1; }
            else suspicion = Math.max(0, suspicion - 1);
            total += income;
            applyFactionMarketPressureFromBusiness(obj, income);
            reports.add(obj.businessName() + " +" + income + (obj.permittedBusiness?" lawful":" illegal"));
        }
        if (open == 0) { logEvent("No open businesses to collect from. Use BASES > BUY PERMIT or OPEN BUSINESS first."); return; }
        addImperialScript(total);
        gainXp("Commerce", Math.max(1, open), "collecting base business income");
        logEvent("BUSINESS INCOME: +" + total + " script from " + String.join("; ", reports) + ".");
        maybeTriggerBaseRaid(baseSecurityRating());
        advanceTurn("runs base commerce.");
    }

    java.util.List<String> visibleRoomIntelLines() {
        ArrayList<String> out = new ArrayList<>();
        if (world == null) { out.add("No zone loaded."); return out; }
        out.add("Current zone type: " + world.zoneType.label + " — " + world.zoneType.descriptor);
        out.addAll(EconomicGenerationBiasAuthority.worldSummaryLines(world));
        out.addAll(EconomicLocalTopologyMetadataSurfaceAuthority.worldSummaryLines(world));
        out.addAll(EconomicTopologyReportingOverlayAuthority.worldSummaryLines(world));
        out.addAll(EconomicTopologyMapIntelBridgeAuthority.roomIntelHeaderLines(world));
        out.add("Room intel is only listed after at least one tile of that room has been seen or remembered by this character.");
        int shown = 0, hidden = 0;
        for(int i=0;i<world.roomProfiles.size();i++){
            if(!roomSeenByPlayer(i)) { hidden++; continue; }
            RoomProfile rp=world.roomProfiles.get(i); Faction owner=world.roomFaction(i);
            String topologyRole = EconomicLocalTopologyMetadataSurfaceAuthority.roomInspectionLine(world, i);
            String mapIntelRole = EconomicTopologyMapIntelBridgeAuthority.roomIntelLine(world, i);
            out.add("room " + i + ": " + (world.roomSpecials.size()>i && world.roomSpecials.get(i)?"SPECIAL ":"") + rp.name + " | owner " + owner.label + " | loot " + rp.scavengeChance + "% | " + rp.descriptor + " | features: " + rp.featureText + (topologyRole == null ? "" : " | " + topologyRole) + (mapIntelRole == null ? "" : " | " + mapIntelRole));
            shown++;
        }
        if(shown == 0) out.add("No room intel recorded yet. Explore rooms with sight or remembered fog before the atlas will describe them.");
        if(hidden > 0) out.add("Unseen rooms withheld: " + hidden + ".");
        return out;
    }

    boolean roomSeenByPlayer(int roomId) {
        if(world == null || roomId < 0 || rememberedTiles == null) return false;
        for(int x=0; x<world.w; x++) for(int y=0; y<world.h; y++) {
            if(world.roomIds[x][y] == roomId && rememberedTiles[x][y]) return true;
        }
        return false;
    }

    String ownershipText(int x, int y) {
        if (world == null) return "No world loaded.";
        int rid = world.roomIdAt(x, y);
        if (rid < 0) return "Corridor/unclaimable passage.";
        if (baseClaimed && rid == claimedRoomId) return "Friendly claimed territory, tinted light green.";
        Faction owner = world.roomFaction(rid);
        if (owner != Faction.NONE) return owner.label + " claimed room, color-coded by faction.";
        return "Neutral unclaimed room, gray/white territory.";
    }

    String hostilitySummary() {
        ArrayList<String> bits = new ArrayList<>();
        for (Faction f : Faction.visibleFactions()) {
            int v = temporaryHostileTurns.getOrDefault(f, 0);
            if (v > 0) bits.add(f.label + " " + v + "t");
        }
        return bits.isEmpty() ? "none" : String.join(", ", bits);
    }
