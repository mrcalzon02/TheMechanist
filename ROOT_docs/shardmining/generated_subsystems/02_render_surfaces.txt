    BufferedImage semanticTileImageForRenderedGlyph(char ch, int worldX, int worldY) {
        // 0.9.10fn: the renderer no longer treats a raw ASCII glyph as the complete
        // visual identity.  Stage 9 hardens this further: once a descriptor has a
        // semantic asset ID, a missing registry image produces a typed missing tile
        // instead of silently falling through to unrelated legacy art.
        CompiledTileDescriptor d = tileDescriptorForRenderedGlyph(ch, worldX, worldY);
        if (d != null) {
            if (d.hasOverlay()) {
                BufferedImage overlay = images.getSemanticAssetImage(d.overlayAssetId);
                if (overlay != null) return overlay;
                if (d.overlayAssetId != null && !d.overlayAssetId.isBlank()) return AssetManager.missingAssetImage(assetTypeForDescriptor(d));
                overlay = images.getTile(d.overlayArtKey, ch);
                if (overlay != null) return overlay;
            }
            BufferedImage primary = images.getSemanticAssetImage(d.primaryAssetId);
            if (primary != null) return primary;
            if (d.primaryAssetId != null && !d.primaryAssetId.isBlank()) return AssetManager.missingAssetImage(assetTypeForDescriptor(d));
            primary = images.getTile(d.primaryArtKey, ch);
            if (primary != null) return primary;
            if (d.primaryArtKey != null && d.primaryArtKey.matches("road_.*_v[1-5]")) {
                String base = d.primaryArtKey.replaceAll("_v[1-5]$", "");
                primary = images.getTile(base, ch);
                if (primary != null) return primary;
            }
            if (d.underlayArtKey != null) {
                BufferedImage under = images.getSemanticAssetImage(d.underlayAssetId);
                if (under != null) return under;
                if (d.underlayAssetId != null && !d.underlayAssetId.isBlank()) return AssetManager.missingAssetImage(assetTypeForDescriptor(d));
                under = images.getTile(d.underlayArtKey, d.underlayGlyph == null ? '.' : d.underlayGlyph.charValue());
                if (under != null) return under;
            }
        }
        return images.getTile(ch);
    }

    void drawTextPanel(Graphics2D g, int x, int y, int w, int h, java.util.List<String> lines, boolean centered) {
        if (w <= 4 || h <= 4) return;
        g.setColor(new Color(0, 0, 0, 224));
        g.fillRoundRect(x, y, w, h, 12, 12);
        drawSlicedFrame(g, x, y, w, h, "inner");
        stampUiFrameId(g, "T", lines == null || lines.isEmpty() ? "text-panel" : lines.get(0), x, y, w, h);
        Shape oldClip = g.getClip();
        g.setClip(x + 8, y + 8, Math.max(1, w - 16), Math.max(1, h - 16));
        g.setFont(smallFont);
        FontMetrics fm = g.getFontMetrics();
        int yy = y + 22;
        for (String line : TextLayoutAuthority.wrapAllPixels(g, g.getFont(), lines, Math.max(8, w - 32))) {
            if (yy > y + h - 8) { g.setClip(oldClip); return; }
            g.setColor(optionColor(GameOptions.TEXT_MAIN));
            if (centered) g.drawString(line, x + (w - fm.stringWidth(line)) / 2, yy); else g.drawString(line, x + 14, yy);
            yy += fm.getHeight();
        }
        g.setClip(oldClip);
    }

    void drawHoverHelp(Graphics2D g, int hoverButtonIndex) {
        if (!options.hoverHelp) return;
        String help = null;
        if (hoverButtonIndex >= 0 && hoverButtonIndex < buttons.size()) help = buttons.get(hoverButtonIndex).tip;
        else if (mouseX < 0 && selectedButton >= 0 && selectedButton < buttons.size() && buttonIsModalInteractive(buttons.get(selectedButton))) help = buttons.get(selectedButton).tip;
        if (help == null || mouseX < 0 || mouseY < 0) return;
        g.setFont(smallFont);
        int maxChars = 34;
        java.util.List<String> lines = TextSurfaceApi.wrap(help, maxChars);
        FontMetrics fm = g.getFontMetrics();
        int tw = 0;
        for (String line : lines) tw = Math.max(tw, fm.stringWidth(line));
        int boxW = Math.min(340, Math.max(190, tw + 22));
        int lineH = fm.getHeight();
        int boxH = Math.max(34, lines.size()*lineH + 14);
        int x = mouseX + 18;
        int y = mouseY + 18;
        if (x + boxW > getWidth() - 12) x = mouseX - boxW - 18;
        if (y + boxH > getHeight() - 12) y = mouseY - boxH - 18;
        x = Math.max(8, x); y = Math.max(8, y);
        g.setColor(new Color(8,8,9,238)); g.fillRect(x, y, boxW, boxH);
        g.setColor(new Color(165,138,82)); g.drawRect(x, y, boxW, boxH);
        g.setColor(new Color(220,205,160));
        int ty = y + 18;
        for (String line : lines) { drawUiTextLine(g, line, x+11, ty); ty += lineH; }
    }

    void drawUiTextLine(Graphics2D g, String s, int x, int y) {
        if (s == null) return;
        FontMetrics fm = g.getFontMetrics();
        int tw = fm.stringWidth(s);
        int th = fm.getHeight();
        Color old = g.getColor();
        g.setColor(new Color(0, 0, 0, 196));
        g.fillRoundRect(x - 5, y - fm.getAscent() - 3, tw + 10, th + 5, 7, 7);
        g.setColor(new Color(128, 105, 58, 135));
        g.drawRoundRect(x - 5, y - fm.getAscent() - 3, tw + 10, th + 5, 7, 7);
        g.setColor(old);
        g.drawString(s, x, y);
    }
